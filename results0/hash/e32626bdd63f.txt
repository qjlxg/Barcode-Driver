<script>
    let zg = null;
    let localStream = null;

    function log(msg) {
        const now = new Date().toLocaleTimeString();
        const logDiv = document.getElementById('log');
        logDiv.innerHTML += `${now} - ${msg}<br>`;
        logDiv.scrollTop = logDiv.scrollHeight;
        console.log(msg);
    }

    async function joinRoom() {
        const appId = parseInt(document.getElementById('appId').value);
        const server = document.getElementById('server').value;
        const userId = document.getElementById('userId').value;
        const roomId = document.getElementById('roomId').value;
        const token = document.getElementById('token').value;

        if (!appId) {
            log('❌ 请填写 AppID');
            return;
        }

        try {
            log(`初始化 ZegoExpressEngine, AppID=${appId}`);
            zg = new ZegoExpressEngine(appId, server);

            // 监听房间状态
            zg.on('roomStateChanged', (roomID, reason, errorCode, extendedData) => {
                log(`房间状态: ${reason}, errorCode=${errorCode}`);
            });

            // 监听远端流
            zg.on('roomStreamUpdate', async (roomID, updateType, streamList) => {
                log(`流更新: ${updateType}, 数量=${streamList.length}`);
                if (updateType === 'ADD') {
                    for (const stream of streamList) {
                        log(`拉取远端流: ${stream.streamID}`);
                        try {
                            const remoteStream = await zg.startPlayingStream(stream.streamID);
                            const audio = document.createElement('audio');
                            audio.srcObject = remoteStream;
                            audio.autoplay = true;
                            audio.id = 'remote_' + stream.streamID;
                            document.getElementById('remoteAudio').appendChild(audio);
                            log(`✅ 远端流播放成功`);
                        } catch (e) {
                            log(`❌ 拉流失败: ${e.message}`);
                        }
                    }
                }
            });

            // 登录房间
            log(`加入房间: ${roomId}`);

            // 如果没有 token，使用临时 token（仅测试环境）
            const loginToken = token || '';

            const result = await zg.loginRoom(roomId, loginToken, {
                userID: userId,
                userName: 'User_' + userId
            });

            log(`✅ 加入房间成功: ${JSON.stringify(result)}`);

        } catch (error) {
            log(`❌ 错误: ${error.message}`);
            console.error(error);
        }
    }

    async function startAudio() {
        if (!zg) {
            log('❌ 请先加入房间');
            return;
        }

        const userId = document.getElementById('userId').value;

        try {
            log('创建本地音频流...');
            localStream = await zg.createStream({
                camera: { video: false, audio: true }
            });

            const streamID = 'stream_' + userId;
            log(`推流: ${streamID}`);
            await zg.startPublishingStream(streamID, localStream);
            log('✅ 推流成功，正在说话...');

        } catch (error) {
            log(`❌ 推流失败: ${error.message}`);
        }
    }

    function stopAudio() {
        if (!zg || !localStream) {
            log('没有正在推送的流');
            return;
        }

        const userId = document.getElementById('userId').value;
        const streamID = 'stream_' + userId;

        try {
            zg.stopPublishingStream(streamID);
            zg.destroyStream(localStream);
            localStream = null;
            log('✅ 停止推流');
        } catch (error) {
            log(`❌ 停止失败: ${error.message}`);
        }
    }

    async function leaveRoom() {
        if (!zg) {
            log('未加入房间');
            return;
        }

        try {
            stopAudio();
            await zg.logoutRoom();
            zg = null;
            log('✅ 已离开房间');
        } catch (error) {
            log(`❌ 离开失败: ${error.message}`);
        }
    }
</script>
