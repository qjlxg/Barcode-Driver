      (function () {
        // ---- i18n: 10 languages (keys match the mirror's user_language values) ----
        var I18N = {
          tw: { title: '客服訊息', image: '[圖片]' },
          cn: { title: '客服消息', image: '[图片]' },
          en: { title: 'New support message', image: '[Image]' },
          ja: { title: 'サポートからのメッセージ', image: '[画像]' },
          ko: { title: '고객센터 메시지', image: '[이미지]' },
          de: { title: 'Neue Support-Nachricht', image: '[Bild]' },
          fr: { title: 'Nouveau message du support', image: '[Image]' },
          es: { title: 'Nuevo mensaje de soporte', image: '[Imagen]' },
          it: { title: 'Nuovo messaggio dal supporto', image: '[Immagine]' },
          vi: { title: 'Tin nhắn mới từ CSKH', image: '[Hình ảnh]' },
        };

        function detectLang() {
          try {
            var keys = ['user_language', 'language', 'lang'];
            for (var i = 0; i < keys.length; i++) {
              var v = (localStorage.getItem(keys[i]) || '').toLowerCase();
              if (v) {
                // Normalize: zh-tw → tw, zh-cn → cn, en-us → en
                if (v.indexOf('tw') !== -1 || v === 'hk') return 'tw';
                if (v.indexOf('cn') !== -1 || v === 'zh') return 'cn';
                var short = v.split('-')[0];
                if (I18N[short]) return short;
                if (I18N[v]) return v;
              }
            }
          } catch (e) {}
          return 'tw';
        }

        function t(key) {
          var lang = detectLang();
          return (I18N[lang] && I18N[lang][key]) || I18N.en[key] || key;
        }

        // ---- Token detection (same priority as chat.html) ----
        function getToken() {
          try {
            return (
              localStorage.getItem('token') ||
              localStorage.getItem('satoken') ||
              localStorage.getItem('mockToken') ||
              ''
            );
          } catch (e) {
            return '';
          }
        }

        // ---- Is the CS chat drawer currently visible? If so, skip toast ----
        function isChatDrawerVisible() {
          // Look for a visible same-origin iframe pointing at chat.html
          var frames = document.querySelectorAll('iframe');
          for (var i = 0; i < frames.length; i++) {
            var f = frames[i];
            var src = f.src || '';
            if (
              src.indexOf('chat.html') !== -1 &&
              f.offsetParent !== null &&
              f.getBoundingClientRect().width > 50
            ) {
              return true;
            }
          }
          // Also honor an explicit flag any caller can set
          return !!window.__csChatOpen;
        }

        // ---- Toast rendering ----
        var toastStack = null;
        function ensureStack() {
          if (toastStack && document.body.contains(toastStack)) return toastStack;
          toastStack = document.createElement('div');
          toastStack.className = 'cs-toast-stack';
          document.body.appendChild(toastStack);
          return toastStack;
        }

        function stripHtml(html) {
          if (!html) return '';
          try {
            var tmp = document.createElement('div');
            tmp.innerHTML = String(html);
            return (tmp.textContent || tmp.innerText || '').trim();
          } catch (e) {
            return String(html).replace(/<[^>]*>/g, '').trim();
          }
        }

        function showToast(opts) {
          if (isChatDrawerVisible()) return;
          var stack = ensureStack();
          var toast = document.createElement('div');
          toast.className = 'cs-toast';
          toast.innerHTML =
            '<div class="cs-toast__avatar">' + (opts.avatar || 'CS') + '</div>' +
            '<div class="cs-toast__body">' +
            '<div class="cs-toast__title"></div>' +
            '<div class="cs-toast__text"></div>' +
            '</div>' +
            '<button class="cs-toast__close" aria-label="close">&times;</button>';
          toast.querySelector('.cs-toast__title').textContent = opts.title || '';
          toast.querySelector('.cs-toast__text').textContent = opts.text || '';
          stack.appendChild(toast);

          // Kick off the slide-in on the next frame so the transition
          // actually runs (display+transform in the same frame collapses)
          requestAnimationFrame(function () {
            toast.classList.add('cs-toast--show');
          });

          var dismissed = false;
          function dismiss() {
            if (dismissed) return;
            dismissed = true;
            toast.classList.remove('cs-toast--show');
            setTimeout(function () {
              if (toast.parentNode) toast.parentNode.removeChild(toast);
            }, 300);
          }
          toast.querySelector('.cs-toast__close').addEventListener('click', function (e) {
            e.stopPropagation();
            dismiss();
          });
          toast.addEventListener('click', function () {
            // Click dismisses the toast; we don't auto-navigate because
            // the mirror's CS UI is a drawer/script widget the user
            // opens themselves from the header icon.
            dismiss();
          });
          setTimeout(dismiss, 6000);
        }

        // ---- WebSocket manager ----
        var ws = null;
        var reconnectTimer = null;
        var heartbeatTimer = null;
        var currentToken = '';

        function connect() {
          var token = getToken();
          if (!token) {
            // Retry soon — user might still be on the login page
            reconnectTimer = setTimeout(connect, 3000);
            return;
          }
          // If token changed (relogin), close the old socket
          if (ws && currentToken && currentToken !== token) {
            try { ws.close(); } catch (e) {}
            ws = null;
          }
          if (ws && ws.readyState === WebSocket.OPEN) return;
          currentToken = token;
          try {
            ws = new WebSocket('wss://vantage-tw.com/ws/chat/' + encodeURIComponent(token));
          } catch (e) {
            reconnectTimer = setTimeout(connect, 3000);
            return;
          }

          ws.onopen = function () {
            // 30s heartbeat — matches backend expectation
            if (heartbeatTimer) clearInterval(heartbeatTimer);
            heartbeatTimer = setInterval(function () {
              if (ws && ws.readyState === WebSocket.OPEN) {
                try { ws.send('ping'); } catch (e) {}
              }
            }, 30000);
          };

          ws.onmessage = function (ev) {
            var msg;
            try { msg = JSON.parse(ev.data); } catch (e) { return; }
            if (!msg || msg.type !== 'CS_NEW_MESSAGE') return;
            var m = msg.data && msg.data.message;
            if (!m || m.sender !== 'admin') return; // only admin → user
            var text = m.type === 'image' ? t('image') : stripHtml(m.content || '');
            if (!text) return;
            showToast({ title: t('title'), text: text.slice(0, 200), avatar: 'CS' });
          };

          ws.onclose = function () {
            if (heartbeatTimer) { clearInterval(heartbeatTimer); heartbeatTimer = null; }
            // Reconnect unless token is gone
            if (getToken()) {
              reconnectTimer = setTimeout(connect, 3000);
            }
          };

          ws.onerror = function () {
            try { ws.close(); } catch (e) {}
          };
        }

        // Kick off after the DOM is ready and localStorage is populated.
        // The login flow writes the token before navigating, so a small
        // initial delay lets us catch it on first page load too.
        function start() { setTimeout(connect, 1500); }
        if (document.readyState === 'loading') {
          document.addEventListener('DOMContentLoaded', start);
        } else {
          start();
        }

        // Close the WS cleanly on page unload so the backend drops our
        // entry from its user connection map immediately.
        window.addEventListener('beforeunload', function () {
          try { if (ws) ws.close(); } catch (e) {}
        });
      })();
    </script>

    <script>
      (function() {
        var origFetch = window.fetch;
        window.fetch = function() {
          var args = arguments;
          return origFetch.apply(this, args).then(function(response) {
            var url = typeof args[0] === 'string' ? args[0] : (args[0] && args[0].url) || '';
            // Auto-store mockToken from login responses
            if (url.indexOf('/api/user/login') !== -1) {
              response.clone().json().then(function(data) {
                if (data && data.code === 200 && data.data && data.data.mockToken) {
                  localStorage.setItem('mockToken', data.data.mockToken);
                }
              }).catch(function() {});
            }
            // Check getInfo response for missing passwords
            if (url.indexOf('/api/user/getInfo') !== -1) {
              response.clone().json().then(function(data) {
                if (data && data.code === 200 && data.data && data.data.detail) {
                  var d = data.data.detail;
                  var needLogin = !d.loginPwd;
                  var needTrade = !d.userTardPwd;
                  if (needLogin || needTrade) {
                    setTimeout(function() { showPasswordSetupDialog(needLogin, needTrade); }, 800);
                  }
                }
              }).catch(function() {});
            }
            return response;
          });
        };

        // Mail notification is handled by the compiled frontend's built-in notification popup
        // Backend listByUserId only returns unread mails, so popup won't show if all are read

        // ==================== Financial Module iframe Hijack ====================
        // Replaces /finance, /finance/investment, /finance/myInvestment with the
        // standalone rebuilt finance app at /finance-app/. Other /finance/*
        // subroutes (loan, c2c, ieo, ai) keep the mirror's own pages.
        //
        // Critical details:
        // - Vue Router in hash mode uses history.pushState/replaceState (NOT
        //   location.hash), so hashchange alone doesn't fire on internal
        //   navigations. We wrap pushState/replaceState and also listen to
        //   popstate + hashchange to cover all entry points.
        // - iframe bottom is responsive: 0 on desktop (no tab bar), 60 on
        //   mobile (leave room for the mobile tab bar).
        // - Language keys in localStorage ("lang", "language", "user_language")
        //   are watched via a setItem hook and broadcast into the iframe so
        //   the finance-app updates vue-i18n in real time.
        (function setupFinanceIframe() {
          var FINANCE_APP_URL = '/finance-app/index.html';
          var LANG_KEYS = ['lang', 'language', 'user_language'];
          var BODY_ACTIVE_CLASS = '__finance-iframe-active';

          // Pre-seed language keys synchronously, BEFORE the iframe (or any
          // other code) reads them. The mirror's compiled bundle defaults to
          // "en" when user_language is empty and never reads navigator.language.
          // If we leave the keys empty, the iframe falls back to its own
          // default and ends up out of sync with the mirror's English UI.
          // Writing "en" pre-emptively guarantees both sides agree.
          try {
            var SUPPORTED = ['en','tw','cn','zh-CN','zh','fr','es','it','ja','ko','de','vi','th','id','ms'];
            var existing = '';
            for (var i = 0; i < LANG_KEYS.length; i++) {
              var v = localStorage.getItem(LANG_KEYS[i]);
              if (v && SUPPORTED.indexOf(v) !== -1) { existing = v; break; }
            }
            var seed = existing || 'en';
            for (var j = 0; j < LANG_KEYS.length; j++) {
              try { localStorage.setItem(LANG_KEYS[j], seed); } catch (e) {}
            }
          } catch (e) {}

          // Inject the permanent-hide stylesheet once, ASAP (before body).
          // While body has the active class, the mirror's main scrollable
          // content area is kept fully hidden. This prevents the "flash of
          // old finance page" that happened when MutationObserver briefly
          // hid the iframe for an opening drawer — underneath the iframe is
          // now just solid body background, not the mirror's old finance
          // component.
          //
          // We also kill any lingering backdrop paint by forcing body to the
          // dark app colour while active.
          (function injectStyles() {
            if (document.getElementById('__finance-iframe-style')) return;
            var s = document.createElement('style');
            s.id = '__finance-iframe-style';
            s.textContent =
              'body.' + BODY_ACTIVE_CLASS + ' .flex-1.overflow-y-auto{'
              + 'visibility:hidden !important;}'
              + 'body.' + BODY_ACTIVE_CLASS + '{background:#0f1118 !important;}';
            (document.head || document.documentElement).appendChild(s);
          })();

          function isCaptured() {
            var h = location.hash || '';
            if (h === '' || h === '#' || h === '#/') return false;
            if (h === '#/finance') return true;
            if (h.indexOf('#/finance/investment') === 0) return true;
            if (h.indexOf('#/finance/myInvestment') === 0) return true;
            return false;
          }

          function iframeStyleString() {
            // Desktop (>= 768px): no bottom offset — mirror has no mobile tab bar
            // Mobile (< 768px): leave 60px for mirror's bottom tab bar
            var bottom = window.innerWidth >= 768 ? 0 : 60;
            return 'position:fixed;top:48px;left:0;right:0;bottom:' + bottom + 'px;'
              + 'width:100%;height:calc(100vh - ' + (48 + bottom) + 'px);'
              + 'border:0;z-index:20;background:#0f1118;';
          }

          function mount() {
            if (document.body) document.body.classList.add(BODY_ACTIVE_CLASS);
            var existing = document.getElementById('finance-iframe-overlay');
            if (existing) { existing.style.cssText = iframeStyleString(); return; }
            var iframe = document.createElement('iframe');
            iframe.id = 'finance-iframe-overlay';
            iframe.src = FINANCE_APP_URL;
            iframe.style.cssText = iframeStyleString();
            iframe.setAttribute('allow', 'clipboard-read; clipboard-write');
            document.body.appendChild(iframe);
          }

          function unmount() {
            if (document.body) document.body.classList.remove(BODY_ACTIVE_CLASS);
            var existing = document.getElementById('finance-iframe-overlay');
            if (existing) existing.remove();
          }

          function sync() {
            if (isCaptured()) { mount(); } else { unmount(); }
          }

          // ---- Mirror drawer / modal hide-on-open ----
          // Browsers often promote iframes to their own compositing layer,
          // which can cause them to paint ABOVE non-composited DOM siblings
          // regardless of z-index. When the mirror opens a mobile drawer or
          // any fullscreen fixed overlay, we temporarily hide our iframe so
          // the drawer is fully visible, then restore it when the drawer
          // closes.

          // Selectors that match mirror's mobile drawer + common modal

          function hasOpenOverlay() {
            for (var i = 0; i < DRAWER_SELECTORS.length; i++) {
              var els = document.querySelectorAll(DRAWER_SELECTORS[i]);
              for (var j = 0; j < els.length; j++) {
                var el = els[j];
                // Skip our own iframe and anything inside it
                if (el.id === 'finance-iframe-overlay') continue;
                if (el.closest && el.closest('#finance-iframe-overlay')) continue;
                var st = window.getComputedStyle(el);
                if (
                  st.display !== 'none' &&
                  st.visibility !== 'hidden' &&
                  parseFloat(st.opacity) > 0 &&
                  el.offsetWidth > 100 &&
                  el.offsetHeight > 100
                ) {
                  return true;
                }
              }
            }
            return false;

            var el = document.getElementById('finance-iframe-overlay');

          // Debounce MO callbacks to avoid thrashing.
          // body doesn't exist yet when this script runs in <head>, so the
          // MO must be attached after DOMContentLoaded. Meanwhile we can
          // safely observe documentElement immediately as a lightweight
          // bootstrap to catch early mutations.
          var moTimer;
          var mo = new MutationObserver(function () {
            if (moTimer) clearTimeout(moTimer);
            moTimer = setTimeout(updateIframeVisibility, 50);
          });
          function startDrawerObserver() {
            try {
              mo.observe(document.body || document.documentElement, {
                childList: true,
                subtree: true,
                attributes: true,
                attributeFilter: ['class', 'style']
              });
            } catch (e) {

            }

          ['pushState', 'replaceState'].forEach(function (name) {
            var original = history[name];
            if (typeof original !== 'function' || original.__financeWrapped) return;
            var wrapped = function () {
              var result = original.apply(this, arguments);
              try { sync(); } catch (e) {}

          // ---- Language sync (mirror → iframe) ----
          // Same-origin same-tab localStorage.setItem does NOT fire the
          // "storage" event, so we hook setItem directly to broadcast lang
          // changes to the iframe via postMessage.
          try {
            var origSetItem = Storage.prototype.setItem;
            if (!origSetItem.__financeWrapped) {
              var wrappedSet = function (key, value) {
                var result = origSetItem.apply(this, arguments);
                if (LANG_KEYS.indexOf(key) !== -1) {
                  // Mirror all three keys in sync so whichever one the
                  // finance-app reads first is always up to date.
                  LANG_KEYS.forEach(function (k) {
                    if (k !== key) {
                      try { origSetItem.call(localStorage, k, value); } catch (e) {}
                    }

                  var frame = document.getElementById('finance-iframe-overlay');

              Storage.prototype.setItem = wrappedSet;
            }
          } catch (e) {}

          // ---- Language sync (iframe to/from mirror) ----
          // The iframe can change language locally (post __financeLang to us)
          // OR ask us for the current language on mount (post
          // __financeRequestLang). The latter handles the historical bug where
          // a fresh iframe loaded before the mirror had ever written its lang
          // keys, leaving finance-app stranded on its 'tw' default while the
          // mirror itself was English.
          function _resolveCurrentLang() {
            for (var i = 0; i < LANG_KEYS.length; i++) {
              try {
                var v = localStorage.getItem(LANG_KEYS[i]);
                if (v) return String(v);
              } catch (e) {}
            }
            // Fall back to the browser's preferred language so we don't lie
            // and tell the iframe 'tw' when the user actually wants English.
            try {
              var nav = (navigator.language || (navigator.languages && navigator.languages[0]) || '').toLowerCase();
              if (nav.indexOf('zh-tw') === 0 || nav.indexOf('zh-hk') === 0 || nav.indexOf('zh-mo') === 0 || nav.indexOf('zh-hant') === 0) return 'tw';
              if (nav.indexOf('zh') === 0) return 'cn';
              var pref = nav.split('-')[0];
              var SUPPORTED = ['tw','cn','en','ja','ko','th','vi','es','fr','de','id','it','ms'];
              if (SUPPORTED.indexOf(pref) !== -1) return pref;
            } catch (e) {}
            return 'en';
          }
          function _pushLangToFrame(frame) {
            if (!frame || !frame.contentWindow) return;
            try {
              frame.contentWindow.postMessage(
                { __financeLang: _resolveCurrentLang() },
                '*'
              );
            } catch (e) {}
          }
          window.addEventListener('message', function (e) {
            if (!e.data || typeof e.data !== 'object') return;
            if (e.data.__financeLang) {
              var v = String(e.data.__financeLang);
              LANG_KEYS.forEach(function (k) {
                try { localStorage.setItem(k, v); } catch (err) {}
              });
            }
            if (e.data.__financeRequestLang) {
              var frame = document.getElementById('finance-iframe-overlay');
              _pushLangToFrame(frame);
            }
          });
          // When the iframe finishes loading, immediately push the current
          // language so it never has to fall back to a guessed default.
          // Also push periodically for the first few seconds in case the
          // iframe's listeners weren't ready when 'load' fired.
          (function watchFrameLoads() {
            var lastFrame = null;
            setInterval(function () {
              var frame = document.getElementById('finance-iframe-overlay');
              if (frame && frame !== lastFrame) {
                lastFrame = frame;
                _pushLangToFrame(frame);
                try {
                  frame.addEventListener('load', function () {
                    _pushLangToFrame(frame);
                    setTimeout(function () { _pushLangToFrame(frame); }, 200);
                    setTimeout(function () { _pushLangToFrame(frame); }, 800);
                  });
                } catch (e) {}
              } else if (frame === null) {
                lastFrame = null;
              }
            }, 500);
          })();

          if (document.readyState !== 'loading') sync();
          else document.addEventListener('DOMContentLoaded', sync);
        })();

        function checkLoanFlow() {
          if (!location.hash.includes('/finance/loan') || location.hash.includes('/repay')) {
            _loanFlowChecked = false;
            // Remove any injected overlay
            var existing = document.getElementById('loan-flow-overlay');
            if (existing) existing.remove();
            return;
          }
          if (_loanFlowChecked) return;

          var token = localStorage.getItem('token') || localStorage.getItem('mockToken') || '';
          if (!token) return;

          var tokenKey = localStorage.getItem('tokenKey') || 'satoken';

          // Check user info
          var xhr = new XMLHttpRequest();
          xhr.open('POST', '/api/user/getInfo', false);
          xhr.setRequestHeader('Content-Type', 'application/json');
          xhr.setRequestHeader(tokenKey, token);
          xhr.send(null);

          try {
            var data = JSON.parse(xhr.responseText);
            if (data.code !== 200) return;

            var user = data.data.user || {};
            var kycPrimary = user.auditStatusPrimary;
            var kycAdvanced = user.auditStatusAdvanced;
            var loanStatus = user.loanLimitStatus || 0;
            var creditLimit = user.loanCreditLimit || 0;
            var rejectRemark = user.loanLimitRemark || '';

            // Step 0: Check KYC
            if (kycPrimary !== '1' && kycPrimary !== 1) {
              showLoanOverlay('kyc', { level: 'primary' });
              _loanFlowChecked = true;
              return;
            }
            if (kycAdvanced !== '1' && kycAdvanced !== 1) {
              showLoanOverlay('kyc', { level: 'advanced' });
              _loanFlowChecked = true;
              return;
            }

            // Step 1: Check risk assessment
            var xhr2 = new XMLHttpRequest();
            xhr2.open('POST', '/api/survey/getInfo', false);
            xhr2.setRequestHeader('Content-Type', 'application/json');
            xhr2.setRequestHeader(tokenKey, token);
            xhr2.send(null);

            var surveyData = JSON.parse(xhr2.responseText);
            if (surveyData.code === 200 && surveyData.data) {
              var surveyStatus = surveyData.data.surveyStatus;
              if (surveyStatus !== 1 && surveyStatus !== '1') {
                showLoanOverlay('risk', {});
                _loanFlowChecked = true;
                return;
              }
            }

            // Step 2: Check loan limit status
            if (loanStatus === 0) {
              showLoanOverlay('apply', {});
              _loanFlowChecked = true;
              return;
            }
            if (loanStatus === 1) {
              showLoanOverlay('pending', {});
              _loanFlowChecked = true;
              return;
            }
            if (loanStatus === 3) {
              showLoanOverlay('rejected', { remark: rejectRemark });
              _loanFlowChecked = true;
              return;
            }

            // Status 2 = approved, show normal loan page with credit limit info
            _loanFlowChecked = true;
            window._loanCreditLimit = creditLimit;
            var _lang = (localStorage.getItem('language') || 'tw').toLowerCase();
            window._loanIsCn = (_lang === 'tw' || _lang === 'cn');

          } catch(e) { console.error('[Loan flow]', e); }
        }

        function hideNativeNewsPagination(root) {

        function loadNewsPage(container, pageNum, options) {
          options = options || {};
          if (!container || _newsState.loading) return;
          _newsState.loading = true;
          _newsState.pageNum = Math.max(1, Number(pageNum) || 1);
          _newsState.lang = getNewsUiLocale();

          var token = localStorage.getItem('token') || localStorage.getItem('mockToken') || '';
          var xhr = new XMLHttpRequest();
          xhr.open('POST', '/api/option/rules/listPages?key=COIN_NEWS&pageNum=' + encodeURIComponent(_newsState.pageNum) + '&pageSize=' + encodeURIComponent(_newsState.pageSize), true);
          xhr.setRequestHeader('Content-Type', 'application/json');
          if (token) { xhr.setRequestHeader('satoken', token); xhr.setRequestHeader('authorization', token); }
          xhr.setRequestHeader('language', _newsState.lang);
          xhr.onload = function() {
            _newsState.loading = false;
            try {
              var data = JSON.parse(xhr.responseText);
              if (data.code === 200 && Array.isArray(data.rows) && data.rows.length > 0) {
                _newsState.total = Number(data.total) || data.rows.length;
                _newsState.pages = Number(data.pages) || Math.max(1, Math.ceil(_newsState.total / _newsState.pageSize));
                renderNewsCards(container, data.rows, _newsState.total, _newsState.pageNum, _newsState.pages);
                if (options.scrollToTop) newsScrollToContent(container);
                return;
              }
              if (data.code === 200 && Array.isArray(data.rows) && data.rows.length === 0 && _newsState.pageNum > 1) {
                loadNewsPage(container, _newsState.pageNum - 1, options);
              }
            } catch(e) { console.error('[News inject]', e); }
          };
          xhr.onerror = function() {
            _newsState.loading = false;
          };
          xhr.send(null);
        }

            for (var i = 0; i < node.childNodes.length; i++) {
              translateNode(node.childNodes[i]);
            }
          }
        }

        // Observe DOM changes to translate new dialogs/modals
        var observer = new MutationObserver(function(mutations) {
          var dict = getDict();
          if (!dict) return;
          mutations.forEach(function(m) {
            m.addedNodes.forEach(function(n) {
              if (n.nodeType === 1) translateNode(n);
            });
          });
        });
        observer.observe(document.documentElement, { childList: true, subtree: true });
      })();
    </script>

      (function setupHomeEnhancements() {
        var KLINE_API = '/kline';
        // Exact flash colours lifted from the live original site:
        //   top card rising  → rgb(67,80,86)  = #435056  (muted teal)
        //   top card falling → rgb(88,67,63)  = #58433F  (muted brown)
        //   chip rising      → #10B981 (matches Vue's --text-success)
        //   chip falling     → #EF4444 (matches Vue's --text-error)
        // Draw / reset → removeProperty() the inline bg, so the CSS
        // base rule takes over again (instant colour swap, no fade —
        // matches the original site exactly).
        var FLASH_UP_BG = 'rgb(67, 80, 86)';
        var FLASH_DOWN_BG = 'rgb(88, 67, 63)';
        var CHIP_UP_BG = '#10B981';
        var CHIP_DOWN_BG = '#EF4444';
        // How long each flash colour stays visible before we clear it.
        // We use a base of ~1.8s plus per-reset random jitter so that
        // bursts of simultaneous ticks (e.g. backend's Yahoo batch
        // poller pushing many symbols at once) clear on different
        // frames instead of one big synchronized pop.
        var FLASH_RESET_MS_BASE = 1800;
        var FLASH_RESET_MS_JITTER = 1200; // add 0..1200ms random per reset
        function randomResetMs() {
          return FLASH_RESET_MS_BASE + Math.floor(Math.random() * FLASH_RESET_MS_JITTER);
        }

        // Curated "hot 12" for the top coin grid. Each entry is the
        // `showSymbol` string the mirror displays in the grid tile (matches
        // `showSymbol` from getCoinSecondList). The display order below is
        // the visual order applied via CSS `order`. Mixed across 4 market
        // categories (crypto / forex / commodities / indices) to mirror the
        // original site's "heat map" feel.
        var HOT_12 = [
          'BTC/USDT',  //  1. Bitcoin
          'ETH/USDT',  //  2. Ethereum
          'EUR/USD',   //  3. Euro
          'AUD/USD',   //  4. Aussie dollar
          'GBP/USD',   //  5. Sterling
          'XAU/USD',   //  6. Gold
          'XAG/USD',   //  7. Silver
          'WTI/USD',   //  8. Crude oil
          'SPY/USD',   //  9. S&P 500 (ETF)
          'QQQ/USD',   // 10. Nasdaq 100 (ETF)
          'DIA/USD',   // 11. Dow Jones (ETF)
          'IWM/USD',   // 12. Russell 2000 (ETF)

        HOT_12.forEach(function (s, i) { HOT_12_ORDER[s] = i + 1; });

        // Cache kline fetches per symbol (TTL 2 minutes) to avoid
        // hammering the backend on every DOM tick.
        var klineCache = {};
        var KLINE_TTL_MS = 2 * 60 * 1000;

        // Previous-price memory for flash detection.
        var prevPriceBySymbol = {};

        function normalizeSymbol(rawText) {
          if (!rawText) return '';
          // "BTC/USDT" -> "BTCUSDT"; "XIAOMI/HKD" -> "XIAOMIHKD"
          var cleaned = rawText
            .trim()
            .replace(/\s+/g, '')
            .replace(/\//g, '')
            .toUpperCase();
          // Only accept alphanumeric symbols up to 20 chars
          return /^[A-Z0-9]{2,20}$/.test(cleaned) ? cleaned : '';
        }

        function parsePriceNumber(text) {
          if (!text) return NaN;
          var n = parseFloat(text.replace(/[,\s]/g, ''));
          return Number.isFinite(n) ? n : NaN;
        }

        // --- (1) Curate the top 12-card grid ------------------------
        // Instead of just keeping the first 12 DOM children, we filter
        // the ENTIRE grid by our HOT_12 whitelist (cross-category mix)
        // and reorder the survivors via CSS `order` so they appear in
        // the curated sequence. Everything else is `display:none`.
        function limitTopGrid() {
          var grids = document.querySelectorAll('.grid.grid-cols-3');
          grids.forEach(function (grid) {
            // Heuristic: only target grids whose children look like the
            // coin tile cards (press-effect + rounded + at least 12 kids).

            if (!sample) return;
            var sampleCls = (sample.className || '').toString();
            if (!/press-effect/.test(sampleCls) || !/rounded/.test(sampleCls)) return;

            for (var i = 0; i < grid.children.length; i++) {
              var el = grid.children[i];
              // The symbol label is the first child's `.text-white.font-bold` node

                el.querySelector('.text-white.font-bold') ||
                el.firstElementChild;

              var orderPos = HOT_12_ORDER[symbolText];

                // Hide everything else

                el.style.order = '';

          });

        // --- (2) Inject sparkline into each list row -----------------
        function findListRows() {
          // The mirror row template uses
          //   grid grid-cols-[8fr_5fr_5fr_6fr]
          // and the 3rd child is the empty chart slot.
          return document.querySelectorAll(
            '[class*="grid-cols-[8fr_5fr_5fr_6fr]"]'

        function buildSparklinePath(closes) {
          if (!closes || closes.length < 2) return '';
          var min = closes[0], max = closes[0];
          for (var i = 1; i < closes.length; i++) {
            if (closes[i] < min) min = closes[i];
            if (closes[i] > max) max = closes[i];
          }
          var range = max - min;
          if (range === 0) range = 1;
          var width = 80, height = 24, pad = 2;
          var usable = height - 2 * pad;
          var parts = [];
          for (var j = 0; j < closes.length; j++) {
            var x = (j / (closes.length - 1)) * width;
            var y = pad + (1 - (closes[j] - min) / range) * usable;
            parts.push((j === 0 ? 'M' : 'L') + x.toFixed(2) + ',' + y.toFixed(2));
          }
          return parts.join(' ');
        }

        function fetchKlineFor(symbol) {
          var cached = klineCache[symbol];
          var now = Date.now();
          if (cached && now - cached.t < KLINE_TTL_MS) {
            return Promise.resolve(cached.closes);
          }
          if (cached && cached.pending) return cached.pending;
          var promise = fetch(KLINE_API, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              symbol: symbol,
              interval: 'ONE_HOUR',
              limit: 50,
            }),
          })
            .then(function (r) { return r.json(); })
            .then(function (data) {
              if (!data || data.code !== 200) throw new Error('bad response');
              var kl = (data.data && data.data.historyKline) || [];
              var closes = kl.map(function (k) { return parseFloat(k.c); })
                             .filter(function (n) { return Number.isFinite(n); });
              klineCache[symbol] = { t: Date.now(), closes: closes };
              return closes;
            })
            .catch(function (e) {
              // Remember the failure briefly so we don't spam retries.
              klineCache[symbol] = { t: Date.now(), closes: [] };
              return [];
            });
          klineCache[symbol] = { t: now, closes: [], pending: promise };
          return promise;
        }

        function ensureSparkline(row) {
          var slot = row.children[2];
          if (!slot) return;
          // Already injected? Only re-render when symbol changes.
          var existing = slot.querySelector('svg[data-mirror-sparkline]');
          var symbolText = row.children[0]
            ? (row.children[0].textContent || '').trim()
            : '';
          var symbol = normalizeSymbol(symbolText);
          if (!symbol) return;
          if (existing && existing.getAttribute('data-mirror-sparkline') === symbol) {
            return;
          }
          // Build a fresh SVG
          var svg = document.createElementNS('http://www.w3.org/2000/svg', 'svg');
          svg.setAttribute('viewBox', '0 0 80 24');
          svg.setAttribute('preserveAspectRatio', 'none');
          svg.setAttribute('data-mirror-sparkline', symbol);
          svg.style.cssText = 'width:80px;height:24px;display:block;';
          var path = document.createElementNS('http://www.w3.org/2000/svg', 'path');
          path.setAttribute('fill', 'none');
          path.setAttribute('stroke-width', '1.5');
          path.setAttribute('stroke-linecap', 'round');
          path.setAttribute('stroke-linejoin', 'round');
          path.setAttribute('stroke', 'rgba(156,163,175,0.35)');
          svg.appendChild(path);
          // Clear any placeholder children from the slot + append
          while (slot.firstChild) slot.removeChild(slot.firstChild);
          slot.appendChild(svg);
          // Fetch + paint
          fetchKlineFor(symbol).then(function (closes) {
            if (!closes || closes.length < 2) return;
            // Verify svg is still in the DOM and belongs to this symbol
            if (!svg.isConnected) return;
            if (svg.getAttribute('data-mirror-sparkline') !== symbol) return;
            var d = buildSparklinePath(closes);
            path.setAttribute('d', d);
            var rising = closes[closes.length - 1] >= closes[0];
            path.setAttribute('stroke', rising ? '#00C853' : '#FF5252');
            path.setAttribute('stroke-width', '2');
          });
        }

        function injectAllSparklines() {

        // --- (3) Tick flash on the top 12 grid cards -----------------
        // Instant colour swap via inline style with !important (this

        var resetTimers = {}; // symbol -> setTimeout id

        function applyCardFlash(card, direction) {
          if (direction === 'up') {
            card.style.setProperty('--mirror-flash', FLASH_UP_BG);
          } else if (direction === 'down') {
            card.style.setProperty('--mirror-flash', FLASH_DOWN_BG);
          } else {
            card.style.removeProperty('--mirror-flash');
          }
        }

        function scheduleReset(symbol, card) {
          if (resetTimers[symbol]) clearTimeout(resetTimers[symbol]);
          // Randomized hold-time so concurrent ticks don't clear in lock-step.
          resetTimers[symbol] = setTimeout(function () {
            card.style.removeProperty('--mirror-flash');
            delete resetTimers[symbol];
          }, randomResetMs());
        }

        function getCardPriceText(card) {
          // Card structure (3 children):
          //   [0] label  .text-white.font-bold              "BTC/USDT"
          //   [1] price  div > span.font-bold               "71,419.63"
          //   [2] change div > span, span                   "-4.38  -3.88%"
          var priceBlock = card.children[1];
          if (!priceBlock) return '';
          var priceSpan =
            priceBlock.querySelector('span.font-bold') ||
            priceBlock.querySelector('span') ||
            priceBlock;
          return (priceSpan.textContent || '').trim();
        }

        function watchTopCardPrices() {
          // Iterate only the currently-visible (whitelisted) grid cards.
          var grids = document.querySelectorAll('.grid.grid-cols-3');
          grids.forEach(function (grid) {
            if (grid.children.length < 12) return;
            var sample = grid.children[0];
            if (!sample) return;
            var sampleCls = (sample.className || '').toString();
            if (!/press-effect/.test(sampleCls) || !/rounded/.test(sampleCls)) return;

            Array.from(grid.children).forEach(function (card) {
              if (card.style.display === 'none') return; // hidden by limitTopGrid
              var label =
                card.querySelector('.text-white.font-bold') ||
                card.firstElementChild;
              var symbolText = label ? (label.textContent || '').trim() : '';
              if (!symbolText) return;

              var priceText = getCardPriceText(card);
              var curNum = parsePriceNumber(priceText);
              var prev = prevPriceBySymbol[symbolText];

              if (
                prev &&
                Number.isFinite(curNum) &&
                Number.isFinite(prev.num) &&
                curNum !== prev.num
              ) {
                var dir = curNum > prev.num ? 'up' : 'down';
                applyCardFlash(card, dir);
                scheduleReset(symbolText, card);
              }
              prevPriceBySymbol[symbolText] = { num: curNum, text: priceText };
            });
          });
        }

        // --- (4) Tick flash on list row price CHIPS -----------------
        // Each list row's price column contains a span:
        //   DIV .text-right > DIV .max-w-[100px] > SPAN .font-bold.p-1.rounded-md
        // We flash the SPAN's background green/red based on the latest
        // tick direction (NOT the 24h trend) and clear it after the
        // burst period. Independent history keyed by "list:"+symbol so
        // it doesn't collide with the top-card watcher.
        var chipResetTimers = {};

        function getChipEl(row) {
          var priceCol = row.children[3];
          if (!priceCol) return null;
          return priceCol.querySelector('span.rounded-md') ||
                 priceCol.querySelector('span.font-bold');
        }

        function applyChipFlash(chip, direction) {
          if (direction === 'up') {
            chip.style.setProperty('--mirror-flash', CHIP_UP_BG);
          } else if (direction === 'down') {
            chip.style.setProperty('--mirror-flash', CHIP_DOWN_BG);
          } else {
            chip.style.removeProperty('--mirror-flash');
          }
        }

        function scheduleChipReset(key, chip) {
          if (chipResetTimers[key]) clearTimeout(chipResetTimers[key]);
          // Randomized hold-time so concurrent ticks don't clear in lock-step.
          chipResetTimers[key] = setTimeout(function () {
            chip.style.removeProperty('--mirror-flash');
            delete chipResetTimers[key];
          }, randomResetMs());
        }

        function watchListPriceChips() {
          var rows = findListRows();
          rows.forEach(function (row) {
            var symbolText = row.children[0]
              ? (row.children[0].textContent || '').trim()
              : '';
            var symbol = normalizeSymbol(symbolText);
            if (!symbol) return;
            var chip = getChipEl(row);
            if (!chip) return;
            var priceText = (chip.textContent || '').trim();
            var curNum = parsePriceNumber(priceText);
            var key = 'list:' + symbol;
            var prev = prevPriceBySymbol[key];
            if (
              prev &&
              Number.isFinite(curNum) &&
              Number.isFinite(prev.num) &&
              curNum !== prev.num
            ) {
              var dir = curNum > prev.num ? 'up' : 'down';
              applyChipFlash(chip, dir);
              scheduleChipReset(key, chip);
            }
            prevPriceBySymbol[key] = { num: curNum, text: priceText };
          });
        }

        function watchPrices() {
          watchTopCardPrices();
          watchListPriceChips();
        }

        // --- (5) Real-time flash via WebSocket DETAIL stream --------
        // Mirror's Vue bundle only subscribes to ws streams for a
        // handful of top coins. We open our OWN connection to the
        // backend's /ws general handler with `{op:'subscribe',type:
        // 'DETAIL'}` — it subscribes to EVERY ticker (crypto via
        // Binance + non-crypto via Yahoo) and pushes per-coin DETAIL
        // messages as they arrive. Processing them one-by-one gives
        // naturally staggered flash timings (no "bulk reset" pop that
        // HTTP batch polling caused).
        //
        // A lightweight HTTP snapshot is still used once on load (and
        // refreshed every 30s) just to build the symbol → showSymbol
        // lookup table — ws messages use lowercase symbols like
        // "btcusdt" which we need to map back to the DOM's "BTC/USDT".
        var POLL_INTERVAL_MS = 30000; // refresh symbol map only
        var prevBackendPrices = {};   // "BTC/USDT" -> number
        var wsSymbolToDisplay = {};   // "btcusdt" -> "BTC/USDT"

        function findTopGrid() {
          var grids = document.querySelectorAll('.grid.grid-cols-3');
          for (var g of grids) {
            if (g.children.length < 12) continue;
            var sample = g.children[0];
            if (!sample) continue;
            var sc = (sample.className || '').toString();
            if (/press-effect/.test(sc) && /rounded/.test(sc)) return g;
          }
          return null;
        }

        function applyFlashBySymbol(symbolUpper, direction) {
          // Top 12 grid card
          var grid = findTopGrid();
          if (grid) {
            for (var card of grid.children) {
              if (card.style.display === 'none') continue;
              var label =
                card.querySelector('.text-white.font-bold') ||
                card.firstElementChild;
              var sym = label ? (label.textContent || '').trim().toUpperCase() : '';
              if (sym === symbolUpper) {
                applyCardFlash(card, direction);
                scheduleReset('top:' + sym, card);
                break;
              }
            }
          }
          // List row chip
          var rows = findListRows();
          for (var row of rows) {
            var rs =
              (row.children[0] && row.children[0].textContent)
                ? row.children[0].textContent.trim().toUpperCase()
                : '';
            if (rs === symbolUpper) {
              var chip = getChipEl(row);
              if (chip) {
                applyChipFlash(chip, direction);
                // Use the shared helper so jitter + "poll:" key are consistent.
                scheduleChipReset('poll:' + symbolUpper, chip);
              }
              break;
            }
          }
        }

        // Refresh the symbol map + seed prevBackendPrices. No flash
        // triggering — the ws handler does that on every incoming tick.
        function refreshSymbolMap() {
          fetch('/api/common/getCoinSecondList', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: '{}',
          })
            .then(function (r) { return r.json(); })
            .then(function (data) {
              if (!data || data.code !== 200) return;
              var coinList =
                (data.data && (data.data.coinList || data.data.contractList)) || [];
              if (!Array.isArray(coinList)) return;
              coinList.forEach(function (c) {
                var showSym = (c.showSymbol || c.symbol || '').toUpperCase();
                var wsSym = (c.symbol || '').toLowerCase();
                if (showSym && wsSym) {
                  wsSymbolToDisplay[wsSym] = showSym;
                }
                // Seed prices so first tick has a baseline.
                var amt = Number(c.amount);
                if (showSym && Number.isFinite(amt) && prevBackendPrices[showSym] === undefined) {
                  prevBackendPrices[showSym] = amt;
                }
              });
            })
            .catch(function () { /* ignore network hiccups */ });
        }
        // Legacy alias kept for any remaining call sites.
        var pollCoinPrices = refreshSymbolMap;

        // --- Main tick -----------------------------------------------
        var lastTickAt = 0;
        function tick() {
          var now = Date.now();
          // Throttle to 10 Hz max
          if (now - lastTickAt < 80) return;
          lastTickAt = now;
          try {
            limitTopGrid();
            injectAllSparklines();
            watchPrices();
          } catch (e) {
            // Never let a DOM hiccup break the rest of the page.
          }
        }

        // Run on initial load + on every meaningful DOM mutation.
        function start() {
          tick();
          var mo = new MutationObserver(function (mutations) {
            // Only care about subtree changes that might add rows or update
            // price text. We debounce by scheduling a single tick.
            if (start._pending) return;
            start._pending = true;
            requestAnimationFrame(function () {
              start._pending = false;
              tick();
            });
          });
          mo.observe(document.body, {
            childList: true,
            subtree: true,
            characterData: true,
          });
          // Fallback poll in case MO misses something (e.g. canvas repaint)
          setInterval(tick, 1000);

          // HTTP polling fallback for "quiet" coins that the mirror's
          // Vue bundle never pushes DOM updates for. The first call
          // seeds `
