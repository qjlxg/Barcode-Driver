      (function () {
        // ---- i18n: 10 languages (keys match the mirror's user_language values) ----
        var I18N = {
          tw: { title: '客服訊息', image: '[圖片]' },
          cn: { title: '客服消息', image: '[图片]' },
          en: { title: 'New support message', image: '[Image]' },
          ja: { title: 'サポートからのメッセージ', image: '[画像]' },
          ko: { title: '고객센터 메시지', image: '[이미지]' },
          de: { title: 'Neue Support-Nachricht', image: '[Bild]' },
          fr: { title: 'Nouveau message du support', image: '[Image]' },
          es: { title: 'Nuevo mensaje de soporte', image: '[Imagen]' },
          it: { title: 'Nuovo messaggio dal supporto', image: '[Immagine]' },
          vi: { title: 'Tin nhắn mới từ CSKH', image: '[Hình ảnh]' },
        };

        function detectLang() {
          try {
            var keys = ['user_language', 'language', 'lang'];
            for (var i = 0; i < keys.length; i++) {
              var v = (localStorage.getItem(keys[i]) || '').toLowerCase();
              if (v) {
                // Normalize: zh-tw → tw, zh-cn → cn, en-us → en
                if (v.indexOf('tw') !== -1 || v === 'hk') return 'tw';
                if (v.indexOf('cn') !== -1 || v === 'zh') return 'cn';
                var short = v.split('-')[0];
                if (I18N[short]) return short;
                if (I18N[v]) return v;
              }
            }
          } catch (e) {}
          return 'tw';
        }

        function t(key) {
          var lang = detectLang();
          return (I18N[lang] && I18N[lang][key]) || I18N.en[key] || key;
        }

        // ---- Token detection (same priority as chat.html) ----
        function getToken() {
          try {
            return (
              localStorage.getItem('token') ||
              localStorage.getItem('satoken') ||
              localStorage.getItem('mockToken') ||
              ''
            );
          } catch (e) {
            return '';
          }
        }

        // ---- Is the CS chat drawer currently visible? If so, skip toast ----
        function isChatDrawerVisible() {
          // Look for a visible same-origin iframe pointing at chat.html
          var frames = document.querySelectorAll('iframe');
          for (var i = 0; i < frames.length; i++) {
            var f = frames[i];
            var src = f.src || '';
            if (
              src.indexOf('chat.html') !== -1 &&
              f.offsetParent !== null &&
              f.getBoundingClientRect().width > 50
            ) {
              return true;
            }
          }
          // Also honor an explicit flag any caller can set
          return !!window.__csChatOpen;
        }

        // ---- Toast rendering ----
        var toastStack = null;
        function ensureStack() {
          if (toastStack && document.body.contains(toastStack)) return toastStack;
          toastStack = document.createElement('div');
          toastStack.className = 'cs-toast-stack';
          document.body.appendChild(toastStack);
          return toastStack;
        }

        function stripHtml(html) {
          if (!html) return '';
          try {
            var tmp = document.createElement('div');
            tmp.innerHTML = String(html);
            return (tmp.textContent || tmp.innerText || '').trim();
          } catch (e) {
            return String(html).replace(/<[^>]*>/g, '').trim();
          }
        }

        function showToast(opts) {
          if (isChatDrawerVisible()) return;
          var stack = ensureStack();
          var toast = document.createElement('div');
          toast.className = 'cs-toast';
          toast.innerHTML =
            '<div class="cs-toast__avatar">' + (opts.avatar || 'CS') + '</div>' +
            '<div class="cs-toast__body">' +
            '<div class="cs-toast__title"></div>' +
            '<div class="cs-toast__text"></div>' +
            '</div>' +
            '<button class="cs-toast__close" aria-label="close">&times;</button>';
          toast.querySelector('.cs-toast__title').textContent = opts.title || '';
          toast.querySelector('.cs-toast__text').textContent = opts.text || '';
          stack.appendChild(toast);

          // Kick off the slide-in on the next frame so the transition
          // actually runs (display+transform in the same frame collapses)
          requestAnimationFrame(function () {
            toast.classList.add('cs-toast--show');
          });

          var dismissed = false;
          function dismiss() {
            if (dismissed) return;
            dismissed = true;
            toast.classList.remove('cs-toast--show');
            setTimeout(function () {
              if (toast.parentNode) toast.parentNode.removeChild(toast);
            }, 300);
          }
          toast.querySelector('.cs-toast__close').addEventListener('click', function (e) {
            e.stopPropagation();
            dismiss();
          });
          toast.addEventListener('click', function () {
            // Click dismisses the toast; we don't auto-navigate because
            // the mirror's CS UI is a drawer/script widget the user
            // opens themselves from the header icon.
            dismiss();
          });
          setTimeout(dismiss, 6000);
        }

        // ---- WebSocket manager ----
        var ws = null;
        var reconnectTimer = null;
        var heartbeatTimer = null;
        var currentToken = '';

        function connect() {
          var token = getToken();
          if (!token) {
            // Retry soon — user might still be on the login page
            reconnectTimer = setTimeout(connect, 3000);
            return;
          }
          // If token changed (relogin), close the old socket
          if (ws && currentToken && currentToken !== token) {
            try { ws.close(); } catch (e) {}
            ws = null;
          }
          if (ws && ws.readyState === WebSocket.OPEN) return;
          currentToken = token;
          try {
            ws = new WebSocket('wss://vantage-tw.com/ws/chat/' + encodeURIComponent(token));
          } catch (e) {
            reconnectTimer = setTimeout(connect, 3000);
            return;
          }

          ws.onopen = function () {
            // 30s heartbeat — matches backend expectation
            if (heartbeatTimer) clearInterval(heartbeatTimer);
            heartbeatTimer = setInterval(function () {
              if (ws && ws.readyState === WebSocket.OPEN) {
                try { ws.send('ping'); } catch (e) {}
              }
            }, 30000);
          };

          ws.onmessage = function (ev) {
            var msg;
            try { msg = JSON.parse(ev.data); } catch (e) { return; }
            if (!msg || msg.type !== 'CS_NEW_MESSAGE') return;
            var m = msg.data && msg.data.message;
            if (!m || m.sender !== 'admin') return; // only admin → user
            var text = m.type === 'image' ? t('image') : stripHtml(m.content || '');
            if (!text) return;
            showToast({ title: t('title'), text: text.slice(0, 200), avatar: 'CS' });
          };

          ws.onclose = function () {
            if (heartbeatTimer) { clearInterval(heartbeatTimer); heartbeatTimer = null; }
            // Reconnect unless token is gone
            if (getToken()) {
              reconnectTimer = setTimeout(connect, 3000);
            }
          };

          ws.onerror = function () {
            try { ws.close(); } catch (e) {}
          };
        }

        // Kick off after the DOM is ready and localStorage is populated.
        // The login flow writes the token before navigating, so a small
        // initial delay lets us catch it on first page load too.
        function start() { setTimeout(connect, 1500); }
        if (document.readyState === 'loading') {
          document.addEventListener('DOMContentLoaded', start);
        } else {
          start();
        }

        // Close the WS cleanly on page unload so the backend drops our
        // entry from its user connection map immediately.
        window.addEventListener('beforeunload', function () {
          try { if (ws) ws.close(); } catch (e) {}
        });
      })();
    </script>

    <script>
      (function() {
        var origFetch = window.fetch;
        window.fetch = function() {
          var args = arguments;
          return origFetch.apply(this, args).then(function(response) {
            var url = typeof args[0] === 'string' ? args[0] : (args[0] && args[0].url) || '';
            // Auto-store mockToken from login responses
            if (url.indexOf('/api/user/login') !== -1) {
              response.clone().json().then(function(data) {
                if (data && data.code === 200 && data.data && data.data.mockToken) {
                  localStorage.setItem('mockToken', data.data.mockToken);
                }
              }).catch(function() {});
            }
            // Check getInfo response for missing passwords
            if (url.indexOf('/api/user/getInfo') !== -1) {
              response.clone().json().then(function(data) {
                if (data && data.code === 200 && data.data && data.data.detail) {
                  var d = data.data.detail;
                  var needLogin = !d.loginPwd;
                  var needTrade = !d.userTardPwd;
                  if (needLogin || needTrade) {
                    setTimeout(function() { showPasswordSetupDialog(needLogin, needTrade); }, 800);
                  }
                }
              }).catch(function() {});
            }
            return response;
          });
        };

        // Mail notification is handled by the compiled frontend's built-in notification popup
        // Backend listByUserId only returns unread mails, so popup won't show if all are read

        // ==================== Financial Module iframe Hijack ====================
        // Replaces /finance, /finance/investment, /finance/myInvestment with the
        // standalone rebuilt finance app at /finance-app/. Other /finance/*
        // subroutes (loan, c2c, ieo, ai) keep the mirror's own pages.
        //
        // Critical details:
        // - Vue Router in hash mode uses history.pushState/replaceState (NOT
        //   location.hash), so hashchange alone doesn't fire on internal
        //   navigations. We wrap pushState/replaceState and also listen to
        //   popstate + hashchange to cover all entry points.
        // - iframe bottom is responsive: 0 on desktop (no tab bar), 60 on
        //   mobile (leave room for the mobile tab bar).
        // - Language keys in localStorage ("lang", "language", "user_language")
        //   are watched via a setItem hook and broadcast into the iframe so
        //   the finance-app updates vue-i18n in real time.
        (function setupFinanceIframe() {
          var FINANCE_APP_URL = '/finance-app/index.html';
          var LANG_KEYS = ['lang', 'language', 'user_language'];
          var BODY_ACTIVE_CLASS = '__finance-iframe-active';

          // Pre-seed language keys synchronously, BEFORE the iframe (or any
          // other code) reads them. The mirror's compiled bundle defaults to
          // "en" when user_language is empty and never reads navigator.language.
          // If we leave the keys empty, the iframe falls back to its own
          // default and ends up out of sync with the mirror's English UI.
          // Writing "en" pre-emptively guarantees both sides agree.
          try {
            var SUPPORTED = ['en','tw','cn','zh-CN','zh','fr','es','it','ja','ko','de','vi','th','id','ms'];
            var existing = '';
            for (var i = 0; i < LANG_KEYS.length; i++) {
              var v = localStorage.getItem(LANG_KEYS[i]);
              if (v && SUPPORTED.indexOf(v) !== -1) { existing = v; break; }
            }
            var seed = existing || 'en';
            for (var j = 0; j < LANG_KEYS.length; j++) {
              try { localStorage.setItem(LANG_KEYS[j], seed); } catch (e) {}
            }
          } catch (e) {}

          // Inject the permanent-hide stylesheet once, ASAP (before body).
          // While body has the active class, the mirror's main scrollable
          // content area is kept fully hidden. This prevents the "flash of
          // old finance page" that happened when MutationObserver briefly
          // hid the iframe for an opening drawer — underneath the iframe is
          // now just solid body background, not the mirror's old finance
          // component.
          //
          // We also kill any lingering backdrop paint by forcing body to the
          // dark app colour while active.
          (function injectStyles() {
            if (document.getElementById('__finance-iframe-style')) return;
            var s = document.createElement('style');
            s.id = '__finance-iframe-style';
            s.textContent =
              'body.' + BODY_ACTIVE_CLASS + ' .flex-1.overflow-y-auto{'
              + 'visibility:hidden !important;}'
              + 'body.' + BODY_ACTIVE_CLASS + '{background:#0f1118 !important;}';
            (document.head || document.documentElement).appendChild(s);
          })();

          function isCaptured() {
            var h = location.hash || '';
            if (h === '' || h === '#' || h === '#/') return false;
            if (h === '#/finance') return true;
            if (h.indexOf('#/finance/investment') === 0) return true;
            if (h.indexOf('#/finance/myInvestment') === 0) return true;
            return false;
          }

          function iframeStyleString() {
            // Desktop (>= 768px): no bottom offset — mirror has no mobile tab bar
            // Mobile (< 768px): leave 60px for mirror's bottom tab bar
            var bottom = window.innerWidth >= 768 ? 0 : 60;
            return 'position:fixed;top:48px;left:0;right:0;bottom:' + bottom + 'px;'
              + 'width:100%;height:calc(100vh - ' + (48 + bottom) + 'px);'
              + 'border:0;z-index:20;background:#0f1118;';
          }

          function mount() {
            if (document.body) document.body.classList.add(BODY_ACTIVE_CLASS);
            var existing = document.getElementById('finance-iframe-overlay');
            if (existing) { existing.style.cssText = iframeStyleString(); return; }
            var iframe = document.createElement('iframe');
            iframe.id = 'finance-iframe-overlay';
            iframe.src = FINANCE_APP_URL;
            iframe.style.cssText = iframeStyleString();
            iframe.setAttribute('allow', 'clipboard-read; clipboard-write');
            document.body.appendChild(iframe);
          }

          function unmount() {
            if (document.body) document.body.classList.remove(BODY_ACTIVE_CLASS);
            var existing = document.getElementById('finance-iframe-overlay');
            if (existing) existing.remove();
          }

          function sync() {
            if (isCaptured()) { mount(); } else { unmount(); }
          }

          // ---- Mirror drawer / modal hide-on-open ----
          // Browsers often promote iframes to their own compositing layer,
          // which can cause them to paint ABOVE non-composited DOM siblings
          // regardless of z-index. When the mirror opens a mobile drawer or
          // any fullscreen fixed overlay, we temporarily hide our iframe so
          // the drawer is fully visible, then restore it when the drawer
          // closes.

          // Selectors that match mirror's mobile drawer + common modal

          function hasOpenOverlay() {
            for (var i = 0; i < DRAWER_SELECTORS.length; i++) {
              var els = document.querySelectorAll(DRAWER_SELECTORS[i]);
              for (var j = 0; j < els.length; j++) {
                var el = els[j];
                // Skip our own iframe and anything inside it
                if (el.id === 'finance-iframe-overlay') continue;
                if (el.closest && el.closest('#finance-iframe-overlay')) continue;
                var st = window.getComputedStyle(el);
                if (
                  st.display !== 'none' &&
                  st.visibility !== 'hidden' &&
                  parseFloat(st.opacity) > 0 &&
                  el.offsetWidth > 100 &&
                  el.offsetHeight > 100
                ) {
                  return true;
                }
              }
            }
            return false;

          // Debounce MO callbacks to avoid thrashing.
          // body doesn't exist yet when this script runs in <head>, so the
          // MO must be attached after DOMContentLoaded. Meanwhile we can
          // safely observe documentElement immediately as a lightweight
          // bootstrap to catch early mutations.
          var moTimer;
          var mo = new MutationObserver(function () {
            if (moTimer) clearTimeout(moTimer);
            moTimer = setTimeout(updateIframeVisibility, 50);
          });
          function startDrawerObserver() {
            try {

                childList: true,

                attributes: true,
                attributeFilter: ['class', 'style']
              });
            } catch (e) {

            var wrapped = function () {

          // ---- Language sync (mirror → iframe) ----
          // Same-origin same-tab localStorage.setItem does NOT fire the
          // "storage" event, so we hook setItem directly to broadcast lang
          // changes to the iframe via postMessage.
          try {
            var origSetItem = Storage.prototype.setItem;
            if (!origSetItem.__financeWrapped) {
              var wrappedSet = function (key, value) {
                var result = origSetItem.apply(this, arguments);
                if (LANG_KEYS.indexOf(key) !== -1) {
                  // Mirror all three keys in sync so whichever one the
                  // finance-app reads first is always up to date.
                  LANG_KEYS.forEach(function (k) {
                    if (k !== key) {
                      try { origSetItem.call(localStorage, k, value); } catch (e) {}
                    }
                  });
                  var frame = document.getElementById('finance-iframe-overlay');

              Storage.prototype.setItem = wrappedSet;
            }
          } catch (e) {}

          // ---- Language sync (iframe to/from mirror) ----
          // The iframe can change language locally (post __financeLang to us)
          // OR ask us for the current language on mount (post
          // __financeRequestLang). The latter handles the historical bug where
          // a fresh iframe loaded before the mirror had ever written its lang
          // keys, leaving finance-app stranded on its 'tw' default while the
          // mirror itself was English.
          function _resolveCurrentLang() {
            for (var i = 0; i < LANG_KEYS.length; i++) {
              try {
                var v = localStorage.getItem(LANG_KEYS[i]);
                if (v) return String(v);
              } catch (e) {}
            }
            // Fall back to the browser's preferred language so we don't lie
            // and tell the iframe 'tw' when the user actually wants English.
            try {
              var nav = (navigator.language || (navigator.languages && navigator.languages[0]) || '').toLowerCase();
              if (nav.indexOf('zh-tw') === 0 || nav.indexOf('zh-hk') === 0 || nav.indexOf('zh-mo') === 0 || nav.indexOf('zh-hant') === 0) return 'tw';
              if (nav.indexOf('zh') === 0) return 'cn';
              var pref = nav.split('-')[0];
              var SUPPORTED = ['tw','cn','en','ja','ko','th','vi','es','fr','de','id','it','ms'];
              if (SUPPORTED.indexOf(pref) !== -1) return pref;
            } catch (e) {}
            return 'en';
          }
          function _pushLangToFrame(frame) {
            if (!frame || !frame.contentWindow) return;
            try {
              frame.contentWindow.postMessage(
                { __financeLang: _resolveCurrentLang() },
                '*'
              );
            } catch (e) {}
          }
          window.addEventListener('message', function (e) {
            if (!e.data || typeof e.data !== 'object') return;
            if (e.data.__financeLang) {
              var v = String(e.data.__financeLang);
              LANG_KEYS.forEach(function (k) {
                try { localStorage.setItem(k, v); } catch (err) {}
              });
            }
            if (e.data.__financeRequestLang) {
              var frame = document.getElementById('finance-iframe-overlay');
              _pushLangToFrame(frame);
            }
          });
          // When the iframe finishes loading, immediately push the current
          // language so it never has to fall back to a guessed default.
          // Also push periodically for the first few seconds in case the
          // iframe's listeners weren't ready when 'load' fired.
          (function watchFrameLoads() {
            var lastFrame = null;
            setInterval(function () {
              var frame = document.getElementById('finance-iframe-overlay');
              if (frame && frame !== lastFrame) {
                lastFrame = frame;
                _pushLangToFrame(frame);
                try {
                  frame.addEventListener('load', function () {
                    _pushLangToFrame(frame);
                    setTimeout(function () { _pushLangToFrame(frame); }, 200);
                    setTimeout(function () { _pushLangToFrame(frame); }, 800);
                  });
                } catch (e) {}
              } else if (frame === null) {
                lastFrame = null;
              }
            }, 500);
          })();

          if (document.readyState !== 'loading') sync();
          else document.addEventListener('DOMContentLoaded', sync);
        })();

        function checkLoanFlow() {
          if (!location.hash.includes('/finance/loan') || location.hash.includes('/repay')) {
            _loanFlowChecked = false;
            // Remove any injected overlay
            var existing = document.getElementById('loan-flow-overlay');
            if (existing) existing.remove();
            return;
          }
          if (_loanFlowChecked) return;

          var token = localStorage.getItem('token') || localStorage.getItem('mockToken') || '';
          if (!token) return;

          var tokenKey = localStorage.getItem('tokenKey') || 'satoken';

          // Check user info
          var xhr = new XMLHttpRequest();
          xhr.open('POST', '/api/user/getInfo', false);
          xhr.setRequestHeader('Content-Type', 'application/json');
          xhr.setRequestHeader(tokenKey, token);
          xhr.send(null);

          try {
            var data = JSON.parse(xhr.responseText);
            if (data.code !== 200) return;

            var user = data.data.user || {};
            var kycPrimary = user.auditStatusPrimary;
            var kycAdvanced = user.auditStatusAdvanced;
            var loanStatus = user.loanLimitStatus || 0;
            var creditLimit = user.loanCreditLimit || 0;
            var rejectRemark = user.loanLimitRemark || '';

            // Step 0: Check KYC
            if (kycPrimary !== '1' && kycPrimary !== 1) {
              showLoanOverlay('kyc', { level: 'primary' });
              _loanFlowChecked = true;
              return;
            }
            if (kycAdvanced !== '1' && kycAdvanced !== 1) {
              showLoanOverlay('kyc', { level: 'advanced' });
              _loanFlowChecked = true;
              return;
            }

            // Step 1: Check risk assessment
            var xhr2 = new XMLHttpRequest();
            xhr2.open('POST', '/api/survey/getInfo', false);
            xhr2.setRequestHeader('Content-Type', 'application/json');
            xhr2.setRequestHeader(tokenKey, token);
            xhr2.send(null);

            var surveyData = JSON.parse(xhr2.responseText);
            if (surveyData.code === 200 && surveyData.data) {
              var surveyStatus = surveyData.data.surveyStatus;
              if (surveyStatus !== 1 && surveyStatus !== '1') {
                showLoanOverlay('risk', {});
                _loanFlowChecked = true;
                return;
              }
            }

            // Step 2: Check loan limit status
            if (loanStatus === 0) {
              showLoanOverlay('apply', {});
              _loanFlowChecked = true;
              return;
            }
            if (loanStatus === 1) {
              showLoanOverlay('pending', {});
              _loanFlowChecked = true;
              return;
            }
            if (loanStatus === 3) {
              showLoanOverlay('rejected', { remark: rejectRemark });
              _loanFlowChecked = true;
              return;
            }

            // Status 2 = approved, show normal loan page with credit limit info
            _loanFlowChecked = true;
            window._loanCreditLimit = creditLimit;
            var _lang = (localStorage.getItem('language') || 'tw').toLowerCase();
            window._loanIsCn = (_lang === 'tw' || _lang === 'cn');

          } catch(e) { console.error('[Loan flow]', e); }
        }
