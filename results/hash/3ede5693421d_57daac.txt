  <script>
    const STORAGE_API_URL = 'remn_monitor_api_url';
    const STORAGE_TOKEN = 'remn_monitor_token';

    const apiUrlEl = document.getElementById('apiUrl');
    const apiTokenEl = document.getElementById('apiToken');
    const configArea = document.getElementById('configArea');
    const proxyBadge = document.getElementById('proxyBadge');
    const manualConfig = document.getElementById('manualConfig');
    const autoRefreshEl = document.getElementById('autoRefresh');
    const refreshSecEl = document.getElementById('refreshSec');
    const btnLoad = document.getElementById('btnLoad');
    const errorBanner = document.getElementById('errorBanner');
    const loadingEl = document.getElementById('loading');
    const summaryEl = document.getElementById('summary');
    const nodesGrid = document.getElementById('nodesGrid');
    const lastUpdateEl = document.getElementById('lastUpdate');

    let refreshTimer = null;
    let useProxyMode = false;

    async function detectProxy() {
      try {
        const res = await fetch('/api/proxy/config');
        const data = await res.json().catch(() => ({}));
        if (data.useProxy) {
          useProxyMode = true;
          proxyBadge.style.display = 'block';
          manualConfig.style.display = 'none';
          return true;
        }
      } catch (_) {}
      useProxyMode = false;
      proxyBadge.style.display = 'none';
      manualConfig.style.display = 'flex';
      return false;
    }

    function loadStored() {
      try {
        const url = localStorage.getItem(STORAGE_API_URL);
        const token = localStorage.getItem(STORAGE_TOKEN);
        if (url) apiUrlEl.value = url;
        if (token) apiTokenEl.value = token;
      } catch (_) {}
    }

    function saveStored() {
      try {
        localStorage.setItem(STORAGE_API_URL, apiUrlEl.value.trim());
        localStorage.setItem(STORAGE_TOKEN, apiTokenEl.value);
      } catch (_) {}
    }

    function showError(msg) {
      errorBanner.textContent = msg;
      errorBanner.style.display = 'block';
    }

    function hideError() {
      errorBanner.style.display = 'none';
    }

    function formatBytes(n) {
      if (n == null || n === 0) return '—';
      const units = ['B', 'KB', 'MB', 'GB', 'TB'];
      let i = 0;
      while (n >= 1024 && i < units.length - 1) { n /= 1024; i++; }
      return n.toFixed(2) + ' ' + units[i];
    }

    function formatBytesLifetime(raw) {
      if (raw == null || raw === '') return '—';
      const bytes = typeof raw === 'number' ? raw : Number(raw);
      if (isNaN(bytes) || bytes < 0) return String(raw);
      return formatBytes(bytes);
    }

    function formatDate(s) {
      if (!s) return '—';
      try {
        const d = new Date(s);
        return d.toLocaleString('ru-RU', { dateStyle: 'short', timeStyle: 'short' });
      } catch (_) { return s; }
    }

    async function fetchStats() {
      if (useProxyMode) {
        try {
          const res = await fetch('/api/proxy/stats');
          const data = await res.json().catch(() => ({}));
          if (res.ok && data && data.response) return data.response;
        } catch (_) {}
        return null;
      }
      const base = (apiUrlEl.value || '').trim().replace(/\/$/, '');
      const token = (apiTokenEl.value || '').trim();
      if (!base) return null;
      try {
        const res = await fetch(`${base}/api/system/stats`, {
          headers: token ? { 'Authorization': `Bearer ${token}` } : {},
        });
        const data = await res.json().catch(() => ({}));
        if (res.ok && data && data.response) return data.response;
      } catch (_) {}
      return null;
    }

    async function fetchNodes() {
      if (useProxyMode) {
        hideError();
        loadingEl.style.display = 'block';
        summaryEl.style.display = 'none';
        nodesGrid.innerHTML = '';
        try {
          const res = await fetch('/api/proxy/nodes');
          const data = await res.json().catch(() => ({}));
          if (!res.ok) {
            throw new Error(data.message || `HTTP ${res.status}`);
          }
          if (!data || !Array.isArray(data.response)) {
            throw new Error('Неверный формат ответа');
          }
          loadingEl.style.display = 'none';
          return data.response;
        } catch (e) {
          loadingEl.style.display = 'none';
          showError(e.message || 'Ошибка загрузки');
          return null;
        }
      }
      const base = (apiUrlEl.value || '').trim().replace(/\/$/, '');
      const token = (apiTokenEl.value || '').trim();
      if (!base) {
        showError('Укажите API Base URL');
        return null;
      }
      saveStored();
      hideError();
      loadingEl.style.display = 'block';
      summaryEl.style.display = 'none';
      nodesGrid.innerHTML = '';

      try {
        const res = await fetch(`${base}/api/nodes`, {
          headers: token ? { 'Authorization': `Bearer ${token}` } : {},
        });
        if (!res.ok) {
          const text = await res.text();
          throw new Error(`HTTP ${res.status}: ${text.slice(0, 200)}`);
        }
        const data = await res.json();
        if (!data || !Array.isArray(data.response)) {
          throw new Error('Неверный формат ответа: ожидается { response: [...] }');
        }
        loadingEl.style.display = 'none';
        return data.response;
      } catch (e) {
        loadingEl.style.display = 'none';
        showError(e.message || 'Ошибка загрузки');
        return null;
      }
    }

    function renderSummary(nodes, stats) {
      const online = nodes.filter(n => n.isConnected && !n.isDisabled).length;
      const offline = nodes.filter(n => !n.isConnected && !n.isDisabled).length;
      const disabled = nodes.filter(n => n.isDisabled).length;
      const usersOnlineSum = nodes.reduce((s, n) => s + (n.usersOnline ?? 0), 0);
      const totalTrafficUsed = nodes.reduce((s, n) => s + (n.trafficUsedBytes ?? 0), 0);
      const totalTrafficLimit = nodes.reduce((s, n) => s + (n.trafficLimitBytes ?? 0), 0);

      const totalUsers = stats && stats.users && typeof stats.users.totalUsers === 'number' ? stats.users.totalUsers : null;
      const os = stats && stats.onlineStats ? stats.onlineStats : null;
      const nodesStats = stats && stats.nodes ? stats.nodes : null;

      summaryEl.innerHTML = `
        <div class="summary-card">
          <div class="value">${nodes.length}</div>
          <div class="label">Всего нод</div>
        </div>
        <div class="summary-card">
          <div class="value green">${online}</div>
          <div class="label">Нод онлайн</div>
        </div>
        <div class="summary-card">
          <div class="value red">${offline}</div>
          <div class="label">Нод офлайн</div>
        </div>
        <div class="summary-card">
          <div class="value orange">${disabled}</div>
          <div class="label">Нод отключено</div>
        </div>
        <div class="summary-card">
          <div class="value">${totalUsers != null ? totalUsers : '—'}</div>
          <div class="label">Всего пользователей</div>
        </div>
        <div class="summary-card">
          <div class="value green">${usersOnlineSum}</div>
          <div class="label">Пользователей онлайн</div>
        </div>
        ${os ? `
        <div class="summary-card">
          <div class="value green">${os.onlineNow ?? '—'}</div>
          <div class="label">Онлайн (система)</div>
        </div>
        <div class="summary-card">
          <div class="value">${os.lastDay != null ? os.lastDay : '—'}</div>
          <div class="label">Заходили за 24 ч</div>
        </div>
        <div class="summary-card">
          <div class="value">${os.lastWeek != null ? os.lastWeek : '—'}</div>
          <div class="label">Заходили за 7 дней</div>
        </div>
        <div class="summary-card">
          <div class="value orange">${os.neverOnline != null ? os.neverOnline : '—'}</div>
          <div class="label">Никогда не заходили</div>
        </div>
        ` : ''}
        <div class="summary-card">
          <div class="value">${formatBytes(totalTrafficUsed)}</div>
          <div class="label">Трафик использовано</div>
        </div>
        <div class="summary-card">
          <div class="value">${totalTrafficLimit ? formatBytes(totalTrafficLimit) : '—'}</div>
          <div class="label">Лимит трафика</div>
        </div>
        ${nodesStats && (nodesStats.totalBytesLifetime != null && nodesStats.totalBytesLifetime !== '') ? `
        <div class="summary-card">
          <div class="value">${formatBytesLifetime(nodesStats.totalBytesLifetime)}</div>
          <div class="label">Трафик нод (всего)</div>
        </div>
        ` : ''}
      `;
      summaryEl.style.display = 'grid';
    }

    function renderNodes(nodes) {
      nodesGrid.innerHTML = nodes.map(node => {
        const status = node.isDisabled ? 'disabled' : node.isConnecting ? 'connecting' : node.isConnected ? 'online' : 'offline';
        const trafficLimit = node.trafficLimitBytes ?? 0;
        const trafficUsed = node.trafficUsedBytes ?? 0;
        const trafficPct = trafficLimit > 0 ? Math.min(100, (trafficUsed / trafficLimit) * 100) : 0;
        const trafficClass = trafficPct >= 90 ? 'danger' : trafficPct >= 70 ? 'warn' : '';

        return `
          <article class="node-card ${status}" data-uuid="${node.uuid}">
            <div class="node-header">
              <span class="node-name">${escapeHtml(node.name || '—')}</span>
              <span class="node-status ${status}">${status === 'online' ? 'Online' : status === 'offline' ? 'Offline' : status === 'connecting' ? 'Connecting' : 'Disabled'}</span>
            </div>
            <div class="node-metrics">
              <div class="metric">
                <span class="key">Адрес</span>
                <span class="val">${escapeHtml(node.address || '—')}${node.port ? ':' + node.port : ''}</span>
              </div>
              <div class="metric">
                <span class="key">Пользователей онлайн</span>
                <span class="val highlight">${node.usersOnline ?? '—'}</span>
              </div>
              <div class="metric">
                <span class="key">Uptime (xray)</span>
                <span class="val">${escapeHtml(node.xrayUptime || '—')}</span>
              </div>
              <div class="metric">
                <span class="key">Версия</span>
                <span class="val">${escapeHtml(node.xrayVersion || node.nodeVersion || '—')}</span>
              </div>
              ${node.cpuCount != null || node.cpuModel ? `
              <div class="metric">
                <span class="key">CPU</span>
                <span class="val">${node.cpuCount != null ? node.cpuCount + ' × ' : ''}${escapeHtml(node.cpuModel || '—')}</span>
              </div>
              ` : ''}
              ${node.totalRam ? `
              <div class="metric">
                <span class="key">RAM</span>
                <span class="val">${escapeHtml(node.totalRam)}</span>
              </div>
              ` : ''}
              ${node.isTrafficTrackingActive ? `
              <div class="metric">
                <span class="key">Трафик</span>
                <span class="val">${formatBytes(trafficUsed)} / ${formatBytes(trafficLimit) || '∞'}</span>
              </div>
              <div class="traffic-bar"><div class="traffic-bar-fill ${trafficClass}" style="width:${trafficPct}%"></div></div>
              ` : ''}
              <div class="metric">
                <span class="key">Последнее изменение</span>
                <span class="val">${formatDate(node.lastStatusChange)}</span>
              </div>
              ${(node.lastStatusMessage || '').trim() ? `
              <div class="metric">
                <span class="key">Сообщение</span>
                <span class="val">${escapeHtml(node.lastStatusMessage)}</span>
              </div>
              ` : ''}
            </div>
            ${(node.tags || []).length ? `<div class="tags">${(node.tags || []).map(t => `<span class="tag">${escapeHtml(t)}</span>`).join('')}</div>` : ''}
          </article>
        `;
      }).join('');
    }

    function escapeHtml(s) {
      if (s == null) return '';
      const div = document.createElement('div');
      div.textContent = s;
      return div.innerHTML;
    }

    async function loadAndRender() {
      const [nodes, stats] = await Promise.all([fetchNodes(), fetchStats()]);
      if (!nodes) return;
      renderSummary(nodes, stats);
      renderNodes(nodes);
      lastUpdateEl.textContent = 'Обновлено: ' + new Date().toLocaleString('ru-RU');
      scheduleRefresh();
    }

    function scheduleRefresh() {
      if (refreshTimer) clearTimeout(refreshTimer);
      if (!autoRefreshEl.checked) return;
      const sec = Math.max(10, parseInt(refreshSecEl.value, 10) || 30);
      refreshTimer = setTimeout(loadAndRender, sec * 1000);
    }

    btnLoad.addEventListener('click', loadAndRender);
    autoRefreshEl.addEventListener('change', () => { if (autoRefreshEl.checked) scheduleRefresh(); else if (refreshTimer) clearTimeout(refreshTimer); });
    refreshSecEl.addEventListener('change', scheduleRefresh);

    (async function init() {
      loadStored();
      await detectProxy();
      loadAndRender();
    })();
  </script>
