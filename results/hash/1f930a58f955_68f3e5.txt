                            id="toggleConfirmPassword"
                            class="absolute right-4 top-1/2 -translate-y-1/2 dark:text-dark-text-secondary light:text-light-text-secondary hover:text-primary transition-colors">
                            <svg id="eyeOpenConfirm" class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15 12a3 3 0 11-6 0 3 3 0 016 0z"></path>
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M2.458 12C3.732 7.943 7.523 5 12 5c4.478 0 8.268 2.943 9.542 7-1.274 4.057-5.064 7-9.542 7-4.477 0-8.268-2.943-9.542-7z"></path>
                            </svg>
                            <svg id="eyeClosedConfirm" class="w-5 h-5 hidden" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M13.875 18.825A10.05 10.05 0 0112 19c-4.478 0-8.268-2.943-9.543-7a9.97 9.97 0 011.563-3.029m5.858.908a3 3 0 114.243 4.243M9.878 9.878l4.242 4.242M9.88 9.88l-3.29-3.29m7.532 7.532l3.29 3.29M3 3l3.59 3.59m0 0A9.953 9.953 0 0112 5c4.478 0 8.268 2.943 9.543 7a10.025 10.025 0 01-4.132 5.411m0 0L21 21"></path>
                            </svg>
                        </button>
                    </div>
                </div>

                <button
                    type="button"
                    id="resetPasswordBtn"
                    class="w-full bg-primary hover:bg-primary-hover text-white font-semibold py-4 px-6 rounded-input transition-all duration-200 transform hover:scale-[1.01] active:scale-[0.99] shadow-md hover:shadow-lg">
                    Изменить пароль
                </button>
            </div>

            <div id="step-success" class="hidden text-center space-y-5">
                <div class="inline-flex items-center justify-center w-16 h-16 rounded-full bg-green-500/20 mb-4">
                    <svg class="w-8 h-8 text-green-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path>
                    </svg>
                </div>
                <h3 class="text-xl font-bold dark:text-dark-text light:text-gray-900">Пароль изменён!</h3>
                <p class="dark:text-dark-text-secondary light:text-gray-600 text-sm">Теперь вы можете войти с новым паролем</p>

                <button
                    type="button"
                    id="backToLoginBtn"
                    class="w-full bg-primary hover:bg-primary-hover text-white font-semibold py-4 px-6 rounded-input transition-all duration-200 transform hover:scale-[1.01] active:scale-[0.99] shadow-md hover:shadow-lg mt-4">
                    Войти
                </button>
            </div>
        </div>
    </div>

    <script>
        const html = document.documentElement;
        const themeToggle = document.getElementById('themeToggle');
        const sunIcon = document.getElementById('sunIcon');
        const moonIcon = document.getElementById('moonIcon');

        const savedTheme = localStorage.getItem('theme') || 'dark';
        if (savedTheme === 'light') {
            html.classList.remove('dark');
            html.classList.add('light');
            sunIcon.classList.remove('hidden');
            moonIcon.classList.add('hidden');
        }

        themeToggle.addEventListener('click', () => {
            if (html.classList.contains('dark')) {
                html.classList.remove('dark');
                html.classList.add('light');
                localStorage.setItem('theme', 'light');
                sunIcon.classList.remove('hidden');
                moonIcon.classList.add('hidden');
            } else {
                html.classList.remove('light');
                html.classList.add('dark');
                localStorage.setItem('theme', 'dark');
                sunIcon.classList.add('hidden');
                moonIcon.classList.remove('hidden');
            }
        });

        const togglePassword = document.getElementById('togglePassword');
        const passwordInput = document.getElementById('password');
        const eyeOpen = document.getElementById('eyeOpen');
        const eyeClosed = document.getElementById('eyeClosed');

        togglePassword.addEventListener('click', () => {
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            eyeOpen.classList.toggle('hidden');
            eyeClosed.classList.toggle('hidden');
        });

        const loginForm = document.getElementById('loginForm');
        const twoFactorForm = document.getElementById('twoFactorForm');

        document.getElementById('email').focus();

        const twoFactorInput = document.getElementById('twoFactorCode');
        twoFactorInput.addEventListener('input', (e) => {
            e.target.value = e.target.value.replace(/[^0-9]/g, '');
        });

        const forgotPasswordBtn = document.getElementById('forgotPasswordBtn');
        const passwordResetModal = document.getElementById('passwordResetModal');
        const closePasswordResetModal = document.getElementById('closePasswordResetModal');

        const stepEmail = document.getElementById('step-email');
        const stepCode = document.getElementById('step-code');
        const stepPassword = document.getElementById('step-password');
        const stepSuccess = document.getElementById('step-success');

        const resetEmailInput = document.getElementById('resetEmail');
        const resetCodeInput = document.getElementById('resetCode');
        const newPasswordInput = document.getElementById('newPassword');
        const confirmPasswordInput = document.getElementById('confirmPassword');

        const sendCodeBtn = document.getElementById('sendCodeBtn');
        const verifyCodeBtn = document.getElementById('verifyCodeBtn');
        const resetPasswordBtn = document.getElementById('resetPasswordBtn');
        const resendCodeBtn = document.getElementById('resendCodeBtn');
        const backToLoginBtn = document.getElementById('backToLoginBtn');
        const backToEmailBtn = document.getElementById('backToEmailBtn');

        const telegramUsernameSpan = document.getElementById('telegramUsernameSpan');
        const resetErrorDiv = document.getElementById('resetError');
        const resetErrorText = document.getElementById('resetErrorText');

        let resendTimer = null;

        forgotPasswordBtn.addEventListener('click', () => {
            passwordResetModal.classList.remove('hidden');
            passwordResetModal.classList.add('flex');
            showStep('email');
            resetEmailInput.focus();
        });

        closePasswordResetModal.addEventListener('click', closeModal);
        passwordResetModal.addEventListener('click', (e) => {
            if (e.target === passwordResetModal) closeModal();
        });

        function closeModal() {
            passwordResetModal.classList.add('hidden');
            passwordResetModal.classList.remove('flex');
            clearErrors();
            clearResendTimer();
            resetEmailInput.value = '';
            resetCodeInput.value = '';
            newPasswordInput.value = '';
            confirmPasswordInput.value = '';
        }

        function showStep(step) {
            stepEmail.classList.add('hidden');
            stepCode.classList.add('hidden');
            stepPassword.classList.add('hidden');
            stepSuccess.classList.add('hidden');
            clearErrors();

            switch(step) {
                case 'email':
                    stepEmail.classList.remove('hidden');
                    break;
                case 'code':
                    stepCode.classList.remove('hidden');
                    resetCodeInput.focus();
                    break;
                case 'password':
                    stepPassword.classList.remove('hidden');
                    newPasswordInput.focus();
                    break;
                case 'success':
                    stepSuccess.classList.remove('hidden');
                    break;
            }
        }

        function showError(message) {
            resetErrorText.textContent = message;
            resetErrorDiv.classList.remove('hidden');
        }

        function clearErrors() {
            resetErrorDiv.classList.add('hidden');
            resetErrorText.textContent = '';
        }

        function setButtonLoading(btn, loading) {
            if (loading) {
                btn.disabled = true;
                btn.dataset.originalText = btn.textContent;
                btn.innerHTML = `
                    <svg class="animate-spin h-5 w-5 mx-auto" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                        <circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle>
                        <path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                    </svg>
                `;
            } else {
                btn.disabled = false;
                btn.textContent = btn.dataset.originalText || 'Отправить';
            }
        }

        function startResendTimer(seconds) {
            clearResendTimer();
            resendCodeBtn.disabled = true;

            let remaining = seconds;
            updateResendBtnText(remaining);

            resendTimer = setInterval(() => {
                remaining--;
                if (remaining <= 0) {
                    clearResendTimer();
                    resendCodeBtn.disabled = false;
                    resendCodeBtn.textContent = 'Отправить повторно';
                } else {
                    updateResendBtnText(remaining);
                }
            }, 1000);
        }

        function updateResendBtnText(seconds) {
            resendCodeBtn.textContent = `Отправить повторно (${seconds}с)`;
        }

        function clearResendTimer() {
            if (resendTimer) {
                clearInterval(resendTimer);
                resendTimer = null;
            }
        }

        sendCodeBtn.addEventListener('click', async () => {
            const email = resetEmailInput.value.trim();
            if (!email) {
                showError('Введите email или логин');
                return;
            }

            clearErrors();
            setButtonLoading(sendCodeBtn, true);

            try {
                const response = await fetch('/password-reset/send-code', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': 'HWz5RgrMbav6Ad9PcIYT94jR8HZDvsxmwckYY0Nz',
                        'Accept': 'application/json'
                    },
                    body: JSON.stringify({ email })
                });

                const data = await response.json();

                if (data.success) {
                    telegramUsernameSpan.textContent = data.telegram_username || 'Telegram';
                    showStep('code');
                    startResendTimer(60);
                } else {
                    showError(data.error || 'Произошла ошибка');
                    if (data.retry_after) {
                        startResendTimer(data.retry_after);
                    }
                }
            } catch (error) {
                showError('Ошибка соединения. Попробуйте позже.');
            } finally {
                setButtonLoading(sendCodeBtn, false);
            }
        });

        verifyCodeBtn.addEventListener('click', async () => {
            const code = resetCodeInput.value.trim();
            if (!code || code.length !== 6) {
                showError('Введите 6-значный код');
                return;
            }

            clearErrors();
            setButtonLoading(verifyCodeBtn, true);

            try {
                const response = await fetch('/password-reset/verify-code', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': 'HWz5RgrMbav6Ad9PcIYT94jR8HZDvsxmwckYY0Nz',
                        'Accept': 'application/json'
                    },
                    body: JSON.stringify({ code })
                });

                const data = await response.json();

                if (data.success) {
                    showStep('password');
                } else {
                    showError(data.error || 'Неверный код');
                    if (data.session_expired) {
                        setTimeout(() => showStep('email'), 2000);
                    }
                }
            } catch (error) {
                showError('Ошибка соединения. Попробуйте позже.');
            } finally {
                setButtonLoading(verifyCodeBtn, false);
            }
        });

        resetPasswordBtn.addEventListener('click', async () => {
            const password = newPasswordInput.value;
            const passwordConfirmation = confirmPasswordInput.value;

            if (!password || password.length < 8) {
                showError('Пароль должен быть не менее 8 символов');
                return;
            }

            if (password !== passwordConfirmation) {
                showError('Пароли не совпадают');
                return;
            }

            clearErrors();
            setButtonLoading(resetPasswordBtn, true);

            try {
                const response = await fetch('/password-reset/reset-password', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': 'HWz5RgrMbav6Ad9PcIYT94jR8HZDvsxmwckYY0Nz',
                        'Accept': 'application/json'
                    },
                    body: JSON.stringify({
                        password,
                        password_confirmation: passwordConfirmation
                    })
                });

                const data = await response.json();

                if (data.success) {
                    showStep('success');
                } else {
                    showError(data.error || 'Произошла ошибка');
                    if (data.session_expired) {
                        setTimeout(() => showStep('email'), 2000);
                    }
                }
            } catch (error) {
                showError('Ошибка соединения. Попробуйте позже.');
            } finally {
                setButtonLoading(resetPasswordBtn, false);
            }
        });

        resendCodeBtn.addEventListener('click', async () => {
            if (resendCodeBtn.disabled) return;

            clearErrors();
            setButtonLoading(resendCodeBtn, true);

            try {
                const response = await fetch('/password-reset/resend-code', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRF-TOKEN': 'HWz5RgrMbav6Ad9PcIYT94jR8HZDvsxmwckYY0Nz',
                        'Accept': 'application/json'
                    }
                });

                const data = await response.json();

                if (data.success) {
                    resetCodeInput.value = '';
                    startResendTimer(60);
                } else {
                    showError(data.error || 'Не удалось отправить код');
                    if (data.retry_after) {
                        startResendTimer(data.retry_after);
                    }
                    if (data.session_expired) {
                        setTimeout(() => showStep('email'), 2000);
                    }
                }
            } catch (error) {
                showError('Ошибка соединения. Попробуйте позже.');
            } finally {
                setButtonLoading(resendCodeBtn, false);
            }
        });

        backToLoginBtn.addEventListener('click', closeModal);

        backToEmailBtn.addEventListener('click', () => {
            clearResendTimer();
            showStep('email');
        });

        resetCodeInput.addEventListener('input', (e) => {
            e.target.value = e.target.value.replace(/[^0-9]/g, '');
        });

        resetEmailInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') sendCodeBtn.click();
        });

        resetCodeInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') verifyCodeBtn.click();
        });

        confirmPasswordInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') resetPasswordBtn.click();
        });

        const toggleNewPassword = document.getElementById('toggleNewPassword');
        const eyeOpenNew = document.getElementById('eyeOpenNew');
        const eyeClosedNew = document.getElementById('eyeClosedNew');

        toggleNewPassword.addEventListener('click', () => {
            const type = newPasswordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            newPasswordInput.setAttribute('type', type);
            eyeOpenNew.classList.toggle('hidden');
            eyeClosedNew.classList.toggle('hidden');
        });

        const toggleConfirmPassword = document.getElementById('toggleConfirmPassword');
        const eyeOpenConfirm = document.getElementById('eyeOpenConfirm');
        const eyeClosedConfirm = document.getElementById('eyeClosedConfirm');

        toggleConfirmPassword.addEventListener('click', () => {
            const type = confirmPasswordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            confirmPasswordInput.setAttribute('type', type);
            eyeOpenConfirm.classList.toggle('hidden');
            eyeClosedConfirm.classList.toggle('hidden');
        });
    </script>

</body>
