<script>
const form = document.getElementById('loginForm');
const error = document.getElementById('errorMessage');
const box = document.getElementById('loginBox');

form.addEventListener('submit', e => {
    e.preventDefault();
    error.classList.add('show');
    box.classList.remove('shake');
    void box.offsetWidth;
    box.classList.add('shake');
});

const forgotLink = document.getElementById('forgotText');
const modal = document.getElementById('forgotModal');
const closeModal = document.getElementById('closeModal');
const sendEmail = document.getElementById('sendEmail');
const modalError = document.getElementById('modalError');
const forgotEmail = document.getElementById('forgotEmail');

forgotLink.addEventListener('click', e=>{
    e.preventDefault();
    forgotEmail.value='';
    modalError.style.display='none';
    modal.style.display='flex';
    updateLanguage(document.getElementById('languageSelect').value);
});

closeModal.addEventListener('click', ()=>modal.style.display='none');
document.querySelector('.modal-overlay').addEventListener('click', ()=>modal.style.display='none');

function isValidEmail(email){return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);}
sendEmail.addEventListener('click', ()=>{
    const email = forgotEmail.value.trim();
    if(!isValidEmail(email)){modalError.style.display='block';return;}
    modalError.style.display='none';
    alert('Password reset link sent to '+email);
    modal.style.display='none';
});

/* Show/Hide password */
const togglePassword = document.getElementById('togglePassword');
const passwordInput = document.getElementById('password');
const eyeOpen = document.getElementById('eyeOpen');
const eyeClosed = document.getElementById('eyeClosed');

togglePassword.addEventListener('click', ()=>{
    if(passwordInput.type==='password'){
        passwordInput.type='text';
        eyeOpen.style.display='none';
        eyeClosed.style.display='block';
    }else{
        passwordInput.type='password';
        eyeOpen.style.display='block';
        eyeClosed.style.display='none';
    }
});
