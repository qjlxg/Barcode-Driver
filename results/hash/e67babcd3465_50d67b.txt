            ctx.fillStyle   = 'hsl(' + this.hue + ',100%,68%)';
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.r, 0, Math.PI * 2);
            ctx.fill();
            ctx.restore();
        };

        function loop() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            pool.forEach(function (p) { p.tick(); p.draw(); });
            requestAnimationFrame(loop);
        }
