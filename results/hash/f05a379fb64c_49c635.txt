<script>
(function() {
  if (window._heroInterval) { clearInterval(window._heroInterval); window._heroInterval = null; }

  var slides = [
    { text: 'безопасному интернету', grad: 'linear-gradient(135deg,#6366f1,#8b5cf6)', phone: 0 },
    { text: 'Gemini, ChatGPT и ИИ',  grad: 'linear-gradient(135deg,#0ea5e9,#6366f1)', phone: 1 },
    { text: 'YouTube без рекламы',    grad: 'linear-gradient(135deg,#ef4444,#f97316)', phone: 2 },
  ];

  var idx = 0;
  var el = document.getElementById('hero-word');
  var phones = [0,1,2].map(function(i){ return document.getElementById('hero-phone-'+i); });
  if (!el) return;

  el.style.backgroundImage = slides[0].grad;

  function showPhone(i) {
    phones.forEach(function(p, n) {
      if (!p) return;
      p.style.opacity = (n === i) ? '1' : '0';
      p.style.transform = (n === i) ? 'translateY(0)' : 'translateY(20px)';
    });
  }

  window._heroInterval = setInterval(function() {
    el.style.opacity = '0';
    el.style.transform = 'translateY(-10px)';
    setTimeout(function() {
      idx = (idx + 1) % slides.length;
      var s = slides[idx];
      el.textContent = s.text;
      el.style.backgroundImage = s.grad;
      el.style.transition = 'none';
      el.style.transform = 'translateY(10px)';
      el.style.opacity = '0';
      showPhone(s.phone);
      requestAnimationFrame(function() {
        requestAnimationFrame(function() {
          el.style.transition = 'opacity .45s ease,transform .45s ease';
          el.style.opacity = '1';
          el.style.transform = 'translateY(0)';
        });
      });
    }, 360);
  }, 3800);
})();
</script>

function installPWA() {
  if (_deferredPrompt) {
    _deferredPrompt.prompt();
    _deferredPrompt.userChoice.then(function() { _deferredPrompt = null; dismissPWAHint(); });
  }
}
