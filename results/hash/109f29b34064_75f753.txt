(function(){
'use strict';

const params = new URLSearchParams(location.search);
const CONFIG = {
  debug: false,  // 有怀疑时手动改 true 即可展开在页面上看调试信息
  eventApi: 'https://promo.adreflux.com/events',
  linkId: location.pathname.replace(/\//g,'') || '49837044',
  pid: params.get('pid') || 'metaweb_int',
  c: params.get('c') || 'secure_dc_3',
};

const SESSION_ID_KEY = 'deepw2a_sid';
let _sid = sessionStorage.getItem(SESSION_ID_KEY);
if (!_sid) { _sid = Date.now().toString(36)+'_'+Math.random().toString(36).slice(2,8); sessionStorage.setItem(SESSION_ID_KEY, _sid); }

const USER_ID_KEY = 'deepw2a_uid';
let _uid = localStorage.getItem(USER_ID_KEY);
if (!_uid) { _uid = 'u_'+Date.now().toString(36); localStorage.setItem(USER_ID_KEY, _uid); }

const STATE = {
  uuid: Date.now().toString(36)+'_'+Math.random().toString(36).slice(2,10),  // 每次加载新 UUID
  sessionId: _sid,  // 整个访问周期不变，串起完整漏斗
  userId: _uid,     // 永久保留，识别独立设备（多次访问可关联）
  stage: 'middle_page',  // 竞品永远先显示中间页（落地页）
  pageEnterTime: Date.now(),
  exitPopupShown: false,
  initData: null, feedData: null, storeData: null,
  isBackFlow: false,     // 是否已切到回流页
};

// 竞品的 localStorage key（按 link_id 分桶累计访问）
const LS_KEY = 'sp_returning_data';
const log = (...a) => { if(CONFIG.debug) console.log('[DeepW2A]',...a); };

function getRecord() {
  const all = JSON.parse(localStorage.getItem(LS_KEY) || '{}');
  if (!all[CONFIG.linkId]) {
    all[CONFIG.linkId] = {
      setting: null,
      record: { middle_page_view:0, impression:0, click_banner:0, firstVisit:Date.now(), lastVisit:Date.now() }
    };
  }
  return { all, bucket: all[CONFIG.linkId] };
}
function bumpRecord(field) {
  const { all, bucket } = getRecord();
  bucket.record[field] = (bucket.record[field] || 0) + 1;
  bucket.record.lastVisit = Date.now();
  localStorage.setItem(LS_KEY, JSON.stringify(all));
  return bucket.record;
}

async function api(method, url, body) {
  try {
    const r = await fetch(url, { method, headers:{'Content-Type':'application/json'}, body: body?JSON.stringify(body):undefined });
    return r.json();
  } catch(e) { log('API err:',url,e.message); return null; }
}

function postEvent(code, data={}) {
  const now = Date.now();
  const obj = { event:code, stage:STATE.stage, uuid:STATE.uuid, sid:STATE.sessionId, uid:STATE.userId, link_id:CONFIG.linkId, pid:CONFIG.pid, channel_id:'4', data, page_time:now-STATE.pageEnterTime };
  // 详细日志：事件名 + 参数
  console.log('%c[Event] '+code, 'color:#22d3ee;font-weight:bold', { stage:STATE.stage, ...data });
  api('POST', CONFIG.eventApi, { id:now, name:code, ts:new Date(now).toISOString().replace('T',' ').slice(0,19), payload:JSON.stringify(obj) });
}

function isReturnVisit() {
  // 老客户判定：按竞品逻辑，累计中间页浏览次数 ≥ threshold（默认5）即认为是老客户
  const { bucket } = getRecord();
  const setting = bucket.setting || {};
  const eventType = setting.event_type || 'middle_page_view';
  const threshold = setting.event_threshold || 5;
  return (bucket.record[eventType] || 0) >= threshold;
}

async function loadData() {
  try {
    const [initResp, feedResp, storeResp, complaintResp] = await Promise.all([
      api('POST', '/api/server/mobback/init', { mch_id:'2549184', channel_id:'4', link_id:CONFIG.linkId, link_type:'app', mbclid:'', uuid:STATE.uuid }),
      api('POST', '/api/server/mobback/feed', { mch_id:'2549184', channel_id:'4', link_id:CONFIG.linkId, link_type:'app', mbclid:'', uuid:STATE.uuid, ad_template_type:2 }),
      api('POST', '/api/server/store/info', { mch_id:'2549184', channel_id:'4', link_id:CONFIG.linkId, link_type:'app', mbclid:'', creative_id:'', landing_code:'', returning_code:'' }),
      api('POST', '/api/server/links_complaint/show', { link_id:CONFIG.linkId, uuid:STATE.uuid }),
    ]);
    STATE.initData = initResp;
    STATE.feedData = feedResp;
    STATE.storeData = storeResp;
    STATE.complaintShow = complaintResp?.data?.complaint_show === true;
    log('data loaded', !!initResp, !!feedResp, !!storeResp, 'complaint:', STATE.complaintShow);
  } catch(e) { log('loadData error:', e.message); }
}

function $(id) { return document.getElementById(id); }

function renderPage() {
  try {
    const store = (STATE.storeData?.data?.appstore_info) || {};
    document.title = store.title || 'App';

    // Header
    if (store.icon) $('app-icon').src = store.icon;
    if (store.title) $('app-title').textContent = store.title;
    if (store.developer) $('app-dev').textContent = store.developer;
    if (store.developer) $('dev-name').textContent = store.developer;
    if (store.developer) $('info-seller').textContent = store.developer;

    // Stats
    const score = store.score || 0;
    const ratings = store.ratings || 0;
    $('rating-mark').textContent = ratings + ' Mark';
    $('rating-value').textContent = score.toFixed(1);
    $('score-num').textContent = score.toFixed(1);
    $('ratings-count').textContent = ratings + ' Ratings';
    if (store.content_rating) {
      $('age-value').textContent = store.content_rating;
      $('info-age').textContent = store.content_rating;
    }
    if (store.primary_genre) {
      $('rank-genre').textContent = store.primary_genre;
      $('info-genre').textContent = store.primary_genre;
    }

    // Stars (4 stars filled)
    const filled = Math.round(score);
    let starsHtml = '';
    for (let i = 0; i < filled; i++) starsHtml += '<img class="page-content__item-text-star" src="/static/images/ios/star-top.png" alt="">';
    $('rating-stars').innerHTML = starsHtml;

    // Screenshots
    const ssDiv = $('screenshots');
    ssDiv.innerHTML = '';
    const screenshots = store.screenshots || [];
    screenshots.forEach(url => {
      ssDiv.insertAdjacentHTML('beforeend',
        '<div class="page-preview__item"><img class="page-preview__item-img" src="'+url+'" loading="lazy"></div>');
    });

    // Description
    const desc = (store.description || '').replace(/\\n/g,'\n');
    $('app-desc').textContent = desc;

    // Release notes
    if (store.version) $('version-title').textContent = 'Version ' + store.version;
    if (store.release_notes) $('release-notes').textContent = store.release_notes;

    // Score histogram (5 bars from 5★ down to 1★)
    const histogram = store.histogram || {};
    const total = ratings || 1;
    const scoreBars = $('score-bars');
    scoreBars.innerHTML = '';
    for (let s = 5; s >= 1; s--) {
      const cnt = histogram[s] || 0;
      const pct = (cnt / total * 100).toFixed(0);
      let starsImgs = '';
      for (let i = 0; i < s; i++) starsImgs += '<img src="/static/images/ios/star-comment.png" alt="Star">';
      scoreBars.insertAdjacentHTML('beforeend',
        '<div class="page-score__star-item"><div class="stars">'+starsImgs+'</div>' +
        '<div class="progress-bar"><div class="progress-bar__track"><div class="progress-bar__fill" style="width:'+pct+'%"></div></div></div></div>');
    }

    // Size & OS
    if (store.size) {
      const mb = (parseInt(store.size) / 1048576).toFixed(2);
      $('info-size').textContent = mb + 'MB';
    }
    if (store.required_os_version) $('info-ios').textContent = 'Requires iOS ' + store.required_os_version + ' or later.';
    if (store.languages) $('info-langs').textContent = store.languages.join(', ');

    // Show page
    $('loading').style.display = 'none';
    $('page').style.display = 'block';
    $('page-footer').style.display = 'block';
  } catch(e) {
    log('render error:', e.message);
    $('loading').textContent = 'Error: ' + e.message;
  }
}

function getJumpUrl() {
  const linkInfo = (STATE.initData?.data?.link_info) || {};
  let url = linkInfo.jump_url;
  if (!url) {
    return /iPhone/.test(navigator.userAgent) ? 'https://apps.apple.com/app/id6757147549' : 'market://details?id=com.example';
  }
  url = url.replace(/\{\{campaign\.id\}\}/g, params.get('af_c_id')||'0')
           .replace(/\{\{adset\.name\}\}/g, params.get('af_adset')||'')
           .replace(/\{\{adset\.id\}\}/g, params.get('af_adset_id')||'0')
           .replace(/\{\{ad\.name\}\}/g, params.get('af_ad')||'')
           .replace(/\{\{ad\.id\}\}/g, params.get('af_ad_id')||'0');
  if (url.includes('onelink.me') && !url.includes('af_sub3=')) {
    url += '&af_sub3='+STATE.uuid+'&af_js_web=true&af_ss_ver=2_10_0';
  }
  return url;
}

function doJump() {
  const url = getJumpUrl();
  STATE.hasJumped = true;
  bumpRecord('click_banner');
  // 详细日志：跳转的真实 URL
  console.log('%c[Jump] → '+url, 'color:#fbbf24;font-weight:bold;font-size:13px');
  console.log('  source:', STATE.isBackFlow ? 'backflow_banner' : 'middle_page_install');
  console.log('  hasJumped:', STATE.hasJumped);
  console.log('  isBackFlow:', STATE.isBackFlow);
  postEvent('jump_link', { source: STATE.isBackFlow?'backflow_banner':'middle_page_install', target:url, has_jumped:STATE.hasJumped });
  // 跳转后 600ms 切到回流页
  setTimeout(() => switchToBackFlow('after_jump'), 600);
  window.location.href = url;
}

// ===== 阶段1: middle_page（中间页 / 落地页） =====
async function runMiddlePage() {
  STATE.stage = 'middle_page';
  log('阶段1: middle_page');

  await loadData();
  renderPage();

  // 累计访问
  bumpRecord('middle_page_view');
  bumpRecord('impression');

  postEvent('middle_page_view', { platform:/Android/.test(navigator.userAgent)?'android':'ios' });

  // ❌ 不自动跳转。等用户点 CTA。
  // 监听 visibilitychange + popstate（用户从商店/其他 App 回来 / 按返回键 → 切到回流页）
  enableReturnTriggers();
}

// ===== 阶段2: back-flow（回流页 — Facebook pure-video-flow） =====
function switchToBackFlow(reason) {
  if (STATE.isBackFlow) return;
  STATE.isBackFlow = true;
  STATE.stage = 'back_flow';
  log('阶段2: back-flow, reason=' + reason);

  postEvent('back_flow_entry', { reason });

  // 用 feed 数据填充
  const item = STATE.feedData?.data?.items?.[0];
  const ad = item?.ad || item || {};
  const asset = ad.asset || {};
  const store = STATE.storeData?.data?.appstore_info || {};

  // 视频源 + 封面图（视频加载前用封面填补黑屏）
  const video = $('bf-video');
  const poster = $('bf-poster');
  const coverUrl = asset.cover_url || '';
  const videoUrl = asset.video_url || coverUrl;
  if (coverUrl) poster.src = coverUrl;
  if (videoUrl) video.src = videoUrl;
  // 视频开始播放后隐藏封面
  video.addEventListener('playing', () => { poster.style.display = 'none'; });
  video.addEventListener('ended', () => { poster.style.display = 'block'; });

  // CTA 文案
  $('bf-cta').textContent = (asset.action_btn || 'CLEAN FREE').toUpperCase();

  // 投诉表单数据
  $('cf-app-logo').src = store.icon || asset.icon_url || '';
  $('cf-app-title').textContent = store.title || ad.ad_app_name || 'App';

  // 投诉理由列表（竞品的11个固定选项）
  const reasons = [
    'Violence, Bullying, or Criminal exploitation','Hate Speech or Harassment',
    'Dangerous Activities or Challenges','Sexual Content or Nudity',
    'Graphic Violence or Gore','Adult Content','False Information',
    'Spam or Fake Engagement','Fraud or Scam','Sharing Personal Information',
    'Counterfeit or Copyright Violation','Other'
  ];
  const reasonsBox = $('cf-reasons');
  reasonsBox.innerHTML = reasons.map((r,i) =>
    '<label style="display:flex;align-items:center;gap:8px;font-size:14px;cursor:pointer">' +
    '<input type="radio" name="cf-reason" value="'+i+'" style="width:18px;height:18px">' +
    '<span>'+r+'</span></label>'
  ).join('');

  // 绑定事件
  $('bf-cta').onclick = doJump;

  // 视频区点击切换播放/暂停
  video.onclick = () => { video.paused ? video.play() : video.pause(); };

  // 静音按钮：竞品放在视频正中央，默认静音，点一次取消静音
  const muteBtn = $('bf-mute-btn');
  if (muteBtn) {
    muteBtn.onclick = (e) => {
      e.stopPropagation();
      video.muted = false;
      video.play();
      muteBtn.classList.add('active');
    };
  }

  // m3u8 HLS 格式：iOS Safari 原生支持，桌面加 HLS.js fallback
  if (videoUrl && videoUrl.endsWith('.m3u8') && !video.canPlayType('application/vnd.apple.mpegurl')) {
    // 动态加载 HLS.js（仅桌面端非 Safari）
    const hlsScript = document.createElement('script');
    hlsScript.src = 'https://cdn.jsdelivr.net/npm/hls.js@1';
    hlsScript.onload = () => {
      if (Hls.isSupported()) {
        const hls = new Hls();
        hls.loadSource(videoUrl);
        hls.attachMedia(video);
        video.play().catch(() => {});
      }
    };
    document.head.appendChild(hlsScript);
  } else if (videoUrl) {
    video.play().catch(() => {});
  }

  // 投诉入口：后端控制 >2次访问才显示
  if (STATE.complaintShow) {
    $('bf-complaint').style.display = 'block';
    $('bf-complaint-trigger').onclick = openComplaintForm;
  }
  // 表单关闭/取消
  $('cf-close').onclick = closeComplaintForm;
  $('cf-cancel').onclick = closeComplaintForm;
  // 表单提交
  $('cf-submit').onclick = submitComplaint;
  // 输入字数统计
  $('cf-detail').oninput = (e) => { $('cf-detail-count').textContent = e.target.value.length; updateSubmitState(); };
  $('cf-email').oninput = updateSubmitState;
  document.querySelectorAll('input[name="cf-reason"]').forEach(r => r.onchange = updateSubmitState);

  // 整页替换 — 所有尺寸在 display 之前设好，避免首帧错误尺寸闪屏
  $('page').style.display = 'none';
  $('page-footer').style.display = 'none';
  document.body.classList.add('backflow-active');
  const bf = $('backflow-page');
  // 逐层设宽高（必须在 display:flex 之前，确保首帧就是正确尺寸）
  bf.style.width = window.innerWidth + 'px';
  bf.style.height = window.innerHeight + 'px';
  const swiper = bf.querySelector('.swiper');
  const wrapper = bf.querySelector('.swiper-wrapper');
  const slide = bf.querySelector('.swiper-slide');
  if (swiper) swiper.style.width = window.innerWidth + 'px';
  if (wrapper) wrapper.style.height = '100%';
  if (slide) slide.style.height = '100%';
  function setSwiperHeight() {
    const ch = ($('bf-complaint')?.offsetHeight || 0);
    if (swiper) swiper.style.height = (window.innerHeight - ch) + 'px';
  }
  setSwiperHeight();
  window.addEventListener('resize', setSwiperHeight);
  window.addEventListener('orientationchange', () => setTimeout(setSwiperHeight, 100));
  // todo : 尺寸就绪，展示页面
  bf.style.display = 'flex';

  // 启用离开意图弹窗（桌面端鼠标移出顶部）
  enableExitIntent();
}

function updateSubmitState() {
  const hasReason = !!document.querySelector('input[name="cf-reason"]:checked');
  const hasEmail = /\S+@\S+\.\S+/.test($('cf-email').value);
  const enabled = hasReason && hasEmail;
  $('cf-submit').style.opacity = enabled ? '1' : '.5';
  $('cf-submit').dataset.enabled = enabled ? '1' : '0';
}

function openComplaintForm() {
  postEvent('complaint_entry_click');
  $('complaint-form').style.display = 'block';
}
function closeComplaintForm() {
  postEvent('complaint_form_close');
  $('complaint-form').style.display = 'none';
}
function submitComplaint() {
  if ($('cf-submit').dataset.enabled !== '1') return;
  const reason = document.querySelector('input[name="cf-reason"]:checked')?.value;
  const detail = $('cf-detail').value;
  const email = $('cf-email').value;
  postEvent('complaint_submit', { reason, has_detail:!!detail, email });
  api('POST', '/api/server/links_complaint/show', { link_id:CONFIG.linkId, reason, detail, email });
  alert('Thank you for your feedback');
  closeComplaintForm();
}

// ===== 触发器：visibilitychange + popstate =====
function enableReturnTriggers() {
  // 触发器1: 页面切回前台 — 竞品源码：任何 visible 都进 back-flow
  document.addEventListener('visibilitychange', () => {
    if (document.visibilityState === 'visible') {
      postEvent('middle_page_visibility');
      if (!STATE.isBackFlow) switchToBackFlow('visibility_visible');
    } else {
      postEvent('middle_page_hide');
    }
  });

  // 触发器2: 浏览器返回键
  // push 一个假的 history entry 进栈，用户按返回时触发 popstate 而不是真离开
  history.pushState({deepw2a:'middle'}, '', location.href + (location.search?'&':'?') + 'sp_page=1');
  window.addEventListener('popstate', () => {
    postEvent('middle_page_back');
    if (!STATE.isBackFlow) {
      switchToBackFlow('popstate');
      // 把假 entry 再 push 一次，避免下次按返回立刻跳走
      history.pushState({deepw2a:'middle'}, '', location.href);
    }
  });
}

function enableExitIntent() {
  // 竞品只是桌面端鼠标移出顶部 → 弹挽留窗，没有返回劫持
  document.addEventListener('mouseout', e => {
    if (e.clientY <= 0 && !STATE.exitPopupShown) {
      STATE.exitPopupShown = true;
      $('exit-popup').style.display = 'flex';
      postEvent('exit_popup_shown', { trigger:'mouseout' });
    }
  });
}
