                function getCookie(name) {
                    var parts = document.cookie ? document.cookie.split(';') : [];

                        var c = parts[i].trim();

                    cl.add('theme-light');

                        cl.add('theme-dark');
                    } else {
                        cl.add('theme-light');
                    }
                }
            } catch (e) {}
        })();

<script type="text/javascript">

    // Global variables for DnD instances
    var dndDimensions = null;
    var dndMetrics = null;
    // Store current builder config to know if we can generate queries
    var currentBuilderConfig = null;

    $('#edit-sql-error-message').hide();
    $('#edit-sql-success-message').hide();

    // Mode Switching Logic
    function switchToQueryMode() {
        $('#dnd-editor-container').hide();
        $('#query-editor-container').show();
        $('#query-editor-actions').show();
        
        // Generate SQL if in simple mode
        if (currentBuilderConfig && currentBuilderConfig.isSimple) {
            generateQueryFromDnD();
            $('#query-mode-warning').show();
        }
        
        // Show "Delete custom query" button if we have a cube (allowing return to DnD)
        if (currentBuilderConfig && currentBuilderConfig.cubeId) {
            $('#delete-custom-query').show();
        } else {
            $('#delete-custom-query').hide();
        }
    }

    function switchToDnDMode() {
        $('#query-editor-container').hide();
        $('#query-editor-actions').hide();
        $('#dnd-editor-container').show();
        
        // Hide warning as we are back in sync/safe mode? 
        // Or check if valid state?
        $('#query-mode-warning').hide();

        // Mark as simple mode so that switching back to query mode triggers regeneration
        if (currentBuilderConfig) {
            currentBuilderConfig.isSimple = 1;
        }
    }

    $('#switch-to-query-editor').click(function(){
        switchToQueryMode();
    });

    function generateQueryFromDnD() {
        if (!currentBuilderConfig || !dndDimensions || !dndMetrics) return;
        
        // We need cube ID from config
         var cubeId = currentBuilderConfig.cubeId;
         var dimIds = dndDimensions.getResult();
         var metIds = dndMetrics.getResult();

         $.ajax({
            type: 'POST',
            url: '/builder/editor/generateQuery',
            data: {
                cube: cubeId,
                dimensions: dimIds,
                metrics: metIds
            },
            success: function(response) {
                if (response.sql) {
                    $('#dashboard-object-edit-sql-query').val(response.sql);
                }
            }
         });
    }

    $(document).on('click', 'button.dashboard-object-edit', function(event){
        let element = $(event.currentTarget);
        let type = element.attr('data-type');
        let uuid = element.attr('data-uuid');
        let cube = element.attr('data-cube');
        let url = "/builder/editor/editSql/REPLACETYPE/REPLACEUUID";
        url = url.replace('REPLACETYPE', type);
        url = url.replace('REPLACEUUID', uuid);
        
        if (cube) {
            url += '?cube=' + cube;
        }
        
        let isClone = element.attr('data-clone') === 'true';

        $.ajax({
            type: 'GET',
            url: url,
            success: function (data, status) {
                $('#dashboard-object-edit-sql-query').val(data['exa_query']);
                if (isClone) {
                    $('#dashboard-object-edit-table-name').val(data['table_title'] + ' - copy');
                    $('#edit-modal #dashboard-object-edit-table-save').attr('data-uuid', 'new');
                    $('#edit-modal #dashboard-object-edit-table-save').attr('data-table_id', '');
                } else {
                    $('#dashboard-object-edit-table-name').val(data['table_title']);
                    $('#edit-modal #dashboard-object-edit-table-save').attr('data-uuid', uuid);
                    $('#edit-modal #dashboard-object-edit-table-save').attr('data-table_id', data['table_id']);
                }
                
                // Populating Cube Dropdown
                var cubes = data['cubes'] || {};
                var $cubeSelect = $('#dnd-cube-selector');
                $cubeSelect.empty();
                $cubeSelect.append('<option value="">Select a Schema...</option>');
                $.each(cubes, function(cid, cname) {
                    $cubeSelect.append($('<option>', { value: cid, text: cname }));
                });
                
                // Initialize Layout Selection
                var layoutType = data['table_type'] ? data['table_type'] : 'simple';
                $('#dashboard-object-edit-table-layout').val(layoutType);
                $('#dashboard-object-edit-table-layout').trigger('change');

                // Handle Builder / Tabs
                // Handle Builder / Tabs
                currentBuilderConfig = data['builder_config'];
                // Set initial cube in selector
                if (currentBuilderConfig && currentBuilderConfig.cubeId) {
                    $cubeSelect.val(currentBuilderConfig.cubeId);
                }

                var hasDnD = (currentBuilderConfig && currentBuilderConfig.cubeId && currentBuilderConfig.html);
                
                
                // Virtual List Class for DnD Editor
                var VirtualDnDList = function(config) {
                    this.container = $(config.containerId);
                    this.list = $(config.listId);
                    this.spacer = $(config.spacerId);
                    this.search = $(config.searchId);
                    this.data = config.data || [];
                    this.itemHeight = config.itemHeight || 30; // Default LI height
                    this.itemPrefix = config.itemPrefix || '';
                    
                    this.filteredData = [];
                    
                    this.init = function() {
                        this.filter();
                        this.container.on('scroll', this.render.bind(this));
                        var that = this;
                        this.search.on('keyup', function() { that.filter(); });
                    };
                    
                    this.filter = function() {
                        var query = this.search.val().toLowerCase();
                        this.filteredData = this.data.filter(function(item) {
                            if (!query) return true;
                            return item.search.indexOf(query) !== -1;
                        });
                        
                        this.spacer.height(this.filteredData.length * this.itemHeight);
                        this.container.scrollTop(0);
                        this.render();
                    };
                    
                    this.render = function() {
                        var scrollTop = this.container.scrollTop();
                        var containerHeight = this.container.height();
                        
                        var startIndex = Math.floor(scrollTop / this.itemHeight);
                        var endIndex = Math.min(
                            this.filteredData.length, 
                            Math.floor((scrollTop + containerHeight) / this.itemHeight) + 10 // buffer
                        );
                        
                        var visibleItems = this.filteredData.slice(startIndex, endIndex);
                        var topOffset = startIndex * this.itemHeight;
                        
                        var html = '';
                        for (var i = 0; i < visibleItems.length; i++) {
                            var item = visibleItems[i];
                            // Reconstruct LI. DragAndDrop2 expects ID.
                            html += '<li id="avail_' + item.id + '" style="height: ' + this.itemHeight + 'px; overflow: hidden; white-space: nowrap; text-overflow: ellipsis; padding: 0 5px; line-height: ' + this.itemHeight + 'px;">' +
                                    item.name + 
                                    '</li>';
                        }
                        
                        this.list.css('top', topOffset + 'px').html(html);
                    };
                };

                // Start: DnD and Virtual List Initialization Helper
                function initDnDInterface() {
                     // Check if data is available (provided by fragment script)
                     if (typeof window.dndFragmentData === 'undefined') return;
                     
                     // Initialize Virtual Lists
                     var vDim = new VirtualDnDList({
                         containerId: '#dim-virtual-container',
                         listId: '#builder-possibleDimensions',
                         spacerId: '#dim-scroll-spacer',
                         searchId: '#dimensionsSearchbar',
                         data: window.dndFragmentData.dimensions,
                         itemPrefix: 'dim'
                     });
                     vDim.init();
                     
                     var vMet = new VirtualDnDList({
                         containerId: '#met-virtual-container',
                         listId: '#builder-possibleMetrics',
                         spacerId: '#met-scroll-spacer',
                         searchId: '#metricsSearchbar',
                         data: window.dndFragmentData.metrics,
                         itemPrefix: 'met'
                     });
                     vMet.init();
                     
                     // Initialize DragAndDrop (SortableJS)
                     if (dndDimensions) { try { dndDimensions.destroy(); } catch(e){} }
                     dndDimensions = new DragAndDrop({
                        dragId: 'builder-possibleDimensions',
                        dropId: 'builder-dimensions',
                        connectionClass: 'connectedDimensionsSortable',
                        searchBarId: 'dimensionsSearchbar',
                        onChange: function() { updateSaveButton(); }
                     });
                     dndDimensions.initialize();
            
                     if (dndMetrics) { try { dndMetrics.destroy(); } catch(e){} }
                     dndMetrics = new DragAndDrop({
                        dragId: 'builder-possibleMetrics',
                        dropId: 'builder-metrics',
                        connectionClass: 'connectedMetricsSortable',
                        searchBarId: 'metricsSearchbar',
                        onChange: function() { updateSaveButton(); }
                     });
                     dndMetrics.initialize();
                }

                // Load SortableJS library ensuring correct version
                function loadSortableLib(callback) {
                    $.ajax({
                        url: '/bundles/bibuilder/js/Sortable.min.js?v=20260610b',
                        dataType: 'script',
                        cache: true,
                        success: function() {
                            if (callback) callback();
                        }
                    });
                }

                // Function to load DnD fragment
                function loadDnDFragment(cubeId) {
                    // Remove instruction alert if present
                    $('#dnd-editor-container > .alert-info').remove();
                    
                    $('#dnd-content-area').html('<div class="alert alert-info"><i class="fa fa-spinner fa-spin"></i> Loading schema...</div>');
                    
                    $.ajax({
                        url: '/builder/editor/dndFragment',
                        data: { cubeId: cubeId },
                        success: function(html) {
                            if (!html) {
                                $('#dnd-content-area').html('<div class="alert alert-warning">Failed to load schema definition.</div>');
                                return;
                            }
                            $('#dnd-content-area').html(html);
                            
                            loadSortableLib(function() {
                                initDnDInterface();
                            });
                            
                            // Update current config tracking (at least the cube part)

    // Helper to escape HTML in dynamic content
    function wdEscapeHtml(text) {
        return String(text)
            .replace(/&/g, '&amp;')
            .replace(/</g, '&lt;')
            .replace(/>/g, '&gt;')
            .replace(/"/g, '&quot;')
            .replace(/'/g, '&#039;');
    }

    // Decide default transformer based on sql_name and type
    function wdComputeDefaultTransformer(sqlName, type) {
        var lname = String(sqlName || '').toLowerCase();
        var t = String(type || '').toLowerCase();
        if (lname.indexOf('%') !== -1) return 'round_percentage';
        var moneyHints = ['revenue','salary','sales','cogs','ek','profit'];
        var containsMoneyHint = moneyHints.some(function(h){ return lname.indexOf(h) !== -1; });
        if (t === 'float' && containsMoneyHint) return 'round_cur';
        if (t === 'float') return 'round';
        if (t === 'string') return 'string';
        return '';
    }

    function wdBuildHeatmapPreview(columns, data) {
        if (!columns || columns.length < 4) {
             return '<div style="padding: 20px; color: red;">Heatmap requires exactly 4 columns: X, Y, KPI, Color code. This table has ' + (columns ? columns.length : 0) + ' columns.</div>';
        }
        
        var x_col = columns[0].sql_name;
        var y_col = columns[1].sql_name;
        var kpi_col = columns[2].sql_name;
        var color_col = columns[3].sql_name;
        
        var x_values = [];
        var y_values = [];
        var matrix = {};
        
        data.forEach(function(row) {
            var x = row[x_col];
            var y = row[y_col];
            var kpi = row[kpi_col];
            var color = row[color_col];
            
            if (x_values.indexOf(x) === -1) x_values.push(x);
            if (y_values.indexOf(y) === -1) y_values.push(y);
            
            var y_key = '_' + y;
            var x_key = '_' + x;
            
            if (!matrix[y_key]) matrix[y_key] = {};
            matrix[y_key][x_key] = { kpi: kpi, color: color };
        });
        
        var html = '<table class="ExcelTable2007 heatmap-table" style="width: 100%; border-collapse: collapse;">';
        html += '<thead><tr>';
        html += '<th class="heading key" style="text-align: left; padding: 8px; border: 1px solid #ddd;">' + wdEscapeHtml(columns[1].name || columns[1].sql_name) + ' \\ ' + wdEscapeHtml(columns[0].name || columns[0].sql_name) + '</th>';
        
        x_values.forEach(function(x) {
            html += '<th style="padding: 8px; border: 1px solid #ddd; text-align: center;">' + wdEscapeHtml(x !== null && x !== undefined ? x : 'NULL') + '</th>';
        });
        html += '</tr></thead><tbody>';
        
        y_values.forEach(function(y) {
            var y_key = '_' + y;
            html += '<tr>';
            html += '<td class="key heading" style="font-weight: bold; padding: 8px; border: 1px solid #ddd; text-align: left;">' + wdEscapeHtml(y !== null && y !== undefined ? y : 'NULL') + '</td>';
            
            x_values.forEach(function(x) {
                var x_key = '_' + x;
                var cellData = matrix[y_key][x_key] || { kpi: '-', color: '#ffffff' };
                html += '<td style="background-color: ' + wdEscapeHtml(cellData.color) + '; padding: 8px; border: 1px solid #eee; text-align: right;">';
                html += wdEscapeHtml(cellData.kpi !== null && cellData.kpi !== undefined ? cellData.kpi : 'NULL');
                html += '</td>';
            });
            html += '</tr>';
        });
        
        html += '</tbody></table>';
        return html;
    }

                    // Apply visual formatting to preview (this mimics Twig extension behavior)
                    var fmt = (col.transformer && col.transformer.length) ? col.transformer : '';
                    val = wdFormatPreviewValue(val, fmt);
                    
                    table += '<td>' + wdEscapeHtml(val) + '</td>';
                });
                table += '</tr>';
            });
        }
        table += '</tbody></table></form>';
        return table;
    }
    
    function wdFormatPreviewValue(val, transformer) {
        if (val === 'NULL' || val === null || val === undefined) return val;
        
        // Simple numeric check
        var num = parseFloat(val);
        var isNum = !isNaN(num) && isFinite(val);
        
        if (!isNum) return val; // if not valid number, return as is (string)
        
        // Helpers
        var doRound = function(v) { 
             // Mimic round-float-with-precision-2
             return v.toFixed(2); // or Math.round(v * 100) / 100? toFixed matches standard business display.
        };
        
        // Transformers map:
        // 'round' -> round 2 decimals
        // 'cur' -> append currency (default € for preview)
        // 'percentage' -> append %
        // 'round_cur' -> round + cur
        // 'round_percentage' -> round + %
        
        // Check for composition first (underscore)
        var parts = transformer.split('_');
        // Common cases seen in dropdown
        
        if (transformer === 'string') return val;
        
        // Formatting Logic
        var formatted = num;
        var suffix = '';
        var prefix = '';
        
        // Check if rounding needed
        if (transformer.indexOf('round') !== -1) {
            formatted = doRound(num);
        } else {
            // Note: some transformers like 'cur' might imply rounding or not?
            // If backend sends 'cur', it usually means raw+cur. 
            // But let's stick to what the transformer name says.
        }
        
        // Check for currency
        if (transformer.indexOf('cur') !== -1) {
             // Basic Currency Symbol (User can adjust in real dashboard, here just preview)
             // We use € as default or generic currency sign.
             // If we want to be smarter, we could assume '€'.
             suffix = ' €'; 
        }
        
        // Check for percentage
        if (transformer.indexOf('percentage') !== -1) {
             suffix = ' %';
        }

        return prefix + formatted + suffix;
    }


    // Read current columns from the form without changing them
    function wdReadColumnsFromForm() {
        var colsByKey = {};
        var order = [];
        // Read inputs and selects under the edit columns form
        $('#query-preview-table-container form#edit_columns_form').find('input, select').each(function() {
            var name = this.name || '';
            var match = name.match(/^columns\[(.+?)\]\[(.+?)\]$/);
            if (!match) return; // skip non-column inputs
            var key = match[1]; // e.g., new-1
            var field = match[2]; // e.g., sql_name, name, detected_type, transformer
            if (!colsByKey[key]) { colsByKey[key] = {}; order.push(key); }
            var val = (this.tagName.toLowerCase() === 'select') ? $(this).val() : $(this).val();
            colsByKey[key][field] = val;
        });
        // Build ordered array
        // Sort order by the numeric part of new-X if possible
        order.sort(function(a,b){
            var na = parseInt(String(a).replace(/[^0-9]/g,''), 10) || 0;
            var nb = parseInt(String(b).replace(/[^0-9]/g,''), 10) || 0;
            return na - nb;
        });
        var cols = [];
        order.forEach(function(k){
            var c = colsByKey[k] || {};
            cols.push({
                sql_name: c['sql_name'] || '',
                name: c['name'] || '',
                type: c['detected_type'] || c['type'] || 'string',
                transformer: c['transformer'] || '',
                background_color: c['background_color'] || '',
                text_color: c['text_color'] || '',
                percentage_bar: (c['percentage_bar'] === 'true')
            });
        });
        return cols;
    }

                        return {

                            background_color: col.background_color || '',
                            text_color: col.text_color || '',

                        };
                    });

                if(!cols || cols.length === 0) {
                    $('#query-preview-table-container').html('<p>' + (response.message || 'Query returned no data.') + '</p>');
                    return;
                }
                
                window.lastPreviewData = response.data || [];
                window.lastPreviewCols = cols;
                
                var layout = $('#dashboard-object-edit-table-layout').val();
                var html = '';
                if (layout === 'heatmap') {
                    html = wdBuildHeatmapPreview(cols, window.lastPreviewData);
                } else {
                    html = wdBuildColumnsForm(cols, window.lastPreviewData);
                }
                $('#query-preview-table-container').html(html);
            },
            error: function(xhr) {
                let errorMsg = 'An unknown error occurred.';
                if (xhr.responseJSON && xhr.responseJSON.error) {
                    errorMsg = xhr.responseJSON.error;
                } else if (xhr.responseText) {
                    try {
                        let response = JSON.parse(xhr.responseText);
                        if(response.error) {
                            errorMsg = response.error;
                        }
                    } catch (e) {
                         errorMsg = xhr.responseText;
                    }
                }
                $('#query-preview-table-container').empty();
                $('#query-error').text(errorMsg).show();
            }
        });
    }

    (function(){
        function setCookie(name, value, days) {
            var expires = '';
            if (days) {
                var date = new Date();
                date.setTime(date.getTime() + (days*24*60*60*1000));
                expires = '; expires=' + date.toUTCString();
            }
            document.cookie = name + '=' + encodeURIComponent(value) + expires + '; path=/';
        }
        function getCookie(name) {
            var parts = document.cookie ? document.cookie.split(';') : [];
            for (var i = 0; i < parts.length; i++) {
                var c = parts[i].trim();
                if (c.indexOf(name + '=') === 0) {
                    return decodeURIComponent(c.substring(name.length + 1));
                }
            }
            return '';
        }
        function applyThemeClass(val){
            val = (val||'').toLowerCase();
            var cl = document.documentElement.classList;
            cl.remove('theme-light','theme-dark','theme-system');
            
            if (val === 'dark') {
                cl.add('theme-dark');
            } else if (val === 'light') {
                cl.add('theme-light');
            } else {
                // System: check OS preference
                if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
                    cl.add('theme-dark');
                } else {
                    cl.add('theme-light');
                }
            }
        }
        function getAppliedTheme(mode){
            mode = (mode||'system').toLowerCase();
            if (mode === 'light' || mode === 'dark') return mode;
            // system: use OS preference
            var isDark = false;
            try { isDark = window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches; } catch(e) {}
            return isDark ? 'dark' : 'light';
        }
        function setToggleAria(btn, currentSetting){
            if (!btn) return;
            var applied = getAppliedTheme(currentSetting);
            var next = (applied === 'dark') ? 'light' : 'dark';
            btn.setAttribute('aria-label', 'Switch to ' + next + ' mode');
            btn.setAttribute('title', 'Switch to ' + next + ' mode');
        }
        function bindToggle(btn){
            if (!btn || btn.__wdBound) return;
            btn.__wdBound = true;
            var current = getCookie('themeMode') || 'system';
            setToggleAria(btn, current);
            btn.addEventListener('click', function(){
                var cur = getCookie('themeMode') || 'system';
                var applied = getAppliedTheme(cur);
                var nxt = (applied === 'dark') ? 'light' : 'dark';
                // If user manually toggles, we save that explicit preference (overriding system)
                // Or should we just toggle between light/dark and keep it explicit?
                // The original code set cookie to 'nxt'.
                setCookie('themeMode', nxt, 365*2);
                applyThemeClass(nxt);
                setToggleAria(btn, nxt);
            });
        }
        
        // Listener for OS theme changes when in system mode
        if (window.matchMedia) {
            var mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
            function handleSystemChange(e) {
                var currentMode = getCookie('themeMode') || 'system';
                if (currentMode === 'system' || currentMode === '') {
                    applyThemeClass('system');
                }
            }
            // Modern browsers
            if (mediaQuery.addEventListener) {
                mediaQuery.addEventListener('change', handleSystemChange);
            } else if (mediaQuery.addListener) {
                // Safari < 14, older browsers
                mediaQuery.addListener(handleSystemChange);
            }
        }

        function init(){
            var btn = document.getElementById('theme-toggle');
            if (btn) { bindToggle(btn); return; }
            // If button is injected later (anonymous navbar), observe DOM and bind once it appears
            if ('MutationObserver' in window) {
                var obs = new MutationObserver(function(){
                    var b = document.getElementById('theme-toggle');
                    if (b) { bindToggle(b); obs.disconnect(); }
                });
                obs.observe(document.documentElement || document.body, {childList: true, subtree: true});
            }
        }
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', init);
        } else { init(); }
    })();
