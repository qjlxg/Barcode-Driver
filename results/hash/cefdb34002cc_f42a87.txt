<script>
const state = { sessions: [] };

function esc(s) {
  return (s || '').replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#039;'}[c]));
}

function fmtDate(ts) {
  try { return new Date(ts).toLocaleString(); } catch { return ts || ''; }
}

async function fetchJSON(url, opts) {
  const r = await fetch(url, opts);
  if (!r.ok) throw new Error('HTTP ' + r.status);
  return await r.json();
}

async function loadRepos() {
  const data = await fetchJSON('/api/repos');
  const sel = document.getElementById('repoFilter');
  const current = sel.value;
  sel.innerHTML = '';

  const all = document.createElement('option');
  all.value = '__all__';
  all.textContent = 'Alle Repos';
  sel.appendChild(all);

  for (const repo of data.repos) {
    const o = document.createElement('option');
    o.value = repo;
    o.textContent = repo;
    sel.appendChild(o);
  }

  if ([...sel.options].some(o => o.value === current)) sel.value = current;
}

async function loadSessions() {
  const repo = document.getElementById('repoFilter').value || '__all__';
  const search = document.getElementById('search').value || '';
  const query = new URLSearchParams({ repo, search });

  const data = await fetchJSON('/api/sessions?' + query.toString());
  state.sessions = data.sessions;

  const rows = document.getElementById('rows');
  rows.innerHTML = '';

  for (const s of state.sessions) {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td><input type="checkbox" class="rowCheck" data-id="${esc(s.session_id)}"></td>
      <td>${esc(fmtDate(s.timestamp))}</td>
      <td>${esc(s.thread_name || '')}</td>
      <td class="id">${esc(s.session_id)}</td>
      <td class="path">${esc(s.repo)}</td>
      <td class="path">${esc(s.cwd)}</td>
      <td><button data-del="${esc(s.session_id)}">Löschen</button></td>
    `;
    rows.appendChild(tr);
  }

  document.getElementById('status').textContent = `${state.sessions.length} Sessions`;
}

function selectedIds() {
  return [...document.querySelectorAll('.rowCheck:checked')].map(x => x.dataset.id);
}

async function deleteIds(ids) {
  if (!ids.length) return;
  if (!confirm(`${ids.length} Session(s) wirklich löschen?`)) return;

  const res = await fetchJSON('/api/delete', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({ ids })
  });

  document.getElementById('status').textContent = `Gelöscht: ${res.deleted.length}, Nicht gefunden: ${res.missing.length}`;
  await loadRepos();
  await loadSessions();
}

document.getElementById('repoFilter').addEventListener('change', loadSessions);
document.getElementById('search').addEventListener('input', loadSessions);
document.getElementById('reloadBtn').addEventListener('click', async () => { await loadRepos(); await loadSessions(); });
document.getElementById('deleteSelectedBtn').addEventListener('click', async () => deleteIds(selectedIds()));
document.getElementById('checkAll').addEventListener('change', (e) => {
  for (const cb of document.querySelectorAll('.rowCheck')) cb.checked = e.target.checked;
});
document.getElementById('rows').addEventListener('click', async (e) => {
  const id = e.target.getAttribute('data-del');
  if (id) await deleteIds([id]);
});

(async () => {
  await loadRepos();
  await loadSessions();
})();
