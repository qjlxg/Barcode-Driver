            if (videoModalVideo) {
                videoModalVideo.pause();
            }
            if (videoModal) videoModal.hidden = true;
        }

        if (deviceLabel) deviceLabel.textContent = info.label;
        if (downloadTitle) downloadTitle.textContent = info.title;
        if (downloadHint) downloadHint.textContent = info.hint;
        if (downloadCta) {
            downloadCta.href = info.url;
            downloadCta.textContent = 'Скачать ' + info.title;
        }
        if (downloadLink) {
            downloadLink.href = info.url;
            downloadLink.textContent = 'Скачать ' + info.title;
        }
        if (switcherTrigger) {
            switcherTrigger.addEventListener('click', openSwitcher);
        }
        if (howToBtn) {
            howToBtn.addEventListener('click', openVideoModal);
        }
        if (videoModalClose) {
            videoModalClose.addEventListener('click', closeVideoModal);
        }
        if (videoModal) {
            videoModal.addEventListener('click', (event) => {
                if (event.target === videoModal) closeVideoModal();
            });
        }
        window.addEventListener('keydown', (event) => {
            if (event.key === 'Escape') closeVideoModal();
        });
    })();
