        var lastIndex = 0;
        var match;
        while ((match = urlRe.exec(value)) !== null) {

          if (row.childNodes.length) row.appendChild(document.createElement("br"));
          appendTextWithLinks(row, item.text);
        }
        if (!row.childNodes.length) row.textContent = "Сообщение";
        messages.appendChild(row);
        messages.scrollTop = messages.scrollHeight;
        if (item.id && item.id > lastId) lastId = item.id;
      }

      function setClosed() {
        chatStatus = "closed";
        resetMessages();
        var row = document.createElement("div");
        row.className = "web-chat-message system";
        row.textContent = "Диалог закрыт.";
        messages.appendChild(row);
        input.disabled = true;
        fileInput.disabled = true;
        form.querySelector("button").disabled = true;
        newButton.hidden = false;
        setStatus("");
      }

      async function call(path, options) {
        var response = await fetch(path, Object.assign({
          credentials: "same-origin",
          headers: {"Content-Type": "application/json"}
        }, options || {}));
        if (!response.ok) throw new Error("request failed");
        return await response.json();
      }

      function safePagePath() {
        return (location.pathname || "/").replace(/\/transfer\/[^\/?#]+/i, "/transfer/[скрыт-токен]");
      }

      async function ensureSession(reset) {
        var data = await call("/chat/session", {
          method: "POST",
          body: JSON.stringify({page_path: safePagePath(), reset: !!reset})
        });
        if (data.status === "closed") {
          setClosed();
          return data;
        }
        chatStatus = data.status || "open";
        input.disabled = false;
        fileInput.disabled = false;
        form.querySelector("button").disabled = false;
        newButton.hidden = true;
        resetMessages();
        (data.messages || []).forEach(renderMessage);
        return data;
      }

      async function poll() {
        if (chatStatus === "closed" || panel.hidden) return;
        try {
          var data = await fetch("/chat/messages?after_id=" + encodeURIComponent(String(lastId)), {
            credentials: "same-origin"
          }).then(function(r) { return r.json(); });
          if (data.status === "closed") {
            setClosed();
            return;
          }
          (data.messages || []).forEach(renderMessage);
        } catch (e) {}
      }

      function startPolling() {
        if (pollTimer) clearInterval(pollTimer);
        pollTimer = setInterval(poll, 4000);
      }

      function focusComposer() {
        var coarsePointer = window.matchMedia && window.matchMedia("(pointer: coarse)").matches;
        var smallScreen = window.matchMedia && window.matchMedia("(max-width: 520px)").matches;
        if (coarsePointer || smallScreen) return;
        try {
          input.focus({preventScroll: true});
        } catch (e) {
          input.focus();
        }
      }

      openButton.addEventListener("click", async function() {
        panel.hidden = false;
        focusComposer();
        setStatus("Открываем чат...");
        try {
          await ensureSession(false);
          setStatus("");
          startPolling();
          poll();
        } catch (e) {
          setStatus("Чат временно недоступен. Попробуйте позже.");
        }
      });

      hideButton.addEventListener("click", function() {
        panel.hidden = true;
      });

      document.addEventListener("click", function(event) {
        if (panel.hidden) return;
        if (!root.contains(event.target)) {
          panel.hidden = true;
        }
      });

      endButton.addEventListener("click", async function() {
        if (chatStatus === "closed") return;
        try {
          await call("/chat/close", {method: "POST", body: JSON.stringify({})});
        } catch (e) {}
        setClosed();
      });

      newButton.addEventListener("click", async function() {
        resetMessages();
        chatStatus = "new";
        setStatus("Открываем новый диалог...");
        try {
          await ensureSession(true);
          setStatus("");
          focusComposer();
          startPolling();
          poll();
        } catch (e) {
          setStatus("Чат временно недоступен. Попробуйте позже.");
        }
      });

      fileInput.addEventListener("change", function() {
        if (fileInput.files && fileInput.files[0]) {
          setStatus("Скриншот выбран: " + fileInput.files[0].name);
        }
      });

      form.addEventListener("submit", async function(event) {
        event.preventDefault();
        var text = input.value.trim();
        var file = fileInput.files && fileInput.files[0] ? fileInput.files[0] : null;
        if ((!text && !file) || chatStatus === "closed") return;
        input.value = "";
        setStatus("Отправляем...");
        try {
          var data;
          if (file) {
            var formData = new FormData();
            formData.append("text", text);
            formData.append("page_path", safePagePath());
            formData.append("screenshot", file);
            var response = await fetch("/chat/upload", {
              method: "POST",
              credentials: "same-origin",
              body: formData
            });
            if (!response.ok) throw new Error("upload failed");
            data = await response.json();
            fileInput.value = "";
          } else {
            data = await call("/chat/messages", {
              method: "POST",
              body: JSON.stringify({text: text, page_path: safePagePath()})
            });
          }
          if (data.status === "closed") {
            setClosed();
            return;
          }
          (data.messages || []).forEach(renderMessage);
          setStatus("");
          poll();
        } catch (e) {
          input.value = text;
          setStatus(file ? "Не удалось отправить скриншот. Проверьте размер и формат." : "Не удалось отправить. Попробуйте еще раз.");
        }
      });
    })();
  </script>
</body>
