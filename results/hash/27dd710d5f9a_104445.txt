        var loadErrHandler = function(e) {
            var target = e.target;
            // Error on <script> and <link>(CSS)
            // <script> will have .src and <link> .href
            var fileRef = (target.src ? target.src : target.href);
            if (("SCRIPT" === target.tagName || "LINK" === target.tagName)
                && criticalFiles.some(
                    function(file) { return fileRef.indexOf(file) !== -1 })) {
                window.onload = function() {
                    // The whole complex part below implements page reloads with
                    // "exponential backoff". The retry attempt is passes as
                    // "rCounter" query parameter
                    var href = window.location.href;

                    var retryMatch = href.match(/.+(\?|&)rCounter=(\d+)/);
                    var retryCountStr = retryMatch ? retryMatch[2] : "0";
                    var retryCount = Number.parseInt(retryCountStr);

                    if (retryMatch == null) {
                        var separator = href.indexOf("?") === -1 ? "?" : "&";
                        var hashIdx = href.indexOf("#");

                        if (hashIdx === -1) {
                            href += separator + "rCounter=1";
                        } else {
                            var hashPart = href.substr(hashIdx);

                            href = href.substr(0, hashIdx)
                                + separator + "rCounter=1" + hashPart;
                        }
                    } else {
                        var separator = retryMatch[1];

                        href = href.replace(
                            /(\?|&)rCounter=(\d+)/,
                            separator + "rCounter=" + (retryCount + 1));
                    }

                    var delay = Math.pow(2, retryCount) * 2000;
                    if (isNaN(delay) || delay < 2000 || delay > 60000)
                        delay = 10000;

(function configureIPadCameraFallback() {
    var isIPad = /iPad/.test(navigator.userAgent)
        || (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);
    var IPADOS_MODERN_MIN = 17;
    // A12 (2018) and newer handle E2EE + simulcast multiparty much more reliably

    var gpuFamily = null;
    var cores = typeof navigator.hardwareConcurrency === 'number'
        ? navigator.hardwareConcurrency
        : null;

    function readAppleGpu() {
        var canvas;
        var gl;
        var ext;
        var renderer;
        var familyMatch;

        try {
            canvas = document.createElement('canvas');
            gl = canvas.getContext('webgl') || canvas.getContext('experimental-webgl');
            if (!gl) {
                return { renderer: null, family: null };
            }
            ext = gl.getExtension('WEBGL_debug_renderer_info');
            renderer = ext
                ? gl.getParameter(ext.UNMASKED_RENDERER_WEBGL)
                : gl.getParameter(gl.RENDERER);
            familyMatch = renderer && String(renderer).match(/Apple\s+A(\d+)/i);
            return {
                renderer: renderer || null,
                family: familyMatch ? parseInt(familyMatch[1], 10) : null
            };
        } catch (error) {
            return { renderer: null, family: null };
        }
    }

    if (!isIPad) {
        window.__viomediIPadPolicy = {
            isIPad: false,
            osMajor: null,
            gpuRenderer: null,
            gpuFamily: null,
            cores: cores,
            modernMin: IPADOS_MODERN_MIN,
            gpuModernMin: APPLE_GPU_MODERN_MIN,
            mode: mode
        };

    if (match) {

    } else {

        if (match) {

        }

        mode = 'legacy-360-weak-or-unknown-gpu';
    } else if (osMajor > 0) {
        mode = 'legacy-360';
    } else {
        mode = 'legacy-360-unknown-version';
    }

    window.__viomediIPadPolicy = {
        isIPad: true,
        osMajor: osMajor || null,
        gpuRenderer: gpuRenderer,
        gpuFamily: gpuFamily,
        cores: cores,
        modernMin: IPADOS_MODERN_MIN,
        gpuModernMin: APPLE_GPU_MODERN_MIN,
        mode: mode,
        p2pAllowWithE2EE: false,
        p2pEnabled: false

    config.constraints.video = config.constraints.video || {};
    config.constraints.video.height = { ideal: 360, max: 360, min: 180 };

    config.p2p.enabled = false;
    config.p2p.allowWithE2EE = false;
