<script>
    const $ = (id) => document.getElementById(id);
    const status = $("status");

    // ── 目标平台：系统/架构/版本/协议四组单选，从已缓存壳（/api/shells）动态生成。壳由 `make deploy-shells` 上传 ──
    let SHELLS = [];
    const splitArch = (a) => { const i = a.lastIndexOf("-"); return [a.slice(0, i), a.slice(i + 1)]; };
    const uniq = (a) => [...new Set(a)].sort();
    const picked = (name) => {
        const el = document.querySelector(`input[name="${name}"]:checked`);
        return el ? el.value : "";
    };

    function fillGroup(elId, name, vals, prev) {
        const box = $(elId);
        box.innerHTML = "";
        vals.forEach((v, i) => {
            const on = prev ? v === prev : i === 0;
            const lab = document.createElement("label");
            lab.className = "opt";
            lab.innerHTML = `<input type="radio" name="${name}" value="${v}"${on ? " checked" : ""}><span>${v}</span>`;
            lab.querySelector("input").onchange = onPick;
            box.appendChild(lab);
        });
    }

    function renderPicker() {
        fillGroup("os-opts", "os", uniq(SHELLS.map(s => splitArch(s.arch)[0])), picked("os"));
        fillGroup("arch-opts", "arch", uniq(SHELLS.map(s => splitArch(s.arch)[1])), picked("arch"));
        fillGroup("ver-opts", "ver", uniq(SHELLS.map(s => s.version)), picked("ver"));
        fillGroup("proto-opts", "proto", uniq(SHELLS.map(s => s.proto || "none")), picked("proto"));
        onPick();
    }

    // 选中的 系统-架构 + 版本 + 协议 → 对应已缓存壳（可能不存在，如 macos+amd64）。
    function currentShell() {
        return SHELLS.find(s => s.arch === picked("os") + "-" + picked("arch")
            && s.version === picked("ver") && (s.proto || "none") === picked("proto"));
    }

    function onPick() {
        const s = currentShell();
        $("gen").disabled = !s;
        $("genlink").disabled = !s;
        $("dllink").innerHTML = "";
        if (!SHELLS.length) {
            status.className = "err"; status.textContent = "无缓存壳（用 make deploy-shells 上传）";
        } else if (!s) {
            status.className = "err";
            status.textContent = `该组合（${picked("os")}-${picked("arch")} · v${picked("ver")} · ${picked("proto")}）暂无壳`;
        } else {
            status.className = ""; status.textContent = "选中：" + s.name;
        }
    }

    // base64(utf8(str))：TextEncoder 出 UTF-8 字节 → binary string → btoa（btoa 不收非 latin1，故先转字节）。
    function b64utf8(str) {
        const bytes = new TextEncoder().encode(str);
        let bin = "";
        for (const b of bytes) bin += String.fromCharCode(b);
        return btoa(bin);
    }

    // 提交 {conf: base64(yaml), os, arch, ver, proto} → 后端清洗 + 注入选中壳 → 直接下载。
    async function build() {
        const yaml = $("cfg").value.trim();
        const s = currentShell();
        if (!yaml) { status.className = "err"; status.textContent = "请先填写配置"; return; }
        if (!s) return;
        const [os, arch] = splitArch(s.arch);  // s.arch="linux-amd64" → [linux, amd64]
        $("gen").disabled = true; $("genlink").disabled = true;
        $("dllink").innerHTML = "";
        status.className = ""; status.textContent = "生成中…";
        try {
            const resp = await fetch("/api/build", {
                method: "POST",
                headers: {"Content-Type": "application/json"},
                body: JSON.stringify({conf: b64utf8(yaml), os, arch, ver: s.version, proto: s.proto || "all"}),
            });
            if (!resp.ok) {
                status.className = "err";
                status.textContent = "失败：" + (await resp.text() || resp.status);
                return;
            }
            const blob = await resp.blob();
            const u = URL.createObjectURL(blob);
            const a = document.createElement("a");
            a.href = u; a.download = s.name; a.click();
            URL.revokeObjectURL(u);
            status.className = "ok"; status.textContent = "已生成，开始下载。";
        } catch (e) {
            status.className = "err"; status.textContent = "出错：" + e;
        } finally {
            onPick();
        }
    }

    $("gen").onclick = build;
    $("genlink").onclick = build;

    async function loadShells() {
        try {
            const r = await fetch("/api/shells");
            if (!r.ok) return;
            const d = await r.json();
            $("srvver").textContent = "server v" + d.server;
            SHELLS = d.shells || [];
            renderPicker();
        } catch (e) {
            /* 静默；状态走 #status */
        }
    }

    loadShells();
</script>
