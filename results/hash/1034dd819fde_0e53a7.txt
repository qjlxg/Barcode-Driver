<script>
// ── State ──────────────────────────────────────────────────────────────────
const state = {
  fileId: null,
  filename: null,
  sheets: [],
  sheet: null,
  dichotomous: false,
  orderMap: null,      // {varname: [options]} после скрейпинга
  scrapeTaskId: null,
  convertTaskId: null,
};

// ── DOM refs ───────────────────────────────────────────────────────────────
const dropZone      = document.getElementById("drop-zone");
const fileInput     = document.getElementById("file-input");
const sheetSelect   = document.getElementById("sheet-select");
const rowsSelect    = document.getElementById("rows-select");
const modeSeg       = document.getElementById("mode-seg");
const formUrl       = document.getElementById("form-url");
const btnScrape     = document.getElementById("btn-scrape");
const formStatus    = document.getElementById("form-status");
const btnConvert    = document.getElementById("btn-convert");
const tableWrap     = document.getElementById("table-wrap");
const emptyState    = document.getElementById("empty-state");
const previewTable  = document.getElementById("preview-table");
const previewHead   = document.getElementById("preview-head");
const previewBody   = document.getElementById("preview-body");
const statusDot     = document.getElementById("status-dot");
const statusText    = document.getElementById("status-text");

// ── Status helpers ─────────────────────────────────────────────────────────
function setStatus(msg, kind = "") {
  statusText.textContent = msg;
  statusDot.className = "dot" + (kind ? " " + kind : "");
}

function setFormStatus(msg, color = "var(--text-dim)") {
  formStatus.textContent = msg;
  formStatus.style.color = color;
}

// ── File upload ────────────────────────────────────────────────────────────
dropZone.addEventListener("click", () => fileInput.click());
dropZone.addEventListener("dragover", e => { e.preventDefault(); dropZone.classList.add("drag-over"); });
dropZone.addEventListener("dragleave", () => dropZone.classList.remove("drag-over"));
dropZone.addEventListener("drop", e => {
  e.preventDefault();
  dropZone.classList.remove("drag-over");
  const f = e.dataTransfer.files[0];
  if (f) uploadFile(f);
});
fileInput.addEventListener("change", () => {
  if (fileInput.files[0]) uploadFile(fileInput.files[0]);
});

async function uploadFile(file) {
  setStatus("Загружаем файл…", "busy");
  dropZone.textContent = "Загружаем…";
  dropZone.className = "";

  const fd = new FormData();
  fd.append("file", file);

  try {
    const res = await fetch("/upload", { method: "POST", body: fd });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error || "Ошибка загрузки");

    state.fileId = data.file_id;
    state.filename = data.filename;
    state.sheets = data.sheets;
    state.sheet = data.best_sheet;
    state.orderMap = null;

    dropZone.textContent = "📄 " + data.filename;
    dropZone.className = "loaded";

    sheetSelect.innerHTML = data.sheets.map(s =>
      `<option${s === data.best_sheet ? " selected" : ""}>${s}</option>`
    ).join("");
    sheetSelect.disabled = false;
    btnConvert.disabled = false;

    setStatus(`${data.filename} | листов: ${data.sheets.length}`, "ok");
    loadPreview();
  } catch (e) {
    setStatus("Ошибка: " + e.message, "err");
    dropZone.textContent = "📂 Загрузить .xlsx";
    dropZone.className = "";
  }
}

// ── Sheet / rows change ────────────────────────────────────────────────────
sheetSelect.addEventListener("change", () => {
  state.sheet = sheetSelect.value;
  loadPreview();
});
rowsSelect.addEventListener("change", loadPreview);

// ── Mode segmented ─────────────────────────────────────────────────────────
modeSeg.querySelectorAll("button").forEach(btn => {
  btn.addEventListener("click", () => {
    modeSeg.querySelectorAll("button").forEach(b => b.classList.remove("active"));
    btn.classList.add("active");
    state.dichotomous = btn.dataset.val === "dichotomous";
  });
});

// ── Preview ────────────────────────────────────────────────────────────────
async function loadPreview() {
  if (!state.fileId) return;
  const sheet = sheetSelect.value;
  const rows = rowsSelect.value;
  setStatus("Загружаем превью…", "busy");

  try {
    const res = await fetch(`/preview/${state.fileId}?sheet=${encodeURIComponent(sheet)}&rows=${rows}`);
    const data = await res.json();
    if (!res.ok) throw new Error(data.error);
    renderPreview(data.rows, data.total_rows, data.total_cols, sheet);
    setStatus(
      `${state.filename} | лист: ${sheet} | строк: ${data.total_rows - 1} | колонок: ${data.total_cols}`,
      "ok"
    );
  } catch (e) {
    setStatus("Ошибка превью: " + e.message, "err");
  }
}

function renderPreview(rows, totalRows, totalCols, sheet) {
  if (!rows || rows.length === 0) {
    emptyState.textContent = "Нет данных";
    emptyState.style.display = "flex";
    previewTable.style.display = "none";
    return;
  }

  const MAX_COLS = 50;
  const headers = rows[0].slice(0, MAX_COLS);

  previewHead.innerHTML = "<tr>" + headers.map(h => {
    const short = h.length > 60 ? h.slice(0, 60) + "…" : h;
    return `<th><div class="th-wrap">${escHtml(short)}<span class="th-tip">${escHtml(h)}</span></div></th>`;
  }).join("") + "</tr>";

  previewBody.innerHTML = rows.slice(1).map(row =>
    "<tr>" + row.slice(0, MAX_COLS).map(v => `<td>${escHtml(v)}</td>`).join("") + "</tr>"
  ).join("");

  emptyState.style.display = "none";
  previewTable.style.display = "";
}

function escHtml(s) {
  return String(s).replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;");
}

// ── Scrape ─────────────────────────────────────────────────────────────────
btnScrape.addEventListener("click", startScrape);

async function startScrape() {
  const url = formUrl.value.trim();
  if (!url) { setFormStatus("Введите ссылку на форму", "var(--red)"); return; }

  btnScrape.disabled = true;
  btnScrape.textContent = "Загружаю…";
  setFormStatus("Подключаюсь…", "var(--text-dim)");

  try {
    const res = await fetch("/scrape", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ url }),
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error);
    state.scrapeTaskId = data.task_id;
    listenScrape(data.task_id);
  } catch (e) {
    setFormStatus("Ошибка: " + e.message, "var(--red)");
    btnScrape.disabled = false;
    btnScrape.textContent = "Загрузить порядок из формы";
  }
}

function listenScrape(taskId) {
  const es = new EventSource(`/status/${taskId}`);

  es.addEventListener("status", e => {
    const d = JSON.parse(e.data);
    setFormStatus(d.message, "var(--text-dim)");
  });

  es.addEventListener("done", e => {
    es.close();
    btnScrape.disabled = false;
    btnScrape.textContent = "Загрузить порядок из формы";

    const d = JSON.parse(e.data);
    if (!d.ok) {
      if (d.error === "captcha_exhausted") {
        setFormStatus("Не удалось получить доступ к форме — Яндекс блокирует запросы. Попробуйте позже или добавьте прокси.", "var(--orange)");
      } else {
        setFormStatus("Ошибка: " + d.error, "var(--red)");
      }
      return;
    }
    const questions = d.result.questions;
    state.scrapeQuestions = questions;
    setFormStatus(`✓ Загружено ${questions.length} вопросов`, "var(--green)");
  });

  es.onerror = () => {
    es.close();
    btnScrape.disabled = false;
    btnScrape.textContent = "Загрузить порядок из формы";
    setFormStatus("Соединение прервано", "var(--red)");
  };
}

// ── Convert ────────────────────────────────────────────────────────────────
btnConvert.addEventListener("click", startConvert);

async function startConvert() {
  if (!state.fileId) return;

  btnConvert.disabled = true;
  btnConvert.textContent = "Конвертирую…";
  setStatus("Конвертация…", "busy");

  const body = {
    file_id: state.fileId,
    sheet: sheetSelect.value,
    dichotomous: state.dichotomous,
  };

  // Если есть результат скрейпинга — передаём вопросы для сопоставления
  if (state.scrapeQuestions) {
    body.scrape_questions = state.scrapeQuestions;
  }

  try {
    const res = await fetch("/convert", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(body),
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error);
    state.convertTaskId = data.task_id;
    listenConvert(data.task_id);
  } catch (e) {
    setStatus("Ошибка: " + e.message, "err");
    btnConvert.disabled = false;
    btnConvert.textContent = "▶ Конвертировать";
  }
}

function listenConvert(taskId) {
  const es = new EventSource(`/status/${taskId}`);

  es.addEventListener("status", e => {
    const d = JSON.parse(e.data);
    setStatus(d.message, "busy");
  });

  es.addEventListener("done", e => {
    es.close();
    btnConvert.disabled = false;
    btnConvert.textContent = "▶ Конвертировать";

    const d = JSON.parse(e.data);
    if (!d.ok) {
      setStatus("Ошибка: " + d.error, "err");
      return;
    }

    setStatus("Готово! Скачиваю файл…", "ok");
    const filename = (state.filename || "result").replace(/\.[^.]+$/, "") + ".sav";
    const a = document.createElement("a");
    a.href = `/download/${taskId}?filename=${encodeURIComponent(filename)}`;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    a.remove();
    setStatus(`Готово → ${filename}`, "ok");
  });

  es.onerror = () => {
    es.close();
    btnConvert.disabled = false;
    btnConvert.textContent = "▶ Конвертировать";
    setStatus("Соединение прервано", "err");
  };
}
</script>
