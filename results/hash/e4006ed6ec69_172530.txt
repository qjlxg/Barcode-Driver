    <script>
        (function() {
            // Кнопка копирования IP
            const copyBtn = document.getElementById('copyIpBtn');
            const ipText = 'play.cubecraft.ru';

            copyBtn.addEventListener('click', function() {
                navigator.clipboard.writeText(ipText).then(() => {
                    const original = copyBtn.textContent;
                    copyBtn.textContent = '✅ Скопировано!';
                    setTimeout(() => {
                        copyBtn.textContent = original;
                    }, 2000);
                }).catch(() => {
                    alert('Не удалось скопировать. IP: ' + ipText);
                });
            });

            // Кнопка "Играть сейчас" — анимация / сообщение
            const joinBtn = document.getElementById('joinBtn');
            joinBtn.addEventListener('click', function() {
                // Имитация запуска игры
                joinBtn.textContent = '⏳ Подключение...';
                joinBtn.style.opacity = '0.7';
                setTimeout(() => {
                    joinBtn.textContent = '✅ Готово!';
                    joinBtn.style.background = '#6fcf97';
                    joinBtn.style.boxShadow = '0 8px 0 #2d6b4a';
                    setTimeout(() => {
                        joinBtn.textContent = '▶  Играть сейчас';
                        joinBtn.style.background = '';
                        joinBtn.style.boxShadow = '';
                        joinBtn.style.opacity = '1';
                    }, 1500);
                }, 800);
            });

            // Динамическое обновление онлайна (имитация)
            const onlineEl = document.getElementById('onlineCount');
            if (onlineEl) {
                function updateOnline() {
                    const base = 87;
                    const variation = Math.floor(Math.random() * 30) - 10; // -10 .. +19
                    let newVal = base + variation;
                    if (newVal < 40) newVal = 40 + Math.floor(Math.random() * 20);
                    onlineEl.textContent = newVal;
                }
                // обновляем каждые 12 секунд
                setInterval(updateOnline, 12000);
                // первый раз через 1 секунду
                setTimeout(updateOnline, 1000);
            }

            // Небольшой эффект: появляющиеся карточки (для красоты)
            const cards = document.querySelectorAll('.card');
            const observer = new IntersectionObserver((entries) => {
                entries.forEach((entry, index) => {
                    if (entry.isIntersecting) {
                        entry.target.style.transition = 'all 0.5s cubic-bezier(0.2, 0.9, 0.3, 1.2)';
                        entry.target.style.opacity = '1';
                        entry.target.style.transform = 'translateY(0)';
                    }
                });
            }, { threshold: 0.2 });

            cards.forEach((card, i) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(30px)';
                observer.observe(card);
            });

            // Немедленно показываем, если они уже видны (для больших экранов)
            setTimeout(() => {
                cards.forEach(card => {
                    const rect = card.getBoundingClientRect();
                    if (rect.top < window.innerHeight - 100) {
                        card.style.opacity = '1';
                        card.style.transform = 'translateY(0)';
                    }
                });
            }, 200);

        })();
    </script>
