    <script>
    document.querySelectorAll('.recipe-header').forEach(function(header) {
        header.addEventListener('click', function() {
            var recipe = this.parentElement;
            var body = recipe.querySelector('.recipe-body');
            var inner = body.querySelector('.recipe-body-inner');

            if (recipe.classList.contains('open')) {
                body.style.height = body.scrollHeight + 'px';
                requestAnimationFrame(function() {
                    body.style.height = '0';
                });
                recipe.classList.remove('open');
            } else {
                recipe.classList.add('open');
                var h = inner.offsetHeight;
                body.style.height = '0';
                requestAnimationFrame(function() {
                    body.style.height = h + 'px';
                });
                body.addEventListener('transitionend', function handler() {
                    if (recipe.classList.contains('open')) {
                        body.style.height = 'auto';
                    }
                    body.removeEventListener('transitionend', handler);
                });
            }
        });
    });
    </script>
