      <button class="modal-close" id="modal-close" aria-label="Закрыть">✕</button>
    </div>

    <div class="section-label">Тема оформления</div>
    <div class="theme-grid" id="theme-grid"></div>
  </div>
</div>

<!-- Topics Modal -->
<div class="modal-overlay" id="topics-modal" role="dialog" aria-modal="true">
  <div class="modal topics-modal">
    <div class="modal-header">
      <div class="modal-title">📚 Топики</div>
      <button class="modal-close" id="topics-modal-close" aria-label="Закрыть">✕</button>
    </div>

    <div class="section-label">Выбери топик, чтобы отфильтровать слова</div>
    <div class="topics-list" id="topics-list"></div>
  </div>
</div>

<script>
  console.log('[Dictionary] v5 loaded');
  const tg = window.Telegram?.WebApp;

  // ── Themes ─────────────────────────────────────────────────
  const THEMES = [
    { id: 'green',  name: 'Green',  swatch: '#6aff8e', headerBg: '#0d0d0f' },
    { id: 'neon',   name: 'Neon',   swatch: 'gradient', headerBg: '#07070d' },
    { id: 'blue',   name: 'Blue',   swatch: '#3d8bff', headerBg: '#0d0d0f' },
    { id: 'pink',   name: 'Pink',   swatch: '#ff6ac1', headerBg: '#0d0d0f' },
    { id: 'yellow', name: 'Yellow', swatch: '#ffd93d', headerBg: '#0d0d0f' },
    { id: 'red',    name: 'Red',    swatch: '#ff5c5c', headerBg: '#0d0d0f' },
    { id: 'white',  name: 'Light',  swatch: '#1a1a1a', headerBg: '#f5f5f7' },
  ];
  const DEFAULT_THEME = 'green';
  const THEME_STORAGE_KEY = 'dict_theme_v1';

  function getStoredTheme() {
    try {
      const v = localStorage.getItem(THEME_STORAGE_KEY);
      if (v && THEMES.some(t => t.id === v)) return v;
    } catch (e) {}
    return DEFAULT_THEME;
  }

  function applyTheme(themeId) {
    const theme = THEMES.find(t => t.id === themeId) || THEMES[0];
    document.documentElement.setAttribute('data-theme', theme.id);

    if (tg) {
      try {
        tg.setHeaderColor(theme.headerBg);
        tg.setBackgroundColor(theme.headerBg);
      } catch (e) {}
    }

    const meta = document.querySelector('meta[name="theme-color"]');
    if (meta) meta.setAttribute('content', theme.headerBg);

    try { localStorage.setItem(THEME_STORAGE_KEY, theme.id); } catch (e) {}

    document.querySelectorAll('.theme-card').forEach(c => {
      c.classList.toggle('selected', c.dataset.themeId === theme.id);
    });

    try { tg?.HapticFeedback?.selectionChanged(); } catch (e) {}
  }

  function hexToGlow(hex) {
    const m = hex.match(/^#([0-9a-f]{6})$/i);
    if (!m) return 'transparent';
    const n = parseInt(m[1], 16);
    const r = (n >> 16) & 255, g = (n >> 8) & 255, b = n & 255;
    return `rgba(${r}, ${g}, ${b}, 0.5)`;
  }

  function buildThemeGrid() {
    const grid = document.getElementById('theme-grid');
    const current = document.documentElement.getAttribute('data-theme');
    grid.innerHTML = THEMES.map(t => {
      const isGradient = t.swatch === 'gradient';
      const swatchStyle = isGradient
        ? ''
        : `style="background:${t.swatch}; --swatch-glow: ${hexToGlow(t.swatch)}"`;
      const selected = t.id === current ? ' selected' : '';
      return `
        <div class="theme-card${selected}" data-theme-id="${t.id}" role="button" tabindex="0">
          <div class="check">✓</div>
          <div class="theme-swatch" ${swatchStyle}></div>
          <div class="theme-name">${t.name}</div>
        </div>
      `;
    }).join('');

    grid.addEventListener('click', (e) => {
      const card = e.target.closest('.theme-card');
      if (!card) return;
      applyTheme(card.dataset.themeId);
    });
  }

  // ── Settings modal ─────────────────────────────────────────
  function openSettings() {
    document.getElementById('settings-modal').classList.add('visible');
    try { tg?.HapticFeedback?.impactOccurred('light'); } catch (e) {}
  }
  function closeSettings() {
    document.getElementById('settings-modal').classList.remove('visible');
  }

  document.getElementById('settings-btn').addEventListener('click', openSettings);
  document.getElementById('modal-close').addEventListener('click', closeSettings);
  document.getElementById('settings-modal').addEventListener('click', (e) => {
    if (e.target.id === 'settings-modal') closeSettings();
  });

  // ── Topics modal ───────────────────────────────────────────
  function openTopics() {
    buildTopicsList();
    document.getElementById('topics-modal').classList.add('visible');
    try { tg?.HapticFeedback?.impactOccurred('light'); } catch (e) {}
  }
  function closeTopics() {
    document.getElementById('topics-modal').classList.remove('visible');
  }

  document.getElementById('topics-modal-close').addEventListener('click', closeTopics);
  document.getElementById('topics-modal').addEventListener('click', (e) => {
    if (e.target.id === 'topics-modal') closeTopics();
  });

  document.getElementById('topics-list').addEventListener('click', (e) => {
    const row = e.target.closest('.topic-row');
    if (!row) return;
    const filter = row.dataset.filter;
    selectTopicFilter(filter);
    try { tg?.HapticFeedback?.selectionChanged(); } catch (e) {}
    closeTopics();
  });

  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') {
      closeSettings();
      closeTopics();
    }
  });

  // ── Telegram init ──────────────────────────────────────────
  if (tg) {
    tg.ready();
    tg.expand();

    try {
      tg.BackButton.show();
      tg.BackButton.onClick(() => {
        const settingsModal = document.getElementById('settings-modal');
        const topicsModal = document.getElementById('topics-modal');
        if (topicsModal.classList.contains('visible')) {
          closeTopics();
        } else if (settingsModal.classList.contains('visible')) {
          closeSettings();
        } else {
          tg.close();
        }
      });
    } catch (e) {}

    try { tg.enableClosingConfirmation(); } catch (e) {}
  }

  // Применяем сохранённую тему ДО построения UI
  applyTheme(getStoredTheme());
  buildThemeGrid();

  // ── State / data ───────────────────────────────────────────
  const topicNames = {};
  let allWords = [];
  let currentTopicFilter = 'all';
  let currentStateFilter = 'all';
  let maxBox = 5;

  function pluralRu(n, forms) {
    const a = Math.abs(n) % 100;
    const b = a % 10;
    if (a > 10 && a < 20) return forms[2];
    if (b > 1 && b < 5) return forms[1];
    if (b === 1) return forms[0];
    return forms[2];
  }

  async function loadWords() {
    try {
      const url = '/api/words';
      const initData = tg?.initData || '';
      const devUserId = new URLSearchParams(location.search).get('user_id');
      const fetchUrl = initData
        ? url
        : (devUserId ? `${url}?user_id=${devUserId}` : url);

      const resp = await fetch(fetchUrl, {
        headers: initData ? { 'X-Telegram-Init-Data': initData } : {},
      });

      if (!resp.ok) {
        if (resp.status === 401) {
          showEmpty('Сессия истекла. Закрой мини-приложение и открой заново через бота.');
        } else {
          showEmpty(`Ошибка загрузки (HTTP ${resp.status}).`);
        }
        return;
      }
      const data = await resp.json();
      renderWords(data.words || [], data.srs || null);
    } catch (err) {
      console.error(err);
      showEmpty('Не удалось загрузить словарь. Проверь, что бот запущен.');
    }
  }

  function renderWords(words, srs) {
    allWords = words;
    if (srs && srs.max_box) maxBox = srs.max_box;

    words.forEach(w => {
      if (w.topic_name && !topicNames[w.topic_id]) {
        topicNames[w.topic_id] = w.topic_name;
      }
    });

    const masteredCount = words.filter(w => w.is_mastered).length;
    const topics = new Set(words.map(w => w.topic_id));

    document.getElementById('total-words').textContent = words.length;
    document.getElementById('mastered-words').textContent = masteredCount;
    document.getElementById('topics-count').textContent = topics.size;

    const dueCount = srs?.due ?? words.filter(w => w.is_due).length;
    const banner = document.getElementById('due-banner');
    if (dueCount > 0) {
      document.getElementById('due-count').textContent = dueCount;
      document.getElementById('due-word').textContent =
        pluralRu(dueCount, ['слово', 'слова', 'слов']);
      banner.classList.add('visible');
    } else {
      banner.classList.remove('visible');
    }

    const orderedTopicIds = [];
    const seen = new Set();
    words.forEach(w => {
      if (!seen.has(w.topic_id)) {
        seen.add(w.topic_id);
        orderedTopicIds.push(w.topic_id);
      }
    });
    buildTabs(orderedTopicIds);

    applyFilters();
  }

  function buildTabs(topicIds) {
    const container = document.getElementById('tabs');
    container.innerHTML = '';

    const allBtn = document.createElement('button');
    allBtn.className = 'tab active';
    allBtn.textContent = 'Все';
    allBtn.dataset.filter = 'all';
    container.appendChild(allBtn);

    topicIds.forEach(tid => {
      const name = topicNames[tid] || `Топик ${tid}`;
      const btn = document.createElement('button');
      btn.className = 'tab';
      btn.textContent = name;
      btn.dataset.filter = String(tid);
      container.appendChild(btn);
    });
  }

  document.getElementById('tabs').addEventListener('click', (e) => {
    const btn = e.target.closest('.tab');
    if (!btn) return;
    selectTopicFilter(btn.dataset.filter);
  });

  // Общая функция выбора топика — используется и табами, и модалкой
  function selectTopicFilter(filter) {
    currentTopicFilter = filter;

    // Синхронизируем активный таб (если такого таба нет — оставляем "Все" неактивной,
    // но сам фильтр всё равно применится)
    document.querySelectorAll('#tabs .tab').forEach(t => {
      t.classList.toggle('active', t.dataset.filter === filter);
    });

    applyFilters();
  }

  // Строим список топиков в модалке на основе уже загруженных слов.
  // Семантически верно: в словаре показываем только те топики, в которых
  // у пользователя реально есть выученные слова.
  function buildTopicsList() {
    const list = document.getElementById('topics-list');
    list.innerHTML = '';

    // Группируем по topic_id с сохранением порядка из allWords
    // (API уже отдаёт слова в порядке сложности топиков)
    const grouped = new Map(); // topic_id -> { name, total, mastered, due }
    allWords.forEach(w => {
      if (!grouped.has(w.topic_id)) {
        grouped.set(w.topic_id, {
          topic_id: w.topic_id,
          name: w.topic_name || topicNames[w.topic_id] || `Топик ${w.topic_id}`,
          total: 0,
          mastered: 0,
          due: 0,
        });
      }
      const g = grouped.get(w.topic_id);
      g.total += 1;
      if (w.is_mastered) g.mastered += 1;
      if (w.is_due) g.due += 1;
    });

    if (grouped.size === 0) {
      list.innerHTML = `
        <div class="topics-empty">
          Топиков пока нет — начни учить слова в боте,
          и они появятся здесь.
        </div>
      `;
      return;
    }

    // Кнопка "Все топики" в самом верху
    const allRow = document.createElement('div');
    allRow.className = 'topic-row' + (currentTopicFilter === 'all' ? ' selected' : '');
    allRow.dataset.filter = 'all';
    allRow.innerHTML = `
      <div class="topic-row-main">
        <div class="topic-row-name">Все топики</div>
        <div class="topic-row-meta">показать все слова</div>
      </div>
      <span class="topic-row-count">${allWords.length}</span>
    `;
    list.appendChild(allRow);

    grouped.forEach(g => {
      const row = document.createElement('div');
      const isSelected = currentTopicFilter === String(g.topic_id);
      row.className = 'topic-row' + (isSelected ? ' selected' : '');
      row.dataset.filter = String(g.topic_id);

      const metaParts = [];
      if (g.mastered > 0) metaParts.push(`🏆 ${g.mastered} закреплено`);
      if (g.due > 0) metaParts.push(`🔁 ${g.due} к повторению`);
      const meta = metaParts.length
        ? metaParts.join(' · ')
        : `${g.total} ${pluralRu(g.total, ['слово', 'слова', 'слов'])}`;

      row.innerHTML = `
        <div class="topic-row-main">
          <div class="topic-row-name">${escapeHtml(g.name)}</div>
          <div class="topic-row-meta">${escapeHtml(meta)}</div>
        </div>
        <span class="topic-row-count">${g.total}</span>
      `;
      list.appendChild(row);
    });
  }

  document.getElementById('stats').addEventListener('click', (e) => {
    const card = e.target.closest('.stat-card');
    if (!card) return;
    const filter = card.dataset.filter;

    // Клик по карточке "Топиков" → открыть модалку со списком всех топиков
    if (filter === 'topics-count-stub') {
      openTopics();
      return;
    }

    if (currentStateFilter === filter) {
      currentStateFilter = 'all';
    } else {
      currentStateFilter = filter;
    }

    document.querySelectorAll('.stat-card').forEach(c => c.classList.remove('active'));
    if (currentStateFilter === 'all') {
      document.querySelector('.stat-card[data-filter="all"]').classList.add('active');
    } else {
      card.classList.add('active');
    }
    applyFilters();
  });

  document.getElementById('due-banner').addEventListener('click', () => {
    if (!tg) return;
    try {
      tg.showPopup({
        title: 'Готовы к повторению',
        message: 'Закрой это окно и в боте открой «📚 Слова» — сверху появится кнопка повторения.',
        buttons: [
          { id: 'close', type: 'default', text: 'Закрыть и перейти' },
          { id: 'cancel', type: 'cancel' },
        ],
      }, (btnId) => {
        if (btnId === 'close') tg.close();
      });
    } catch (e) {
      tg.close();
    }
  });

  let searchTimer;
  document.getElementById('search').addEventListener('input', () => {
    clearTimeout(searchTimer);
    searchTimer = setTimeout(applyFilters, 150);
  });

  function applyFilters() {
    const query = document.getElementById('search').value.trim().toLowerCase();
    let filtered = allWords;

    if (currentTopicFilter !== 'all') {
      filtered = filtered.filter(w => String(w.topic_id) === currentTopicFilter);
    }

    if (currentStateFilter === 'mastered') {
      filtered = filtered.filter(w => w.is_mastered);
    } else if (currentStateFilter === 'due') {
      filtered = filtered.filter(w => w.is_due);
    }

    if (query) {
      filtered = filtered.filter(w =>
        w.word_en.toLowerCase().includes(query) ||
        w.word_ru.toLowerCase().includes(query)
      );
    }
    displayWords(filtered);
  }

  function displayWords(words) {
    const container = document.getElementById('words-container');

    if (words.length === 0) {
      const isSearching = document.getElementById('search').value.trim().length > 0;
      const isFiltered = currentTopicFilter !== 'all' || currentStateFilter !== 'all';

      if (allWords.length === 0) {
        showEmpty('Начни изучать слова в боте — они появятся здесь.');
      } else if (isSearching) {
        showEmpty('По этому запросу ничего не найдено.');
      } else if (isFiltered) {
        showEmpty('Нет слов под этот фильтр. Попробуй другой.');
      } else {
        showEmpty('Тут пока пусто.');
      }
      return;
    }

    container.innerHTML = words.map((w, i) => {
      const acc = Math.round(w.correct / Math.max(w.attempts, 1) * 100);
      const accClass = acc >= 80 ? 'acc-high' : acc >= 50 ? 'acc-mid' : 'acc-low';
      const topicName = w.topic_name || topicNames[w.topic_id] || '';
      const boxBadge = renderBoxBadge(w);
      const cardClass = w.is_due ? 'word-card is-due' : 'word-card';

      return `
        <div class="${cardClass}" style="animation-delay: ${Math.min(i * 0.03, 0.6)}s">
          <div class="word-main">
            <div class="word-en">${escapeHtml(w.word_en)}</div>
            <div class="word-ru">${escapeHtml(w.word_ru)}</div>
            <span class="topic-tag">${escapeHtml(topicName)}</span>
          </div>
          <div class="word-meta">
            ${boxBadge}
            <span class="badge ${accClass}">${acc}%</span>
          </div>
        </div>
      `;
    }).join('');
  }

  function renderBoxBadge(w) {
    if (w.box == null) {
      return `<span class="badge box-badge box-none" title="Не в SRS">📦 —</span>`;
    }
    const cls = `box-${w.box}`;
    const dueCls = w.is_due ? ' is-due' : '';
    const icon = w.is_mastered ? '🏆' : '📦';
    const title = w.is_due
      ? 'Готово к повторению'
      : (w.next_review_at ? `Следующее повторение: ${formatDate(w.next_review_at)}` : '');
    return `<span class="badge box-badge ${cls}${dueCls}" title="${escapeHtml(title)}">${icon} ${w.box}/${maxBox}</span>`;
  }

  function formatDate(iso) {
    try {
      const d = new Date(iso + (iso.endsWith('Z') ? '' : 'Z'));
      const now = new Date();
      const diffDays = Math.round((d - now) / (1000 * 60 * 60 * 24));
      if (diffDays <= 0) return 'сегодня';
      if (diffDays === 1) return 'завтра';
      if (diffDays < 30)  return `через ${diffDays} ${pluralRu(diffDays, ['день', 'дня', 'дней'])}`;
      return d.toLocaleDateString('ru-RU', { day: 'numeric', month: 'short' });
    } catch (e) {
      return iso;
    }
  }

  function showEmpty(msg) {
    document.getElementById('words-container').innerHTML = `
      <div class="empty-state">
        <div class="empty-emoji">📖</div>
        <div class="empty-title">Ничего нет</div>
        <div class="empty-text">${escapeHtml(msg)}</div>
      </div>
    `;
  }

  function escapeHtml(str) {
    if (str == null) return '';
    return String(str)
      .replace(/&/g,'&amp;')
      .replace(/</g,'&lt;')
      .replace(/>/g,'&gt;')
      .replace(/"/g,'&quot;')
      .replace(/'/g,'&#39;');
  }

  loadWords();
</script>
</body>
