    <script>
        let config = {};

        // ============ AUTH ============
        function getToken() { return sessionStorage.getItem('ex24_token'); }
        function setToken(t) { sessionStorage.setItem('ex24_token', t); }
        function clearToken() { sessionStorage.removeItem('ex24_token'); }

        function showLogin() {
            document.getElementById('loginOverlay').classList.remove('hidden');
        }
        function hideLogin() {
            document.getElementById('loginOverlay').classList.add('hidden');
        }

        // Enter key on password field
        document.getElementById('loginPassword').addEventListener('keydown', e => {
            if (e.key === 'Enter') doLogin();
        });

        async function doLogin() {
            const pw = document.getElementById('loginPassword').value;
            const errEl = document.getElementById('loginError');
            errEl.textContent = '';
            try {
                const resp = await fetch('/api/auth/login', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ password: pw })
                });
                const data = await resp.json();
                if (data.success && data.token) {
                    setToken(data.token);
                    hideLogin();
                    loadConfig();
                } else {
                    errEl.textContent = data.error || 'Неверный пароль';
                    document.getElementById('loginPassword').select();
                }
            } catch {
                errEl.textContent = 'Ошибка подключения к серверу';
            }
        }

        async function doLogout() {
            try {
                await authFetch('/api/auth/logout', { method: 'POST' });
            } catch {}
            clearToken();
            document.getElementById('loginPassword').value = '';
            showLogin();
        }

        // Обёртка fetch с авторизацией
        async function authFetch(url, opts = {}) {
            const token = getToken();
            if (!opts.headers) opts.headers = {};
            if (token) opts.headers['Authorization'] = 'Bearer ' + token;
            const resp = await fetch(url, opts);
            if (resp.status === 401) {
                clearToken();
                showLogin();
                throw new Error('Unauthorized');
            }
            return resp;
        }

        function toggleEditor(header) {
            const body = header.nextElementSibling;
            const toggle = header.querySelector('.toggle');
            body.classList.toggle('open');
            toggle.textContent = body.classList.contains('open') ? '▲ свернуть' : '▼ развернуть';
        }

        function esc(str) {
            const d = document.createElement('div');
            d.textContent = str || '';
            return d.innerHTML;
        }

        async function uploadImage(type) {
            const input = document.getElementById(type + 'ImageFile');
            if (!input.files || !input.files[0]) return;
            const formData = new FormData();
            formData.append('file', input.files[0]);
            formData.append('type', type);
            try {
                const token = getToken();
                const resp = await fetch('/api/config/upload-image', {
                    method: 'POST',
                    headers: token ? { 'Authorization': 'Bearer ' + token } : {},
                    body: formData
                });
                if (resp.status === 401) { clearToken(); showLogin(); return; }
                const data = await resp.json();
                if (resp.ok) {
                    showToast(`✅ Изображение «${type}» загружено: ${data.fileName}`, 'success');
                    setTimeout(loadConfig, 500);
                } else {
                    showToast(data.error || 'Ошибка загрузки', 'error');
                }
            } catch (e) { showToast('Ошибка подключения', 'error'); }
            input.value = '';
        }

        async function unbanUser(userId) {
            if (!confirm(`Разблокировать пользователя ${userId}?`)) return;
            config.bannedUsers = (config.bannedUsers || []).filter(id => id !== userId);
            try {
                const resp = await authFetch('/api/config/full', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(config)
                });
                if (resp.ok) {
                    showToast(`✅ Пользователь ${userId} разблокирован`, 'success');
                    loadConfig();
                } else showToast('Ошибка сохранения', 'error');
            } catch (e) { if (e.message !== 'Unauthorized') showToast('Ошибка подключения', 'error'); }
        }

        async function exportBanList() {
            try {
                const resp = await authFetch('/api/config/bannedusers/export');
                const text = await resp.text();
                const blob = new Blob([text], { type: 'text/plain' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'banlist.txt';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                showToast('✅ Бан-лист выгружен', 'success');
            } catch (e) { if (e.message !== 'Unauthorized') showToast('Ошибка выгрузки', 'error'); }
        }
