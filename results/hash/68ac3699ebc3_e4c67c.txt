    (function() {

      const container = document.getElementById('sakuraContainer');

        const size = Math.floor(Math.random() * 16 + 10); // 10-25px
        const leftPos = Math.random() * 100;
        const delay = Math.random() * 15; // задержка анимации
        const duration = Math.random() * 8 + 10; // 10-18 секунд падения
        const swayDuration = Math.random() * 4 + 3; // качание
        
        petal.style.width = size + 'px';
        petal.style.height = size + 'px';
        petal.style.left = leftPos + '%';
        petal.style.animationDelay = delay + 's';
        petal.style.animationDuration = duration + 's, ' + swayDuration + 's';

        const hue = 330 + Math.floor(Math.random() * 25); // оттенки розового
