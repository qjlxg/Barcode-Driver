function _parseLastGame(lgt){
    if(!lgt||lgt==='None_baza'||lgt==='—')return null;
    try{
        const m=lgt.match(/^(\d{2})-(\d{2})-(\d{4})\s+(\d{2}):(\d{2}):(\d{2})$/);
        if(!m)return null;
        return Date.UTC(+m[3],+m[2]-1,+m[1],+m[4]-3,+m[5],+m[6]);
    }catch(e){return null}
}
function fmtLastGame(lgt){
    const utcMs=_parseLastGame(lgt);
    if(utcMs===null)return'—';
    const diffSec=Math.max(0,Math.floor((Date.now()-utcMs)/1000));
    const h=Math.floor(diffSec/3600),min=Math.floor((diffSec%3600)/60),s=diffSec%60;
    if(h>0)return h+'h '+min+'m';
    if(min>0)return min+'m '+s+'s';
    return s+'s';
}

function renderGrid(computers){
    const grid=document.getElementById('grid');
    grid._lastComputers=computers; // сохраняем для перерендера при toggleSort

    // Handle empty state
    if(computers.length===0){
        grid.innerHTML='<div class="empty"><div class="ico">&#128187;</div><p>No computers found</p></div>';
        grid._prevCards={};
        return;
    }
    // Clear empty placeholder if present
    if(grid.firstChild&&grid.firstChild.classList&&grid.firstChild.classList.contains('empty')) grid.innerHTML='';

    // Применяем выбранный режим сортировки
    if(currentSort==='komp'){
        computers=[...computers].sort((a,b)=>(parseInt(a.komp)||0)-(parseInt(b.komp)||0));
    } else if(currentSort==='games'){
        computers=[...computers].sort((a,b)=>(parseInt(b.games_played)||0)-(parseInt(a.games_played)||0));
    } else if(currentSort==='ranked'){
        computers=[...computers].sort((a,b)=>(parseInt(b.games_played_rating)||0)-(parseInt(a.games_played_rating)||0));
    } else if(currentSort==='playtime'){
        computers=[...computers].sort((a,b)=>(parseFloat(b.total_game_time_seconds)||0)-(parseFloat(a.total_game_time_seconds)||0));
    } else if(currentSort==='lastgame'){
        computers=[...computers].sort((a,b)=>{
            const an=_parseLastGame(a.last_game_time)||0, bn=_parseLastGame(b.last_game_time)||0;
            return an-bn;
        });
    }

    // Сортировка: пары рядом, остальные в конце
    const inMatchStates=['picking','buying','in_game'];
    if(sortPairs&&currentSort==='komp'){
        const byKomp={};
        computers.forEach(pc=>byKomp[String(pc.komp)]=pc);
        const placed=new Set();
        const sorted=[];
        computers.forEach(pc=>{
            const k=String(pc.komp);
            if(placed.has(k)) return;
            const st=pc.bot_status||'idle';
            const opp=pc.opponent_komp_clean||'';
            if(inMatchStates.includes(st)&&opp&&byKomp[opp]&&!placed.has(opp)){
                sorted.push(pc); sorted.push(byKomp[opp]);
                placed.add(k); placed.add(opp);
            }
        });
        computers.forEach(pc=>{ if(!placed.has(String(pc.komp))) sorted.push(pc); });
        computers=sorted;
    }

    // Build pair color map — каждая пара (komp↔opponent) получает свой цвет обводки
    const PAIR_COLORS=['#4fc3f7','#ef5350','#66bb6a','#ffa726','#ab47bc','#26c6da','#ec407a','#d4e157','#ff7043','#80cbc4'];
    const pairColorMap={}; // komp(str) -> color
    let colorIdx=0;
    computers.forEach(pc=>{
        const st=pc.bot_status||'idle';
        const opp=pc.opponent_komp_clean||'';
        const k=String(pc.komp);
        if(!inMatchStates.includes(st)||!opp) return;
        if(k in pairColorMap) return; // уже назначен
        const color=opp in pairColorMap ? pairColorMap[opp] : PAIR_COLORS[colorIdx++%PAIR_COLORS.length];
        pairColorMap[k]=color;
        pairColorMap[opp]=color;
    });

    const prevCards=grid._prevCards||{};
    const newCards={};
    const newOrder=[];

    // Lookup: komp -> pc record (для проверки opponent)
    const byKompAll={};
    computers.forEach(pc=>{ byKompAll[String(pc.komp)]=pc; });

    computers.forEach((pc,i)=>{
        const st=pc.bot_status||'idle';
        const wl=String(pc.win_or_losses||'').toLowerCase();
        const wlCls=wl==='win'?'win':wl==='losses'?'loss':'';
        const wlTxt=wl==='win'?'WIN':wl==='losses'?'LOSS':(sv(wl)!=='—'?sv(wl).toUpperCase():'—');
        const vAccept=(pc.v_accept&&pc.v_accept!=='None_baza')?pc.v_accept:1;
        const opp=pc.opponent_komp_clean||'';

        // 👬 если оба имеют friend_accept=true И у них разные bot_password → разные владельцы → друг
        let isPartyFriend=false;
        if(opp && String(pc.friend_accept||'').toLowerCase()==='true'){
            const oppPc=byKompAll[String(opp)];
            if(oppPc && String(oppPc.friend_accept||'').toLowerCase()==='true'){
                const myBp=String(pc.bot_password||'');
                const oppBp=String(oppPc.bot_password||'');
                isPartyFriend=myBp&&oppBp&&myBp!==oppBp;
            }
        }

        let partHtml='';
        if(inMatchStates.includes(st)&&opp) partHtml='<div class="partner-badge paired">&#9876; Против PC #'+opp+(isPartyFriend?' &#128107;':'')+'</div>';
        else if(inMatchStates.includes(st)) partHtml='<div class="partner-badge solo">&#9876; В матче (соло)</div>';
        else partHtml='<div class="partner-badge solo">&#8212; Не в матче</div>';

        const pm=pc.ping_retool_ms;
        const pingHtml=pm&&pm!=='None_baza'?'<span class="bot-ping '+pingCls(pm)+'">BOT PING '+pm+' ms</span>':'';

        let rankedStr=sv(pc.games_played_rating);
        if(String(pc.auto_ranked_active).toLowerCase()==='true'&&pc.auto_ranked_hours&&pc.auto_ranked_hours!=='None_baza')
            rankedStr+=' <span style="color:var(--text-dim);font-size:.7em">(auto >'+pc.auto_ranked_hours+'h)</span>';

        const pairColor=pairColorMap[String(pc.komp)]||null;
        const key=String(pc.komp);

        // Fingerprint: all data + pair color (determines if card needs rebuild)
        const fp=JSON.stringify(pc)+'|'+(pairColor||'');

        let card;
        if(prevCards[key]&&prevCards[key].fp===fp){
            // No changes — reuse existing element as-is
            card=prevCards[key].el;
        } else {
            // New or changed — create or update element in place
            card=prevCards[key]?prevCards[key].el:document.createElement('div');
            card.className='pc-card s-'+st;
            if(!prevCards[key]) card.style.animationDelay=(i*.04)+'s'; // animate only new cards
            if(pairColor){
                card.style.border='2px solid '+pairColor;
                card.style.boxShadow='0 0 18px '+pairColor+', 0 0 6px '+pairColor+'aa, inset 0 0 20px '+pairColor+'15';
            } else {
                card.style.border='';
                card.style.boxShadow='';
            }
            card.innerHTML=
                '<div class="pc-hdr"><span class="pc-num">PC #'+sv(pc.komp)+'</span><span class="pc-st"><span class="d"></span>'+statusLabel(st,pc)+'</span></div>'+
                partHtml+pingHtml+
                '<div class="stats">'+
                    '<div class="si"><div class="sl">Win / Loss</div><div class="sv '+wlCls+'">'+wlTxt+'</div></div>'+
                    '<div class="si"><div class="sl">Game Mode</div><div class="sv">'+modeLabel(pc.game_mode)+'</div></div>'+
                    '<div class="si"><div class="sl">Region</div><div class="sv">'+sv(pc.region)+'</div></div>'+
                    '<div class="si"><div class="sl">V Accept</div><div class="sv">'+vAccept+'</div></div>'+
                    '<div class="si"><div class="sl">Games Played</div><div class="sv">'+sv(pc.games_played)+'</div></div>'+
                    '<div class="si"><div class="sl">Ranked</div><div class="sv">'+rankedStr+'</div></div>'+
                    '<div class="si"><div class="sl">Play Time</div><div class="sv">'+fmtTime(pc.total_game_time_seconds)+'</div></div>'+
                    '<div class="si"><div class="sl">Last Game</div><div class="sv">'+fmtLastGame(pc.last_game_time)+'</div></div>'+
                '</div>'+
                '<div class="pc-ftr"><div style="display:flex;align-items:center;gap:4px"><button class="pc-del-btn" onclick="deletePC(this,'+sv(pc.komp)+')">&#128465;</button><span>v'+sv(pc.VERSION)+'</span></div><span>'+sv(pc.timestamp)+'</span></div>';
        }

        newCards[key]={fp, el: card};
        newOrder.push(card);
    });

    // Remove cards no longer present in data
    Object.keys(prevCards).forEach(k=>{
        if(!newCards[k]&&prevCards[k].el.parentNode===grid) grid.removeChild(prevCards[k].el);
    });

    // Reorder/append cards to match sorted order (no recreate, just DOM move)
    newOrder.forEach((card,i)=>{
        if(grid.children[i]!==card) grid.insertBefore(card, grid.children[i]||null);
    });

    grid._prevCards=newCards;
}
