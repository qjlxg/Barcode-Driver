function toggleLang() {
  const cur = 'en';
  fetch('/settings', {
    method: 'POST',
    headers: {'Content-Type': 'application/x-www-form-urlencoded'},
    body: `action=change_language&language=${cur === 'en' ? 'fa' : 'en'}`
  }).then(() => location.reload());
