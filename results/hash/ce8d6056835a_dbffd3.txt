        /* Stars generator */
        function generateStars() {
            var container = document.getElementById('starsContainer');
            for (var i = 0; i < 80; i++) {
                var star = document.createElement('div');
                star.className = 'star';
                star.style.left = Math.random() * 100 + '%';
                star.style.top = Math.random() * 100 + '%';
                star.style.width = (Math.random() * 2 + 1) + 'px';

        /* Animated counters */
        function initCounters() {
            var targets = { statProjects: 3, statUptime: 99.9, statCommits: 2400, statNodes: 1 };
            var animated = {};
            var observer = new IntersectionObserver(function(entries) {
                entries.forEach(function(entry) {
                    if (entry.isIntersecting && !animated[entry.target.id]) {
                        animated[entry.target.id] = true;
                        animateCounter(entry.target.id, targets[entry.target.id]);
                    }
                });
            }, { threshold: 0.5 });
            Object.keys(targets).forEach(function(id) {
                var el = document.getElementById(id);
                if (el) observer.observe(el);
            });
        }
        function animateCounter(id, target) {
            var el = document.getElementById(id);
            if (!el) return;
            var isFloat = target % 1 !== 0;
            var duration = 1800;
            var start = performance.now();
            function step(now) {
                var progress = Math.min((now - start) / duration, 1);
                var eased = 1 - Math.pow(1 - progress, 3);
                var val = eased * target;
                el.textContent = isFloat ? val.toFixed(1) : Math.round(val);
                if (progress < 1) requestAnimationFrame(step);
            }
            requestAnimationFrame(step);
        }

        /* Scroll observer for card animations */
        function initScrollObserver() {
            var observer = new IntersectionObserver(function(entries) {
                entries.forEach(function(entry) {
                    if (entry.isIntersecting) {
                        var delay = parseInt(entry.target.dataset.delay || 0);
                        setTimeout(function() { entry.target.classList.add('visible'); }, delay);
                    }
                });
            }, { threshold: 0.15 });
            document.querySelectorAll('.project-card, .roadmap-card, .stat-item, .manifesto').forEach(function(card) { observer.observe(card); });
        }

        async function checkAuth() {
            var hdr = _authHeaders();
            var hasAuth = hdr['X-Session-Token'] || hdr['X-API-Key'];
            document.querySelectorAll('.auth-only').forEach(function(el) { el.style.display = 'none'; });
            document.querySelectorAll('.vpn-card').forEach(function(el) { el.style.display = 'none'; });

                var res = await fetch(DOTID + '/me', {headers: hdr});

            } catch (e) {}

        async function login() {
            var key = document.getElementById('keyInput').value.trim();
            var errorEl = document.getElementById('loginError');
            if (!key) {
                errorEl.textContent = currentLang === 'EN' ? 'Enter your access key' : (currentLang === 'KZ' ? '\u041a\u0456\u043b\u0442\u0442\u0456 \u0435\u043d\u0433\u0456\u0437\u0456\u04a3\u0456\u0437' : '\u0412\u0432\u0435\u0434\u0438 \u043a\u043b\u044e\u0447 \u0434\u043e\u0441\u0442\u0443\u043f\u0430');
                errorEl.style.display = 'block';
                return;
            }
            errorEl.style.display = 'none';
            try {
                // Nexus login — получаем session token
                var forceLogin = false;
                async function doNexusLogin() {
                    var nxRes = await fetch(DOTID + '/nexus/login', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ api_key: key, client_type: 'web', force: forceLogin })
                    });
                    if (nxRes.status === 401 || nxRes.status === 404) {
                        errorEl.textContent = currentLang === 'EN' ? 'Invalid key!' : (currentLang === 'KZ' ? '\u041a\u0456\u043b\u0442 \u049b\u0430\u0442\u0435!' : '\u041d\u0435\u0432\u0435\u0440\u043d\u044b\u0439 \u043a\u043b\u044e\u0447!');
                        errorEl.style.display = 'block';
                        errorEl.style.color = 'var(--red)';
                        return null;
                    }
                    if (!nxRes.ok) {
                        errorEl.textContent = currentLang === 'EN' ? 'Server error' : (currentLang === 'KZ' ? 'Сервер қатесі' : 'Ошибка сервера');
                        errorEl.style.display = 'block';
                        errorEl.style.color = 'var(--red)';
                        return null;
                    }
                    return await nxRes.json();
                }

                var nxData = await doNexusLogin();
                if (!nxData) return;

                // Конфликт — сессия на другом устройстве
                if (nxData.status === 'conflict') {
                    var confirmMsg = currentLang === 'EN'
                        ? 'You are already logged in on another device (' + nxData.existing_ip + '). Log out there and continue here?'
                        : (currentLang === 'KZ'
                            ? 'Сіз басқа құрылғыда кіргенсіз (' + nxData.existing_ip + '). Сол жерден шығып, осында жалғастырасыз ба?'
                            : 'Вы уже авторизованы на другом устройстве (' + nxData.existing_ip + '). Выйти там и продолжить здесь?');
                    if (confirm(confirmMsg)) {
                        forceLogin = true;
                        nxData = await doNexusLogin();
                        if (!nxData) return;
                    } else {
                        return;
                    }
                }

                // 2FA required
                if (nxData.requires_2fa) {
                    errorEl.textContent = currentLang === 'EN' ? '📱 Check Telegram to confirm login...' : (currentLang === 'KZ' ? '📱 Кіруді Telegram-да растаңыз...' : '📱 Подтвердите вход в Telegram...');
                    errorEl.style.display = 'block';
                    errorEl.style.color = 'var(--teal)';
                    await waitFor2FA(key, errorEl);
                    return;
                }

                // Сессия создана — сохраняем
                if (nxData.session_token) {
                    localStorage.setItem('dot_session', nxData.session_token);
                    localStorage.removeItem('dot_key'); // больше не нужен
                    var redirect = window._loginRedirect;
                    window._loginRedirect = null;
                    if (redirect) { location.href = redirect; } else { location.reload(); }
                } else {
                    // Fallback — если Nexus отключен на сервере
                    localStorage.setItem('dot_key', key);
                    var redirect = window._loginRedirect;
                    window._loginRedirect = null;
                    if (redirect) { location.href = redirect; } else { location.reload(); }
                }
            } catch (e) {
                errorEl.textContent = currentLang === 'EN' ? 'Server error' : (currentLang === 'KZ' ? 'Сервер қатесі' : 'Ошибка сервера');
                errorEl.style.display = 'block';
                errorEl.style.color = 'var(--red)';
            }
        }

        async function waitFor2FA(key, errorEl) {
            var attempts = 0;
            var maxAttempts = 60;
            var errors = 0;
            var interval = setInterval(async function() {
                attempts++;
                try {
                    // Используем Nexus confirm — сразу получаем session_token при approved
                    var res = await fetch(DOTID + '/nexus/confirm', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ api_key: key, client_type: 'web' })
                    });
                    if (res.status === 429) {
                        errors++;
                        if (errors >= 3) {
                            clearInterval(interval);
                            errorEl.textContent = currentLang === 'EN' ? '\u26a0\ufe0f Too many requests, try again later' : (currentLang === 'KZ' ? '\u26a0\ufe0f \u0422\u044b\u043c \u043a\u04e9\u043f \u0441\u04b1\u0440\u0430\u0443, \u043a\u0435\u0439\u0456\u043d\u0456\u0440\u0435\u043a \u049b\u0430\u0439\u0442\u0430\u043b\u0430\u04a3\u044b\u0437' : '\u26a0\ufe0f \u0421\u043b\u0438\u0448\u043a\u043e\u043c \u043c\u043d\u043e\u0433\u043e \u0437\u0430\u043f\u0440\u043e\u0441\u043e\u0432, \u043f\u043e\u043f\u0440\u043e\u0431\u0443\u0439 \u043f\u043e\u0437\u0436\u0435');
                            errorEl.style.color = 'var(--red)';
                        }
                        return;
                    }
                    errors = 0;
                    var data = await res.json();
                    if (data.status === 'approved' && data.session_token) {
                        clearInterval(interval);
                        localStorage.setItem('dot_session', data.session_token);
                        localStorage.removeItem('dot_key');
                        var redirect = window._loginRedirect;
                        window._loginRedirect = null;
                        if (redirect) { location.href = redirect; } else { location.reload(); }
                    } else if (data.status === 'denied') {
                        clearInterval(interval);
                        errorEl.textContent = currentLang === 'EN' ? '\u274c Login denied via Telegram' : (currentLang === 'KZ' ? '\u274c \u041a\u0456\u0440\u0443 Telegram \u0430\u0440\u049b\u044b\u043b\u044b \u049b\u0430\u0431\u044b\u043b\u0434\u0430\u043d\u0431\u0430\u0434\u044b' : '\u274c \u0412\u0445\u043e\u0434 \u043e\u0442\u043a\u043b\u043e\u043d\u0451\u043d \u0447\u0435\u0440\u0435\u0437 Telegram');
                        errorEl.style.color = 'var(--red)';
                    }
                } catch (e) {
                    errors++;
                    if (errors >= 5) {
                        clearInterval(interval);
                        errorEl.textContent = currentLang === 'EN' ? '\u26a0\ufe0f Connection error' : (currentLang === 'KZ' ? '\u26a0\ufe0f \u0411\u0430\u0439\u043b\u0430\u043d\u044b\u0441 \u049b\u0430\u0442\u0435\u0441\u0456' : '\u26a0\ufe0f \u041e\u0448\u0438\u0431\u043a\u0430 \u0441\u043e\u0435\u0434\u0438\u043d\u0435\u043d\u0438\u044f');
                        errorEl.style.color = 'var(--red)';
                    }
                }
                if (attempts >= maxAttempts) {
                    clearInterval(interval);
                    errorEl.textContent = currentLang === 'EN' ? '⏳ 2FA timeout' : (currentLang === 'KZ' ? '⏳ Уақыт аяқталды' : '⏳ Время ожидания истекло');
                    errorEl.style.color = 'var(--red)';
                }
            }, 2000);
        }

        function parseHostPortUserPass(input) {
            input = input.trim();
            if (!input) return null;
            try {
                if (input.startsWith('socks5:') || input.startsWith('http:') || input.startsWith('https:')) {
                    var url = new URL(input);
                    var scheme = url.protocol.replace(':', '');
                    var host = url.hostname;
                    var port = url.port || (scheme === 'https' ? '443' : '80');
                    var username = decodeURIComponent(url.username || '');
                    var password = decodeURIComponent(url.password || '');
                    return { scheme: scheme, host: host, port: port, user: username, pass: password };
                }
            } catch (e) {}

            var parts = input.split(':');
            if (parts.length === 4) {
                return { scheme: 'socks5', host: parts[0].trim(), port: parts[1].trim(), user: decodeURIComponent(parts[2].trim()), pass: decodeURIComponent(parts[3].trim()) };
            }
            return null;
        }

        function convertDotCalc() {
            var inp = document.getElementById('dotcalc-input').value;
            var name = (document.getElementById('dotcalc-name').value || '').trim();
            document.getElementById('dotcalc-error').style.display = 'none';
            var parsed = parseHostPortUserPass(inp);
            if (!parsed) {
                var dict = translations[currentLang] || {};
                showDotCalcError(dict.dotCalcErrParse || 'Parse error');
                var panel = document.querySelector('.modal-panel');
                if (panel) {
                    panel.classList.remove('conversion-success');
                    panel.classList.add('conversion-error');
                    setTimeout(function() { panel.classList.remove('conversion-error'); }, 600);
                }
                return;
            }

            var userPart = parsed.user ? encodeURIComponent(parsed.user) : '';
            var passPart = parsed.pass ? encodeURIComponent(parsed.pass) : '';
            var auth = userPart || passPart ? userPart + ':' + passPart + '@' : '';
            var scheme = selectedProto === 'auto' ? (parsed.scheme || 'socks5') : selectedProto;
            var tag = name ? '#' + encodeURIComponent(name) : '';
            var uri = scheme + '://' + auth + parsed.host + ':' + parsed.port + tag;
            document.getElementById('dotcalc-result').value = uri;
            document.getElementById('dotcalc-copy').disabled = false;

            var panel2 = document.querySelector('.modal-panel');
            if (panel2) {
                panel2.classList.remove('conversion-error');
                panel2.classList.add('conversion-success');
                setTimeout(function() { panel2.classList.remove('conversion-success'); }, 800);
            }
        }
