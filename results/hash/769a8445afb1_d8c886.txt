     function FocusOnInput() {
        document.getElementById("username").focus();

         $('document').ready(function() 
         { 
            // ASTPPENT-4515 Kinjal start
            const agent = window.navigator.userAgent.toLowerCase();
            const browser = agent.indexOf('edge') > -1 ? 'edge'
              : agent.indexOf('edg') > -1 ? 'chromium based edge'
              : agent.indexOf('opr') > -1 && window.opr ? 'opera'
              : agent.indexOf('chrome') > -1 && window.chrome ? 'chrome'
              : agent.indexOf('trident') > -1 ? 'ie'
              : agent.indexOf('firefox') > -1 ? 'firefox'
              : agent.indexOf('safari') > -1 ? 'safari'
              : 'other';
            // ASTPPENT-4515 Kinjal End
             $("#login_form").validate({
                 rules: {
                     username: {
                         required: true,
                     },
                     password: {
                         required: true,
                     },
                },
                 messages: {
                     username: { 
                         required: '<span class="text-danger">Account No is Required</span>',
                     },
                     password: {
                        required: '<span class="text-danger">Password is Required</span>',
                     },
                 },
                 errorClass: "error_label",
                 submitHandler: function(form) {
                                var username_decrypt = $("#username").val();
                var password_decrypt = $("#password").val();
                var password = "123456";
                username = CryptoJSAesJson.encrypt(username_decrypt, password);
                password1 = CryptoJSAesJson.encrypt(password_decrypt, password);
                $("#username").val(username);
                $("#password").val(password1);
                // Kinjal issue no 4515
                if(browser == 'safari' ){
                    $('<input>').attr({
                        type: 'hidden',id: 'new_username',name: 'new_username',value: username
                    }).appendTo('form');
                    $('<input>').attr({
                        type: 'hidden',id: 'new_password',name: 'new_password',value: password1
                    }).appendTo('form');
                    form.submit();
                                        localStorage.setItem('user_logout', ''); 
                                        $('#login_form')[0].reset();
                }else{
                    form.submit();
                                        localStorage.setItem('user_logout', ''); 
                                        $("#username").val(username_decrypt);
                    $("#password").val(password_decrypt);
                }
                // END
                			$(':input[type="submit"]').prop('disabled', true);
			$('input[type="text"]').keyup(function() {
			if($(this).val() != '') {
			$(':input[type="submit"]').prop('disabled', false);
			}
			});
                 }
             });  
         });
