    <script>
        // Переключение языков
        function switchLang(lang) {
            const setLang = (texts) => {
                document.getElementById('title').textContent = texts.title;
                document.getElementById('username-label').textContent = texts.username;
                document.getElementById('password-label').textContent = texts.password;
                document.getElementById('submit-btn').textContent = texts.submit;
                document.getElementById('ru-btn').classList.toggle('active', lang === 'ru');
                document.getElementById('en-btn').classList.toggle('active', lang === 'en');
                // Обновляем текст ошибки, если она была
                if (document.getElementById('general-error').style.display !== 'none') {
                    document.getElementById('general-error').textContent = texts.error;
                }
            };

            const translations = {
                ru: {
                    title: 'Файловый сервер',
                    username: 'Имя пользователя',
                    password: 'Пароль',
                    submit: 'Войти',
                    error: 'Ошибка: неверный логин или пароль'
                },
                en: {
                    title: 'File Server',
                    username: 'Username',
                    password: 'Password',
                    submit: 'Log In',
                    error: 'Error: invalid username or password'
                }
            };

            setLang(translations[lang]);
        }

        // Обработка входа
        function handleLogin(e) {
            e.preventDefault();
            // Сброс ошибок
            document.getElementById('username-error').style.display = 'none';
            document.getElementById('password-error').style.display = 'none';
            document.getElementById('general-error').style.display = 'none';

            const username = document.getElementById('username').value.trim();
            const password = document.getElementById('password').value;

            const lang = document.getElementById('ru-btn').classList.contains('active') ? 'ru' : 'en';
            const errorMsg = lang === 'ru' ? 'Ошибка: неверный логин или пароль' : 'Error: invalid username or password';

            if (!username) {
                document.getElementById('username-error').textContent = lang === 'ru' ? 'Введите имя пользователя' : 'Enter username';
                document.getElementById('username-error').style.display = 'block';
                return;
            }
            if (!password) {
                document.getElementById('password-error').textContent = lang === 'ru' ? 'Введите пароль' : 'Enter password';
                document.getElementById('password-error').style.display = 'block';
                return;
            }

            // Показываем ошибку при попытке входа
            document.getElementById('general-error').textContent = errorMsg;
            document.getElementById('general-error').style.display = 'block';
        }

        // Установка текущего года
        document.getElementById('year').textContent = new Date().getFullYear();
    </script>
