          }

          oneObs.unobserve(e.target);
        }
      });
    }, { threshold: 0.4 });

const LOCALE_KEY = 'bidby:locale';
const VALID_LOCALES = ['en', 'ru', 'ar'];
const RTL_LOCALES = ['ar'];
let currentLocale = 'ru';

function persistLocale(locale) {
  try { localStorage.setItem(LOCALE_KEY, locale); } catch(e) {}
}

function applyLocale(locale) {
  if (!VALID_LOCALES.includes(locale)) locale = 'ru';
  currentLocale = locale;
  const msgs = LANDING_I18N[locale];

  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.dataset.i18n;
    if (msgs[key] !== undefined) el.textContent = msgs[key];
  });

  document.documentElement.lang = locale;
  document.documentElement.dir = RTL_LOCALES.includes(locale) ? 'rtl' : 'ltr';

  if (msgs['page-title']) document.title = msgs['page-title'];

  const codes = { en: 'EN', ru: 'RU', ar: 'AR' };
  const btn = document.getElementById('lang-current');
  if (btn) btn.textContent = codes[locale];

  document.querySelectorAll('.lang-item').forEach(item => {
    item.classList.toggle('is-active', item.dataset.lang === locale);
  });

  persistLocale(locale);
}

function toggleLangMenu() {
  document.getElementById('lang-menu').classList.toggle('is-open');
}

function setLang(locale) {
  applyLocale(locale);
  document.getElementById('lang-menu').classList.remove('is-open');
}

document.addEventListener('click', e => {
  if (!e.target.closest('#lang-switcher'))
    document.getElementById('lang-menu')?.classList.remove('is-open');
});

const _queryLocale = (() => {
  try { return new URLSearchParams(window.location.search).get('lang'); } catch(e) { return null; }
})();
const _stored = (() => { try { return localStorage.getItem(LOCALE_KEY); } catch(e) { return null; } })();
applyLocale(VALID_LOCALES.includes(_queryLocale) ? _queryLocale : (VALID_LOCALES.includes(_stored) ? _stored : 'ru'));
