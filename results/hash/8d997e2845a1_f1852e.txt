const PROGRESS = document.getElementById('progress');
const QCOUNT  = document.getElementById('qcount');
const QEL     = document.getElementById('question');
const OPTS    = document.getElementById('options');
const REACT   = document.getElementById('reaction');
let idx = 0;

// прогресс-лента
QUESTIONS.forEach(()=>{const d=document.createElement('div');d.className='dot';PROGRESS.appendChild(d);});
const DOTS = [...PROGRESS.children];

function show(id){
  document.querySelectorAll('.screen').forEach(s=>s.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  window.scrollTo({top:0,behavior:'smooth'});
}

function render(){
  const item = QUESTIONS[idx];
  QCOUNT.textContent = `Вопрос ${idx+1} из ${QUESTIONS.length}`;
  QEL.textContent = item.q;
  REACT.textContent = ''; REACT.classList.remove('show');
  DOTS.forEach((d,i)=>d.classList.toggle('on', i<=idx));
  OPTS.innerHTML = '';
  item.options.forEach((text,i)=>{
    const b=document.createElement('button');
    b.className='opt';
    b.innerHTML=`<span class="mark">${i+1}</span><span>${text}</span>`;
    b.onclick=()=>pick(b,i,item);
    OPTS.appendChild(b);
  });
}

function pick(btn, i, item){
  [...OPTS.children].forEach(o=>{o.classList.remove('picked');o.style.pointerEvents='none';});
  btn.classList.add('picked');
  REACT.textContent = item.reaction || "Идеально. Дальше!";
  REACT.classList.add('show');
  setTimeout(()=>{
    idx++;
    if(idx < QUESTIONS.length){
      OPTS.querySelectorAll('.opt').forEach(o=>o.style.pointerEvents='');
      render();
    } else {
      show('screen-win');
      celebrate();
    }
  }, 1400);
}

document.getElementById('start').onclick=()=>{show('screen-quiz');render();};
document.getElementById('yes').onclick=()=>{
  document.getElementById('afterYes').textContent = "Тогда наряжайся. Столик за мной. 🍷";
  celebrate();
};

  e.style.animationDuration=(7+Math.random()*9)+'s';
  e.style.animationDelay=(-Math.random()*12)+'s';
  E.appendChild(e);
}

/* лёгкий «дождь из лепестков/искр» на победе */
function celebrate(){
  if(window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;
  for(let i=0;i<36;i++){
    const p=document.createElement('div');
    p.textContent = Math.random()>.5 ? '❤' : '✦';
    p.style.cssText=`position:fixed;z-index:5;top:-30px;left:${Math.random()*100}vw;
      font-size:${10+Math.random()*16}px;color:${Math.random()>.5?'#d4738c':'#e3bd78'};
      pointer-events:none;opacity:.9;`;
    document.body.appendChild(p);
    const dur=2600+Math.random()*2200;
    p.animate([
      {transform:`translateY(0) rotate(0deg)`,opacity:.9},
      {transform:`translateY(105vh) rotate(${(Math.random()*2-1)*360}deg)`,opacity:0}
    ],{duration:dur,easing:'cubic-bezier(.4,0,.6,1)'}).onfinish=()=>p.remove();
  }
}
