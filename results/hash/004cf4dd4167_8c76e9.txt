  <script>
    (() => {
      const canvas = document.getElementById("network");
      const ctx = canvas.getContext("2d", { alpha: true });
      const reducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)").matches;

      if (!ctx || reducedMotion) return;

      let width = 0;
      let height = 0;
      let ratio = 1;
      let points = [];
      let frame = 0;
      let lastTime = 0;
      let running = true;
      const pointer = { x: -9999, y: -9999, active: false };

      const config = {
        maxPoints: 46,
        minPoints: 24,
        connectionDistance: 168,
        pointerDistance: 220,
        speed: 0.115
      };

      function pointCount() {
        const area = width * height;
        return Math.max(
          config.minPoints,
          Math.min(config.maxPoints, Math.round(area / 33000))
        );
      }

      function makePoint() {
        return {
          x: Math.random() * width,
          y: Math.random() * height,
          vx: (Math.random() - 0.5) * config.speed,
          vy: (Math.random() - 0.5) * config.speed,
          r: Math.random() * 0.75 + 0.45
        };
      }

      function resize() {
        width = window.innerWidth;
        height = window.innerHeight;
        ratio = Math.min(window.devicePixelRatio || 1, 1.5);

        canvas.width = Math.round(width * ratio);
        canvas.height = Math.round(height * ratio);
        canvas.style.width = width + "px";
        canvas.style.height = height + "px";
        ctx.setTransform(ratio, 0, 0, ratio, 0, 0);

        const count = pointCount();
        points = Array.from({ length: count }, (_, index) => points[index] || makePoint());
      }

      function update(delta) {
        const step = Math.min(delta, 32);

        for (const point of points) {
          point.x += point.vx * step;
          point.y += point.vy * step;

          if (point.x < -20) point.x = width + 20;
          if (point.x > width + 20) point.x = -20;
          if (point.y < -20) point.y = height + 20;
          if (point.y > height + 20) point.y = -20;

          if (pointer.active) {
            const dx = point.x - pointer.x;
            const dy = point.y - pointer.y;
            const distanceSq = dx * dx + dy * dy;
            const limitSq = config.pointerDistance * config.pointerDistance;

            if (distanceSq > 1 && distanceSq < limitSq) {
              const distance = Math.sqrt(distanceSq);
              const force = (1 - distance / config.pointerDistance) * 0.008 * step;
              point.x += (dx / distance) * force;
              point.y += (dy / distance) * force;
            }
          }
        }
      }

      function draw() {
        ctx.clearRect(0, 0, width, height);

        for (let i = 0; i < points.length; i++) {
          const a = points[i];

          ctx.beginPath();
          ctx.arc(a.x, a.y, a.r, 0, Math.PI * 2);
          ctx.fillStyle = "rgba(244, 243, 239, 0.28)";
          ctx.fill();

          for (let j = i + 1; j < points.length; j++) {
            const b = points[j];
            const dx = a.x - b.x;
            const dy = a.y - b.y;
            const distanceSq = dx * dx + dy * dy;
            const limit = config.connectionDistance;

            if (distanceSq < limit * limit) {
              const distance = Math.sqrt(distanceSq);
              const alpha = (1 - distance / limit) * 0.10;
              ctx.beginPath();
              ctx.moveTo(a.x, a.y);
              ctx.lineTo(b.x, b.y);
              ctx.strokeStyle = `rgba(244, 243, 239, ${alpha})`;
              ctx.lineWidth = 0.65;
              ctx.stroke();
            }
          }

          if (pointer.active) {
            const dx = a.x - pointer.x;
            const dy = a.y - pointer.y;
            const distanceSq = dx * dx + dy * dy;
            const limit = config.pointerDistance;

            if (distanceSq < limit * limit) {
              const distance = Math.sqrt(distanceSq);
              const alpha = (1 - distance / limit) * 0.15;
              ctx.beginPath();
              ctx.moveTo(a.x, a.y);
              ctx.lineTo(pointer.x, pointer.y);
              ctx.strokeStyle = `rgba(182, 240, 156, ${alpha})`;
              ctx.lineWidth = 0.7;
              ctx.stroke();
            }
          }
        }
      }

      function animate(time) {
        if (!running) return;
        const delta = lastTime ? time - lastTime : 16;
        lastTime = time;

        update(delta);
        draw();
        frame = requestAnimationFrame(animate);
      }

      window.addEventListener("resize", resize, { passive: true });

      window.addEventListener("pointermove", (event) => {
        pointer.x = event.clientX;
        pointer.y = event.clientY;
        pointer.active = true;
      }, { passive: true });

      document.addEventListener("pointerleave", () => {
        pointer.active = false;
      });

      document.addEventListener("visibilitychange", () => {
        running = !document.hidden;
        if (running) {
          lastTime = 0;
          cancelAnimationFrame(frame);
          frame = requestAnimationFrame(animate);
        } else {
          cancelAnimationFrame(frame);
        }
      });

      resize();
      frame = requestAnimationFrame(animate);
    })();
  </script>
