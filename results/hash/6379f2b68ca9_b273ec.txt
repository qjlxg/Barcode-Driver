async function pay(){
const btn=document.getElementById('payBtn');
const status=document.getElementById('status');
btn.disabled=true;
status.className='loading';
status.style.display='block';
status.textContent='Создаём платёж...';
try{
const device=document.getElementById('device').value;
const months=document.getElementById('months').value;
const email=document.getElementById('email').value.trim();
if(!email){alert('Укажите email');btn.disabled=false;status.style.display='none';return;}
const r=await fetch('/create-payment',{
method:'POST',
headers:{'Content-Type':'application/json'},
body:JSON.stringify({device_type:device,months:parseInt(months),email:email})
});
const d=await r.json();
if(d.url){
window.location.href=d.url;
}else{
status.className='error';
status.textContent='Ошибка: '+d.error;
btn.disabled=false;
}
}catch(e){
status.className='error';
status.textContent='Ошибка соединения';
btn.disabled=false;
}
}
document.querySelectorAll('.instr-tab').forEach(function(t){

if(panel)panel.classList.add('active');
});
});
