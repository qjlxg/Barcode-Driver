// State
let meals = [], categories = [], tags = [], fridge = [], plan = [], shoppingList = [];

const DAY_SHORT = ['Mo','Di','Mi','Do','Fr','Sa','So'];

// API helpers
async function api(path, opts = {}) {
  const res = await fetch('/api' + path, {
    headers: { 'Content-Type': 'application/json' },
    ...opts,
    body: opts.body ? JSON.stringify(opts.body) : undefined,
  });
  if (res.status === 204) return null;
  return res.json();
}

// Navigation
document.querySelectorAll('.nav-item').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));
    document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
    btn.classList.add('active');
    document.getElementById('view-' + btn.dataset.view).classList.add('active');
    document.querySelector('.sidebar').classList.remove('open');

    // Refresh data on view switch
    const view = btn.dataset.view;
    if (view === 'planner') loadPlan();
    if (view === 'fridge') loadFridge();
    if (view === 'suggestions') loadSuggestions();
    if (view === 'shopping') loadShopping();
    if (view === 'dashboard') loadDashboard();
    if (view === 'generator') initGenerator();
    if (view === 'settings') loadBringSettings();
  });
});

// Initial load
async function init() {
  categories = await api('/categories');
  meals = await api('/meals');
  tags = await api('/tags');
  renderCategoryFilters();
  renderTagFilters();
  renderMeals();
  loadDashboard();
}

// Dashboard
async function loadDashboard() {
  fridge = await api('/fridge');
  plan = await api('/plan?week_offset=0');
  shoppingData = await api('/shopping-list?week_offset=0');
  shoppingList = shoppingData ? shoppingData.combined : [];
  const suggestions = await api('/suggestions');

  document.getElementById('stat-meals').textContent = meals.length;
  document.getElementById('stat-fridge').textContent = fridge.length;
  document.getElementById('stat-planned').textContent = plan.length;
  document.getElementById('stat-shopping').textContent = (shoppingList || []).filter(i => !i.in_fridge).length;

  // Week preview
  const wp = document.getElementById('dash-week-preview');
  if (plan.length === 0) {
    wp.innerHTML = '<p style="color:var(--text-muted);font-size:14px">Noch nichts geplant. Gehe zum Wochenplan!</p>';
  } else {
    wp.innerHTML = plan.map(p => `
      <div style="display:flex;justify-content:space-between;padding:6px 0;border-bottom:1px solid var(--border);font-size:14px">
        <span>${DAY_SHORT[p.day_of_week]}</span>
        <span style="font-weight:500">${esc(p.meal_name)}</span>
        <span style="color:var(--text-muted)">${p.servings}P</span>
      </div>
    `).join('');
  }

  // Top suggestions
  const sg = document.getElementById('dash-suggestions');
  if (suggestions.length === 0) {
    sg.innerHTML = '<p style="color:var(--text-muted);font-size:14px">Füge Zutaten zum Kühlschrank hinzu, um Vorschläge zu erhalten.</p>';
  } else {
    sg.innerHTML = suggestions.slice(0, 5).map(s => `
      <div style="display:flex;justify-content:space-between;padding:6px 0;border-bottom:1px solid var(--border);font-size:14px">
        <span style="font-weight:500">${esc(s.name)}</span>
        <span style="color:${s.match_percentage >= 75 ? 'var(--green)' : s.match_percentage >= 50 ? 'var(--yellow)' : 'var(--orange)'}">${s.match_percentage}%</span>
      </div>
    `).join('');
  }
}

// Meals
function renderCategoryFilters() {
  const sel = document.getElementById('meal-cat-filter');
  const modalSel = document.getElementById('meal-category');
  categories.forEach(c => {
    sel.innerHTML += `<option value="${c.id}">${c.icon} ${c.name}</option>`;
    modalSel.innerHTML += `<option value="${c.id}">${c.icon} ${c.name}</option>`;
  });
}

function renderTagFilters() {
  const sel = document.getElementById('meal-tag-filter');
  tags.forEach(t => {
    sel.innerHTML += `<option value="${t}">${t}</option>`;
  });
}

function renderMeals() {
  const search = document.getElementById('meal-search').value.toLowerCase();
  const catFilter = document.getElementById('meal-cat-filter').value;
  const tagFilter = document.getElementById('meal-tag-filter').value;

  let filtered = meals.filter(m => {
    if (search && !m.name.toLowerCase().includes(search)) return false;
    if (catFilter && m.category_id !== parseInt(catFilter)) return false;
    if (tagFilter && !m.tags.includes(tagFilter)) return false;
    return true;
  });

  // Show/hide "fill missing" button
  const missingCount = meals.filter(m => m.ingredients.length === 0).length;
  const fillBtn = document.getElementById('fill-missing-btn');
  if (fillBtn) {
    fillBtn.style.display = missingCount > 0 ? 'inline-flex' : 'none';
    fillBtn.textContent = `Fehlende Zutaten ergänzen (${missingCount})`;
  }

  const grid = document.getElementById('meal-grid');
  grid.innerHTML = filtered.map(m => `
    <div class="meal-card" onclick="openMealModal(${m.id})">
      <div class="meal-card-header">
        <div>
          <div class="meal-card-name">${esc(m.name)}</div>
          <div class="meal-card-category">${m.category_icon} ${esc(m.category_name)}</div>
        </div>
        <span style="font-size:12px;color:var(--text-muted)">${m.default_servings}P</span>
      </div>
      ${m.ingredients.length ? `<div class="ing-count">${m.ingredients.length} Zutaten</div>` : '<div class="ing-count" style="color:var(--yellow)">Keine Zutaten</div>'}
      <div class="meal-card-tags">
        ${m.tags.map(t => `<span class="tag tag-${t.toLowerCase()}">${t}</span>`).join('')}
      </div>
    </div>
  `).join('');
}

// Meal Modal
function openMealModal(id) {
  const modal = document.getElementById('meal-modal');
  const title = document.getElementById('meal-modal-title');
  const deleteBtn = document.getElementById('meal-delete-btn');
  document.getElementById('meal-edit-id').value = '';
  document.getElementById('meal-name').value = '';
  document.getElementById('meal-category').value = categories[0]?.id || '';
  document.getElementById('meal-servings').value = 4;
  document.getElementById('meal-tags').value = '';
  document.getElementById('meal-instructions').value = '';
  document.getElementById('meal-ingredients-list').innerHTML = '';

  if (id) {
    const m = meals.find(x => x.id === id);
    if (!m) return;
    title.textContent = 'Gericht bearbeiten';
    deleteBtn.style.display = 'inline-flex';
    document.getElementById('meal-edit-id').value = id;
    document.getElementById('meal-name').value = m.name;
    document.getElementById('meal-category').value = m.category_id || '';
    document.getElementById('meal-servings').value = m.default_servings;
    document.getElementById('meal-tags').value = m.tags.join(', ');
    document.getElementById('meal-instructions').value = m.instructions;
    m.ingredients.forEach(i => addIngredientRow(i.name, i.amount, i.unit));
  } else {
    title.textContent = 'Neues Gericht';
    deleteBtn.style.display = 'none';
    addIngredientRow();
  }
  modal.classList.add('open');
}

function closeMealModal() {
  document.getElementById('meal-modal').classList.remove('open');
}

function addIngredientRow(name = '', amount = '', unit = '') {
  const list = document.getElementById('meal-ingredients-list');
  const row = document.createElement('div');
  row.className = 'ing-row';
  row.innerHTML = `
    <div class="autocomplete-wrapper ing-name">
      <input class="input" placeholder="Zutat" value="${esc(name)}" oninput="ingAutocomplete(this)">
      <div class="autocomplete-list"></div>
    </div>
    <input class="input ing-amount" placeholder="Menge" value="${esc(amount)}">
    <input class="input ing-unit" placeholder="Einheit" value="${esc(unit)}">
    <button class="ing-remove" onclick="this.parentElement.remove()">&times;</button>
  `;
  list.appendChild(row);
}

async function saveMeal() {
  const id = document.getElementById('meal-edit-id').value;
  const data = {
    name: document.getElementById('meal-name').value.trim(),
    category_id: parseInt(document.getElementById('meal-category').value) || null,
    default_servings: parseInt(document.getElementById('meal-servings').value) || 4,
    tags: document.getElementById('meal-tags').value.split(',').map(t => t.trim()).filter(Boolean),
    instructions: document.getElementById('meal-instructions').value.trim(),
    ingredients: [],
  };

  document.querySelectorAll('#meal-ingredients-list .ing-row').forEach(row => {
    const name = row.querySelector('.ing-name input').value.trim();
    if (name) {
      data.ingredients.push({
        name,
        amount: row.querySelector('.ing-amount').value.trim(),
        unit: row.querySelector('.ing-unit').value.trim(),
      });
    }
  });

  if (!data.name) return alert('Bitte Name eingeben');

  if (id) {
    await api('/meals/' + id, { method: 'PUT', body: data });
  } else {
    await api('/meals', { method: 'POST', body: data });
  }

  meals = await api('/meals');
  tags = await api('/tags');
  renderMeals();
  closeMealModal();
}

async function deleteMeal() {
  const id = document.getElementById('meal-edit-id').value;
  if (!id || !confirm('Gericht wirklich löschen?')) return;
  await api('/meals/' + id, { method: 'DELETE' });
  meals = await api('/meals');
  renderMeals();
  closeMealModal();
}

// Autocomplete for ingredients
async function ingAutocomplete(input) {
  const val = input.value.trim();
  const list = input.parentElement.querySelector('.autocomplete-list');
  if (val.length < 1) { list.classList.remove('open'); return; }
  const results = await api('/ingredients?q=' + encodeURIComponent(val));
  if (results.length === 0) { list.classList.remove('open'); return; }
  list.innerHTML = results.map(r => `<div class="autocomplete-item" onmousedown="selectAC(this, '${esc(r)}')">${esc(r)}</div>`).join('');
  list.classList.add('open');
  input.addEventListener('blur', () => setTimeout(() => list.classList.remove('open'), 150), { once: true });
}

function selectAC(el, val) {
  const input = el.closest('.autocomplete-wrapper').querySelector('input');
  input.value = val;
  el.closest('.autocomplete-list').classList.remove('open');
}

// Weekly Plan
async function loadPlan() {
  plan = await api('/plan?week_offset=' + currentWeekOffset);
  const label = document.getElementById('week-label');
  if (currentWeekOffset === 0) label.textContent = 'Diese Woche';
  else if (currentWeekOffset === 1) label.textContent = 'Nächste Woche';
  else if (currentWeekOffset === -1) label.textContent = 'Letzte Woche';
  else label.textContent = `Woche ${currentWeekOffset > 0 ? '+' : ''}${currentWeekOffset}`;
  renderPlan();
}

function changeWeek(delta) {
  currentWeekOffset += delta;
  loadPlan();
}

function renderPlan() {
  const grid = document.getElementById('week-grid');
  grid.innerHTML = DAYS.map((day, i) => {
    const dayMeals = plan.filter(p => p.day_of_week === i);
    return `
      <div class="day-col">
        <div class="day-header">${day}${dayMeals.length ? `<button class="plan-meal-remove" onclick="clearDay(${i})" title="Tag leeren" style="float:right">&times;</button>` : ''}</div>
        ${dayMeals.map(p => `
          <div class="plan-meal">
            <div class="plan-meal-type">${p.meal_type === 'lunch' ? 'Mittag' : 'Abend'}</div>
            <div class="plan-meal-name">${esc(p.meal_name)}</div>
            <div class="plan-meal-servings">
              <span class="servings-control">
                <button class="servings-btn" onclick="updatePlanServings(${p.id}, ${p.servings - 1}, ${p.day_of_week}, '${p.meal_type}')">&minus;</button>
                <span class="servings-val">${p.servings}P</span>
                <button class="servings-btn" onclick="updatePlanServings(${p.id}, ${p.servings + 1}, ${p.day_of_week}, '${p.meal_type}')">+</button>
              </span>
            </div>
            <button class="plan-meal-remove" onclick="removePlan(${p.id})">&times;</button>
          </div>
        `).join('')}
        <button class="day-add" onclick="openPlanModal(${i})">+ Gericht</button>
      </div>
    `;
  }).join('');
}

async function updatePlanServings(planId, servings, dayOfWeek, mealType) {
  if (servings < 1) return;
  await api('/plan/' + planId, { method: 'PUT', body: { day_of_week: dayOfWeek, meal_type: mealType, servings } });
  loadPlan();
}

async function generateShoppingList() {
  // Load plan, check which meals have no ingredients
  plan = await api('/plan?week_offset=' + currentWeekOffset);
  if (plan.length === 0) {
    alert('Kein Wochenplan vorhanden. Erstelle zuerst einen Plan.');
    return;
  }

  // Refresh meals to get latest ingredient data
  meals = await api('/meals');

  // Find unique meals in plan that have no ingredients
  const planMealIds = [...new Set(plan.map(p => p.meal_id))];
  const missing = [];
  for (const mealId of planMealIds) {
    const m = meals.find(x => x.id === mealId);
    if (m && m.ingredients.length === 0) {
      missing.push(m);
    }
  }

  if (missing.length === 0) {
    // All good, navigate to shopping view and load
    await navigateToShopping();
    return;
  }

  // Start populate flow
  populateQueue = [...missing];
  showNextPopulate();
}

async function navigateToShopping() {
  shoppingData = await api('/shopping-list?week_offset=' + currentWeekOffset);
  shoppingList = shoppingData ? shoppingData.combined : [];
  // Switch to shopping view
  document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));
  document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));
  document.querySelector('.nav-item[data-view="shopping"]').classList.add('active');
  document.getElementById('view-shopping').classList.add('active');
  renderShoppingList();
}

function showNextPopulate() {
  if (populateQueue.length === 0) {
    // Done, refresh everything
    closePopulateModal();
    finishPopulate();
    return;
  }

  const meal = populateQueue[0];
  document.getElementById('populate-meal-id').value = meal.id;
  document.getElementById('populate-modal-title').textContent = `Zutaten für: ${meal.name}`;
  document.getElementById('populate-modal-desc').textContent =
    `Dieses Gericht hat noch keine Zutaten. Bitte ergänze die Zutaten, damit die Einkaufsliste vollständig ist. (Noch ${populateQueue.length} Gerichte übrig)`;
  document.getElementById('populate-ingredients-list').innerHTML = '';
  popAddIngRow();
  popAddIngRow();
  popAddIngRow();
  document.getElementById('populate-modal').classList.add('open');
}

  document.getElementById('populate-modal').classList.remove('open');
}

async function popSaveIngredients() {
  const mealId = parseInt(document.getElementById('populate-meal-id').value);
  const meal = meals.find(m => m.id === mealId);
  if (!meal) return;

  const ingredients = [];
  document.querySelectorAll('#populate-ingredients-list .ing-row').forEach(row => {
    const name = row.querySelector('.ing-name input').value.trim();
    if (name) {
      ingredients.push({
        name,
        amount: row.querySelector('.ing-amount').value.trim(),
        unit: row.querySelector('.ing-unit').value.trim(),
      });
    }
  });

  if (ingredients.length > 0) {
    // Update the meal with new ingredients (keep existing data)
    await api('/meals/' + mealId, {
      method: 'PUT',
      body: {
        name: meal.name,
        category_id: meal.category_id,
        default_servings: meal.default_servings,
        tags: meal.tags,
        instructions: meal.instructions,
        ingredients: ingredients,
      },
    });
    // Refresh meals
    meals = await api('/meals');
  }

  // Move to next
  populateQueue.shift();
  showNextPopulate();
}

let genConfig = []; // [{day, dayName, lunch: bool, lunchServings, dinner: bool, dinnerServings}]
let generatedPlan = []; // results from API

function initGenerator() {
  // Build config with defaults: all dinners enabled, all lunches enabled except Monday(0) and Thursday(3)
  const NO_LUNCH_DAYS = [0, 3]; // Monday, Thursday
  genConfig = DAYS.map((name, i) => ({
    day: i,
    dayName: name,
    lunch: !NO_LUNCH_DAYS.includes(i),
    lunchServings: 4,
    dinner: true,
    dinnerServings: 4,
  }));
  renderGenConfig();
  // Reset to step 1
  document.getElementById('gen-step-config').style.display = '';
  document.getElementById('gen-step-review').style.display = 'none';
}

function renderGenConfig() {
  const tbody = document.getElementById('gen-config-body');
  tbody.innerHTML = genConfig.map(c => `
    <tr style="border-bottom:1px solid var(--border)">
      <td style="padding:8px;font-weight:500">${c.dayName}</td>
      <td style="text-align:center;padding:8px">
        <input type="checkbox" class="gen-check" ${c.lunch ? 'checked' : ''} onchange="genConfig[${c.day}].lunch=this.checked">
      </td>
      <td style="text-align:center;padding:8px">
        <input type="number" class="input gen-servings" value="${c.lunchServings}" min="1" ${c.lunch ? '' : 'disabled'}
          onchange="genConfig[${c.day}].lunchServings=parseInt(this.value)||4"
          data-gen-lunch-servings="${c.day}">
      </td>
      <td style="text-align:center;padding:8px">
        <input type="checkbox" class="gen-check" ${c.dinner ? 'checked' : ''} onchange="genConfig[${c.day}].dinner=this.checked">
      </td>
      <td style="text-align:center;padding:8px">
        <input type="number" class="input gen-servings" value="${c.dinnerServings}" min="1" ${c.dinner ? '' : 'disabled'}
          onchange="genConfig[${c.day}].dinnerServings=parseInt(this.value)||4"
          data-gen-dinner-servings="${c.day}">
      </td>
    </tr>
  `).join('');

  // Wire up checkbox to enable/disable servings input
  tbody.querySelectorAll('.gen-check').forEach(cb => {
    cb.addEventListener('change', () => {
      const row = cb.closest('tr');
      const inputs = row.querySelectorAll('.gen-servings');
      const checks = row.querySelectorAll('.gen-check');
      checks.forEach((ch, idx) => { inputs[idx].disabled = !ch.checked; });
    });
  });
}

function genSetAllServings() {
  const val = prompt('Portionen für alle Mahlzeiten:', '4');
  if (!val) return;
  const n = parseInt(val) || 4;
  genConfig.forEach(c => { c.lunchServings = n; c.dinnerServings = n; });
  renderGenConfig();
}

async function generatePlan() {
  // Build slots from config
  const slots = [];
  genConfig.forEach(c => {
    if (c.lunch) slots.push({ day_of_week: c.day, meal_type: 'lunch', servings: c.lunchServings });
    if (c.dinner) slots.push({ day_of_week: c.day, meal_type: 'dinner', servings: c.dinnerServings });
  });

  if (slots.length === 0) return alert('Keine Mahlzeiten ausgewählt!');

  generatedPlan = await api('/plan/generate', {
    method: 'POST',
    body: { slots, week_offset: currentWeekOffset },
  });

  // Switch to review step
  document.getElementById('gen-step-config').style.display = 'none';
  document.getElementById('gen-step-review').style.display = '';
  renderGenReview();
}

function renderGenReview() {
  const grid = document.getElementById('gen-review-grid');
  grid.innerHTML = DAYS.map((day, i) => {
    const dayMeals = generatedPlan.filter(p => p.day_of_week === i);
    if (dayMeals.length === 0) {
      return `
        <div class="day-col" style="opacity:0.5">
          <div class="day-header">${day}</div>
          <p style="font-size:12px;color:var(--text-muted);text-align:center;padding:20px 0">Keine Mahlzeit</p>
        </div>
      `;
    }
    return `
      <div class="day-col">
        <div class="day-header">${day}</div>
        ${dayMeals.map(p => `
          <div class="gen-review-meal">
            <div class="gen-review-meal-info">
              <div class="gen-review-meal-meta">${p.meal_type === 'lunch' ? 'Mittag' : 'Abend'} &bull; ${p.servings}P</div>
              <div class="gen-review-meal-name">${esc(p.meal_name)}</div>
            </div>
            <div class="gen-review-actions">
              <button class="btn btn-sm" onclick="openSwapModal(${p.id})" title="Austauschen">&#x21c4;</button>
              <button class="btn btn-sm btn-danger" onclick="genRemoveEntry(${p.id})" title="Entfernen">&times;</button>
            </div>
          </div>
        `).join('')}
      </div>
    `;
  }).join('');
}

async function sendToBring() {
  if (!shoppingData || !shoppingData.combined || shoppingData.combined.length === 0) {
    alert('Keine Einkaufsliste vorhanden. Bitte zuerst generieren.');
    return;
  }

  // Build items - combine amount + unit as spec
  const items = shoppingData.combined
    .filter(item => !item.in_fridge)  // Skip items already in fridge
    .map(item => ({
      name: item.name,
      spec: item.amount ? `${item.amount} ${item.unit}`.trim() : item.unit || '',
    }));

  if (items.length === 0) {
    alert('Alle Zutaten sind bereits im Kühlschrank!');
    return;
  }

  const btn = document.getElementById('bring-send-btn');
  const origText = btn.textContent;
  btn.textContent = 'Sende...';
  btn.disabled = true;

  try {
    const result = await api('/bring/send', { method: 'POST', body: { items } });
    if (result.error) {
      alert('Fehler: ' + result.error);
    } else {
      btn.textContent = `${result.count} Artikel gesendet!`;
      btn.style.background = 'var(--green)';
      btn.style.color = 'white';
      setTimeout(() => {
        btn.textContent = origText;
        btn.style.background = 'var(--green-bg)';
        btn.style.color = 'var(--green)';
        btn.disabled = false;
      }, 3000);
      return;
    }
  } catch (e) {
    alert('Fehler beim Senden. Bring! Einstellungen prüfen.');
  }
  btn.textContent = origText;
  btn.disabled = false;
}

// Init
init();
</script>
