<script>
(function() {
  'use strict';

  // ========== PARTICLE SYSTEM ==========
  var canvas = document.getElementById('particle-canvas');
  var ctx = canvas.getContext('2d');
  var particles = [];
  var w, h;

  function resizeCanvas() {
    w = canvas.width = window.innerWidth;
    h = canvas.height = window.innerHeight;
  }
  resizeCanvas();
  window.addEventListener('resize', resizeCanvas);

  function Particle() { this.reset(); }
  Particle.prototype.reset = function() {
    this.x = Math.random() * w;
    this.y = Math.random() * h;
    this.size = Math.random() * 1.5 + 0.3;
    this.speedX = (Math.random() - 0.5) * 0.3;
    this.speedY = (Math.random() - 0.5) * 0.3;
    this.opacity = Math.random() * 0.3 + 0.05;
    this.fadeDir = Math.random() > 0.5 ? 1 : -1;
  };
  Particle.prototype.update = function() {
    this.x += this.speedX;
    this.y += this.speedY;
    this.opacity += this.fadeDir * 0.001;
    if (this.opacity > 0.35) this.fadeDir = -1;
    if (this.opacity < 0.03) this.fadeDir = 1;
    if (this.x < -10 || this.x > w + 10 || this.y < -10 || this.y > h + 10) this.reset();
  };
  Particle.prototype.draw = function() {
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
    ctx.fillStyle = 'rgba(201, 160, 80, ' + this.opacity + ')';
    ctx.fill();
  };

  var particleCount = Math.min(80, Math.floor(w * h / 15000));
  for (var i = 0; i < particleCount; i++) particles.push(new Particle());

  function animateParticles() {
    ctx.clearRect(0, 0, w, h);
    for (var i = 0; i < particles.length; i++) {
      particles[i].update();
      particles[i].draw();
    }
    // Draw connections
    for (var i = 0; i < particles.length; i++) {
      for (var j = i + 1; j < particles.length; j++) {
        var dx = particles[i].x - particles[j].x;
        var dy = particles[i].y - particles[j].y;
        var dist = Math.sqrt(dx * dx + dy * dy);
        if (dist < 150) {
          ctx.beginPath();
          ctx.moveTo(particles[i].x, particles[i].y);
          ctx.lineTo(particles[j].x, particles[j].y);
          ctx.strokeStyle = 'rgba(201, 160, 80, ' + (0.03 * (1 - dist / 150)) + ')';
          ctx.lineWidth = 0.5;
          ctx.stroke();
        }
      }
    }
    requestAnimationFrame(animateParticles);
  }
  animateParticles();

  // ========== NAV SCROLL ==========
  var navEl = document.getElementById('main-nav');
  window.addEventListener('scroll', function() {
    if (window.scrollY > 60) {
      navEl.classList.add('scrolled');
    } else {
      navEl.classList.remove('scrolled');
    }
  });

  // ========== SCROLL REVEAL ==========
  var observerOptions = { threshold: 0.15, rootMargin: '0px 0px -50px 0px' };
  var observer = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        var delay = parseInt(entry.target.getAttribute('data-delay')) || 0;
        setTimeout(function() { entry.target.classList.add('visible'); }, delay);
        observer.unobserve(entry.target);
      }
    });
  }, observerOptions);

  var revealEls = document.querySelectorAll('.product-card, .about-card, .reveal');
  for (var i = 0; i < revealEls.length; i++) observer.observe(revealEls[i]);

  // ========== BLEND BAR ANIMATION ==========
  var blendObserver = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        entry.target.style.width = entry.target.getAttribute('data-width') + '%';
        blendObserver.unobserve(entry.target);
      }
    });
  }, { threshold: 0.5 });

  var blendFills = document.querySelectorAll('.blend-bar-fill');
  for (var i = 0; i < blendFills.length; i++) blendObserver.observe(blendFills[i]);

  // ========== ANNOUNCEMENT ==========
  function loadAnnouncement() {
    var ann = localStorage.getItem('oryx_announcement');
    var bar = document.getElementById('announcement-bar');
    if (ann && ann.trim()) {
      bar.textContent = ann;
      bar.style.display = 'block';
      document.getElementById('main-nav').style.top = '36px';
    } else {
      bar.style.display = 'none';
      document.getElementById('main-nav').style.top = '0';
    }
  }
  loadAnnouncement();

  // ========== ANALYTICS (localStorage) ==========
  function incStat(key) {
    var val = parseInt(localStorage.getItem(key) || '0') + 1;
    localStorage.setItem(key, String(val));
    return val;
  }
  incStat('oryx_visits');

  // Expose globals
  window.enterSite = function() {
    document.getElementById('age-gate').classList.add('hidden');
    document.body.style.overflow = '';
    incStat('oryx_age_pass');
    sessionStorage.setItem('oryx_age_verified', '1');
  };

  window.denySite = function() {
    window.location.href = 'https://www.google.com';
  };

  // Check if already verified this session
  if (sessionStorage.getItem('oryx_age_verified') === '1') {
    document.getElementById('age-gate').classList.add('hidden');
  } else {
    document.body.style.overflow = 'hidden';
  }

  // ========== MOBILE MENU ==========
  window.toggleMobile = function() {
    document.getElementById('mobile-menu').classList.toggle('active');
  };

  // ========== NEWSLETTER ==========
  window.handleNewsletter = function(e) {
    e.preventDefault();
    var email = document.getElementById('newsletter-email').value;
    if (!email) return false;

    var subs = JSON.parse(localStorage.getItem('oryx_subscribers') || '[]');
    for (var i = 0; i < subs.length; i++) {
      if (subs[i].email === email) {
        showToast('You are already subscribed!');
        return false;
      }
    }
    subs.push({ email: email, date: new Date().toISOString().split('T')[0] });
    localStorage.setItem('oryx_subscribers', JSON.stringify(subs));

    document.querySelector('.newsletter-form').style.display = 'none';
    document.getElementById('newsletter-success').style.display = 'block';
    showToast('Successfully subscribed!');
    return false;
  };

  // ========== TOAST ==========
  window.showToast = function(msg) {
    var t = document.getElementById('toast');
    t.textContent = msg;
    t.classList.add('show');
    setTimeout(function() { t.classList.remove('show'); }, 3000);
  };

  // ========== ADMIN ==========
  var ADMIN_PASS = 'oryx2026';

  window.promptAdmin = function() {
    var pw = prompt('Enter admin password:');
    if (pw === ADMIN_PASS) {
      openAdmin();
    } else if (pw !== null) {
      showToast('Incorrect password.');
    }
  };

  function openAdmin() {
    var panel = document.getElementById('admin-panel');
    panel.classList.add('active');

    document.getElementById('admin-visits').textContent = localStorage.getItem('oryx_visits') || '0';
    document.getElementById('admin-age-pass').textContent = localStorage.getItem('oryx_age_pass') || '0';

    var subs = JSON.parse(localStorage.getItem('oryx_subscribers') || '[]');
    document.getElementById('admin-subs-count').textContent = String(subs.length);

    var tbody = document.getElementById('admin-subs-list');
    // Clear existing rows safely
    while (tbody.firstChild) {
      tbody.removeChild(tbody.firstChild);
    }

    if (subs.length === 0) {
      var emptyRow = document.createElement('tr');
      var emptyCell = document.createElement('td');
      emptyCell.setAttribute('colspan', '2');
      emptyCell.style.color = 'var(--text-dim)';
      emptyCell.style.fontStyle = 'italic';
      emptyCell.textContent = 'No subscribers yet';
      emptyRow.appendChild(emptyCell);
      tbody.appendChild(emptyRow);
    } else {
      for (var i = 0; i < subs.length; i++) {
        var tr = document.createElement('tr');
        var tdEmail = document.createElement('td');
        tdEmail.textContent = subs[i].email;
        var tdDate = document.createElement('td');
        tdDate.textContent = subs[i].date;
        tr.appendChild(tdEmail);
        tr.appendChild(tdDate);
        tbody.appendChild(tr);
      }
    }

    document.getElementById('admin-announcement').value = localStorage.getItem('oryx_announcement') || '';
  }

  window.closeAdmin = function() {
    document.getElementById('admin-panel').classList.remove('active');
  };

  window.saveAnnouncement = function() {
    var text = document.getElementById('admin-announcement').value;
    localStorage.setItem('oryx_announcement', text);
    loadAnnouncement();
    showToast('Announcement saved.');
  };

  // Keyboard shortcut: Ctrl+Shift+A for admin
  document.addEventListener('keydown', function(e) {
    if (e.ctrlKey && e.shiftKey && e.key === 'A') {
      e.preventDefault();
      promptAdmin();
    }
  });

})();
