    <script>
      document.body.style.opacity = '1';

      function getUrlParam(name) {
        const params = new URLSearchParams(window.location.search);
        return params.get(name) === 'true';
      }

      function dashboard() {
        return {
          darkMode: localStorage.getItem('darkMode') !== 'false',
          autoRefresh: localStorage.getItem('autoRefresh') === 'true',
          countdown:  300 ,
          countdownInterval: null,
          search: '',
          searchOpen: false,
          filter: localStorage.getItem('filter') || 'all',
          sort: localStorage.getItem('sort') || 'default',
          toasts: [],
          initialLoad: true,

          
          showHeader: !getUrlParam('hideHeader'),
          showStats: !getUrlParam('hideStats'),
          showServersHeader: !getUrlParam('hideServersHeader'),
          showControls: !getUrlParam('hideControls'),
          showProxies: !getUrlParam('hideProxies'),
          showStatusInfo: !getUrlParam('hideStatusInfo'),
          showFooter: !getUrlParam('hideFooter'),
          hideBackground: getUrlParam('hideBackground'),

          
          badgeMode: false,
          badgeProxy: null,
          badgeTheme: 'dark',
          badgeStyle: '',
          badgeContainerStyle: '',
          badgeShowName: true,
          badgeShowLatency: true,
          badgeVariant: 'default',
          badgeSize: 'md',
          badgeRounded: 'md',
          badgeTransparent: false,

          get badgeClasses() {
            const classes = [];
            if (this.badgeVariant === 'flat') classes.push('badge-style-flat');
            if (this.badgeVariant === 'pill') classes.push('badge-style-pill');
            if (this.badgeVariant === 'dot') classes.push('badge-dot-only');
            if (this.badgeSize === 'sm') classes.push('badge-size-sm');
            if (this.badgeSize === 'lg') classes.push('badge-size-lg');
            if (this.badgeVariant !== 'pill' && this.badgeVariant !== 'dot') {
              classes.push('badge-rounded-' + this.badgeRounded);
            }
            if (this.badgeTransparent) classes.push('badge-transparent');
            return classes.join(' ');
          },

          proxies: [
            
            { name: "🇩🇪Germany-XHTTP", stableId: "e3942bbd352c0dea", index:  0 , status:  true , latency: "180ms", latencyMs:  180  },
            
            { name: "🇩🇪Germany-VLESS", stableId: "fb8730a9fc5fdf96", index:  1 , status:  true , latency: "244ms", latencyMs:  244  },

            { name: "🇸🇪Sweden-XHTTP", stableId: "7785e6cccec1acad", index:  3 , status:  true , latency: "197ms", latencyMs:  197  },
            
            { name: "🇸🇪Sweden-VLESS", stableId: "1ec187ea8c1e5e6b", index:  4 , status:  true , latency: "335ms", latencyMs:  335  },
            
            { name: "🇸🇪Sweden-GRPC", stableId: "4539cb389295a575", index:  5 , status:  true , latency: "410ms", latencyMs:  410  },
            
            { name: "🇪🇪Estonia-VLESS", stableId: "9661795c73ed0205", index:  6 , status:  true , latency: "883ms", latencyMs:  883  },
            
            { name: "🇪🇪Estonia-XHTTP", stableId: "b6f276808802ba33", index:  7 , status:  true , latency: "258ms", latencyMs:  258  },
            
            { name: "🇪🇪Estonia-GRPC", stableId: "6a0d4899f6e4b97d", index:  8 , status:  true , latency: "246ms", latencyMs:  246  },
            
            { name: "🇬🇧 United Kingdom", stableId: "9e6b0831116ac3d8", index:  9 , status:  true , latency: "191ms", latencyMs:  191  },
            
          ],

          get stats() {
            const online = this.proxies.filter(p => p.status).length;
            const withLatency = this.proxies.filter(p => p.status && p.latencyMs > 0);
            return {
              total: this.proxies.length,
              online,
              offline: this.proxies.length - online,
              avgLatency: withLatency.length ? Math.round(withLatency.reduce((s, p) => s + p.latencyMs, 0) / withLatency.length) : 0
            };
          },

          get filteredProxies() {
            let r = this.proxies;
            if (this.search) {
              const q = this.search.toLowerCase();
              r = r.filter(p => p.name.toLowerCase().includes(q) || (p.serverInfo && p.serverInfo.toLowerCase().includes(q)));
            }
            if (this.filter === 'online') r = r.filter(p => p.status);
            else if (this.filter === 'offline') r = r.filter(p => !p.status);

            return [...r].sort((a, b) => {
              if (this.sort === 'latency-asc') {
                if (a.latencyMs === 0 && b.latencyMs === 0) return 0;
                if (a.latencyMs === 0) return 1;
                if (b.latencyMs === 0) return -1;
                return a.latencyMs - b.latencyMs;
              }
              if (this.sort === 'latency-desc') {
                if (a.latencyMs === 0 && b.latencyMs === 0) return 0;
                if (a.latencyMs === 0) return 1;
                if (b.latencyMs === 0) return -1;
                return b.latencyMs - a.latencyMs;
              }
              if (this.sort === 'status') return b.status - a.status;
              if (this.sort === 'name') return this.stripLeadingEmoji(a.name).localeCompare(this.stripLeadingEmoji(b.name));
              return a.index - b.index;
            });
          },

          init() {
            const params = new URLSearchParams(window.location.search);

            
            const stableId = params.get('stableId');
            if (stableId) {
              this.badgeProxy = this.proxies.find(p => p.stableId === stableId);
              if (this.badgeProxy) {
                this.badgeMode = true;

                
                const theme = params.get('theme');
                if (theme === 'light' || theme === 'dark') {
                  this.badgeTheme = theme;
                  this.darkMode = theme === 'dark';
                }

                
                const variant = params.get('variant');
                if (['default', 'flat', 'pill', 'dot'].includes(variant)) {
                  this.badgeVariant = variant;
                  if (variant === 'dot') {
                    this.badgeShowName = false;
                    this.badgeShowLatency = false;
                  }
                }

                
                const size = params.get('size');
                if (['sm', 'md', 'lg'].includes(size)) {
                  this.badgeSize = size;
                }

                
                const rounded = params.get('rounded');
                if (['none', 'sm', 'md', 'lg', 'full'].includes(rounded)) {
                  this.badgeRounded = rounded;
                }

                
                if (params.get('showName') === 'false') {
                  this.badgeShowName = false;
                }

                
                if (params.get('showLatency') === 'false') {
                  this.badgeShowLatency = false;
                }

                
                if (params.get('transparent') === 'true') {
                  this.badgeTransparent = true;
                }

                
                const styles = [];
                const width = params.get('width');
                const height = params.get('height');
                if (width) styles.push('width: ' + width + (isNaN(width) ? '' : 'px'));
                if (height) styles.push('height: ' + height + (isNaN(height) ? '' : 'px'));
                if (styles.length) this.badgeStyle = styles.join('; ');

                
                const containerStyles = [];
                const align = params.get('align');
                if (align === 'center') containerStyles.push('justify-content: center');
                if (align === 'right') containerStyles.push('justify-content: flex-end');
                if (width || height) containerStyles.push('width: 100%');
                if (containerStyles.length) this.badgeContainerStyle = containerStyles.join('; ');
              }
            }

            if (this.autoRefresh) this.startCountdown();
            setTimeout(() => this.initialLoad = false, 500);
          },

          toggleDarkMode() {
            this.darkMode = !this.darkMode;
            localStorage.setItem('darkMode', this.darkMode);
          },

          toggleAutoRefresh() {
            this.autoRefresh = !this.autoRefresh;
            localStorage.setItem('autoRefresh', this.autoRefresh);
            this.autoRefresh ? this.startCountdown() : this.stopCountdown();
          },

          setFilter(f) { this.filter = f; localStorage.setItem('filter', f); },
          saveSort() { localStorage.setItem('sort', this.sort); },

          getSegments(ms) {
            if (ms === 0) return 0;
            if (ms < 200) return 5;
            if (ms < 500) return 4;
            if (ms < 800) return 3;
            if (ms < 1200) return 2;
            return 1;
          },

          stripLeadingEmoji(str) {
            return str.replace(/^[\p{Emoji}\p{Emoji_Modifier}\p{Emoji_Component}\p{Emoji_Modifier_Base}\p{Emoji_Presentation}\u200d\ufe0f\s]+/u, '').trim() || str;
          },

          startCountdown() {
            this.countdown =  300 ;
            this.countdownInterval = setInterval(() => {
              if (--this.countdown <= 0) this.refreshData();
            }, 1000);
          },

          stopCountdown() {
            clearInterval(this.countdownInterval);
            this.countdownInterval = null;
            this.countdown =  300 ;
          },

          async refreshData() {
            try {
              const res = await fetch('./api/v1/public/proxies');
              const json = await res.json();
              if (json.success && json.data) {
                json.data.forEach(updated => {
                  const proxy = this.proxies.find(p => p.name === updated.name);
                  if (proxy) {
                    proxy.status = updated.online;
                    proxy.latencyMs = updated.latencyMs;
                    proxy.latency = updated.latencyMs > 0 ? updated.latencyMs + 'ms' : 'n/a';
                  }
                });
                
                if (this.badgeMode && this.badgeProxy) {
                  const updated = json.data.find(p => p.stableId === this.badgeProxy.stableId);
                  if (updated) {
                    this.badgeProxy.status = updated.online;
                    this.badgeProxy.latencyMs = updated.latencyMs;
                    this.badgeProxy.latency = updated.latencyMs > 0 ? updated.latencyMs + 'ms' : 'n/a';
                  }
                }
              }
            } catch (e) {
              console.error('Failed to refresh:', e);
            }
            this.countdown =  300 ;
          },

          async copyUrl(proxy) {
            const url = new URL(proxy.url, location.href).toString();
            try { await navigator.clipboard.writeText(url); }
            catch { const t = Object.assign(document.createElement('textarea'), {value: url, style: 'position:fixed;opacity:0'}); document.body.appendChild(t); t.select(); document.execCommand('copy'); t.remove(); }
            this.showToast('Copied!');
          },

          showToast(msg) {
            const t = { id: Date.now(), message: msg, visible: true };
            this.toasts.push(t);
            setTimeout(() => { t.visible = false; setTimeout(() => this.toasts = this.toasts.filter(x => x.id !== t.id), 200); }, 2000);
          }
        };
      }
