// план семейств: холодный первым, затем перемешанный баланс остальных
function planFamilies(n){
  const others=Object.keys(TPL).filter(k=>k!=='COLD');
  for(let i=others.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[others[i],others[j]]=[others[j],others[i]];}
  const seq=['COLD']; let pool=others.slice();
  while(seq.length<n){ if(!pool.length)pool=others.slice(); seq.push(pool.shift()); }
  return seq.slice(0,n);
}

    // коридор снизу вверх до дальней цели
    s+=`<line x1="${x(cx)}" y1="${y(0)}" x2="${x(cx)}" y2="${y(far)}" stroke="${col}" stroke-opacity=".26" stroke-width="1"/>`;
    // база коридора на рубеже (латеральная позиция упражнения) + номер
    s+=`<circle cx="${x(cx)}" cy="${y(0)}" r="3" fill="${col}"/>`;
    s+=`<text x="${x(cx)+6}" y="${y(0)-5}" fill="${col}" font-size="9" font-weight="700">${ex.no}</text>`;
    const seenD=new Set();
    ex.targets.forEach(t=>{
      let lat=t.lat||0; lat=Math.max(-(half-0.6),Math.min(half-0.6,lat));
      const tx=Math.max(1.6,Math.min(WID-1.6,cx+lat));
      const cxp=x(tx), cyp=y(t.d);
      s+=marker(t.shape,cxp,cyp,col);
      if(!seenD.has(t.d)){seenD.add(t.d);
        const rightmost=Math.max(...ex.targets.filter(o=>o.d===t.d).map(o=>Math.max(-(half-0.6),Math.min(half-0.6,o.lat||0))));
        s+=`<text class="g-dist" x="${x(Math.min(WID-1.6,cx+rightmost))+8}" y="${cyp+3}" font-size="8.5" text-anchor="start">${t.d}</text>`;}
      s+=`<circle class="tgt" cx="${cxp}" cy="${cyp}" r="8" fill="transparent" data-ex="${i}"/>`;
    });
  });
  s+=`</svg>`;
  return s;
}
function stageWidth(){
  const wrap=document.querySelector('.gallery-wrap');
  if(wrap){const cs=getComputedStyle(wrap);
    const w=wrap.clientWidth-parseFloat(cs.paddingLeft)-parseFloat(cs.paddingRight);
    if(w>240)return w;}
  return 720;
}

function tipHTML(ex){
  const f=FAM[ex.fam];
  const dists=ex.targets.map(t=>`${t.d}м/${t.size}см`).join(', ');
  const row=(k,v)=>`<tr><td style="padding:1px 8px 1px 0;color:var(--faint);white-space:nowrap">${k}</td><td>${esc(v)}</td></tr>`;
  return `<div style="display:flex;align-items:center;gap:7px;margin-bottom:5px">
    <span style="width:9px;height:9px;border-radius:50%;background:${f.c};flex:none"></span>
    <b>№${ex.no} · ${esc(ex.name)}</b></div>
    <div style="font-family:var(--mono);font-size:10.5px;color:var(--faint);text-transform:uppercase;letter-spacing:.04em;margin-bottom:6px">${esc(f.t)}</div>
    <table style="font-size:11.5px;color:var(--muted);border-collapse:collapse;font-variant-numeric:tabular-nums">
    ${row('цели',dists)}${row('выстр.',ex.shots)}${row('время',ex.time)}${row('позиция',ex.pos)}${row('очки',ex.pts)}</table>
    <div style="margin-top:7px;font-size:11.5px;line-height:1.42;color:var(--ink)">${esc(ex.proc)}</div>`;
}
function wireTips(){
  const tip=document.getElementById('genTip');
  document.querySelectorAll('#galleryStage .tgt').forEach(el=>{
    const show=e=>{tip.innerHTML=tipHTML(genState[+el.dataset.ex]);tip.style.display='block';move(e);};
    const move=e=>{const t=(e.touches&&e.touches[0])||e;
      tip.style.left=Math.min(t.clientX+14,innerWidth-290)+'px';
      tip.style.top=Math.min(t.clientY+14,innerHeight-tip.offsetHeight-10)+'px';};
    el.addEventListener('mouseenter',show); el.addEventListener('mousemove',move);
    el.addEventListener('mouseleave',()=>tip.style.display='none');
    el.addEventListener('touchstart',show,{passive:true});
  });
}

function buildBriefing(){
  const doc=document.getElementById('briefingDoc');
  const maxN=s=>{const m=String(s).match(/\d+/g);return m?Math.max(...m.map(Number)):0;};
  const shots=genState.reduce((a,e)=>a+maxN(e.shots),0);
  const pts=genState.reduce((a,e)=>a+(parseFloat(e.pts)||0),0);
  const ptsStr=(Math.round(pts*100)/100).toString();
  let h=`<div class="bf-head">
    <div class="bf-kicker">Брифинг матча · МК-снайпинг · .22LR</div>
    <div class="bf-title">План огня — ${genState.length} ${plural(genState.length,'упражнение','упражнения','упражнений')}</div>
    <div class="bf-meta">галерея 315 × 50 м · настрел ≈ ${shots} патр. · макс. ${ptsStr} очк.</div>
  </div>`;
  h+=`<div class="bf-plan">${galleryMarkup(700)}</div>`;
  h+=`<div class="bf-rules"><div class="bf-sec">Общие правила</div><ul>
    <li>Угол безопасности — 90°. Перенос и обслуживание оружия — только стволом в направлении мишеней, затвор открыт.</li>
    <li>Стартовое положение объявляется на каждом упражнении; по умолчанию — стоя на отметке, оружие разряжено, ствол вверх, затвор открыт, магазин присоединён.</li>
    <li>Начало по команде судьи «ОГОНЬ», окончание — по истечении времени или команде «СТОП».</li>
    <li>Поражение целей вне указанного порядка засчитывается как промах.</li>
    <li>Дальномеры разрешены; дистанции стрелок уточняет самостоятельно.</li>
    <li>За нарушение техники безопасности или регламента судья вправе назначить штраф.</li>
  </ul></div>`;
  h+=`<div class="bf-sec">Упражнения</div>`;
  h+=genState.map(ex=>{
    const f=FAM[ex.fam];
    const tg=ex.targets.map(t=>`${t.d} м / ${t.size} см`).join(' · ');
    const row=(k,v)=>`<div class="bf-row"><span class="bf-k">${k}</span><span class="bf-v">${esc(v)}</span></div>`;
    return `<div class="bf-ex">
      <div class="bf-ex-head">
        <span class="bf-no" style="background:${f.c}">${ex.no}</span>
        <div><div class="bf-ex-name">${esc(ex.name)}</div><div class="bf-ex-fam">${esc(f.t)}</div></div>
        <div class="bf-ex-pts">${esc(ex.pts)}<span>очк.</span></div>
      </div>
      <div class="bf-grid">
        ${row('Дистанция / цели',tg)}
        ${row('Выстрелов',ex.shots)}
        ${row('Время',ex.time)}
        ${row('Позиция',ex.pos)}
      </div>
      <div class="bf-proc"><span class="bf-k">Порядок</span>${esc(ex.proc)}</div>
    </div>`;
  }).join('');
  h+=`<div class="bf-foot">Сформировано конструктором матча · раскладка по принципам практического МК-снайпинга</div>`;
  doc.innerHTML=h;
}
function plural(n,a,b,c){const n1=n%10,n2=n%100;return n1===1&&n2!==11?a:(n1>=2&&n1<=4&&(n2<10||n2>=20)?b:c);}
