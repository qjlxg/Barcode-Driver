(function(){
    if (typeof jsyaml === 'undefined') {

    function toast(msg, type='success') {
        const t = document.createElement('div');
        t.className = `toast toast-${type}`;
        t.textContent = msg;
        document.getElementById('toastContainer').appendChild(t);
        setTimeout(() => t.remove(), 3000);
    }

    function copyTextToClipboard(text, successMsg) {
        if (!text) { toast('没有内容可复制', 'error'); return; }
        if (navigator.clipboard && window.isSecureContext) {
            navigator.clipboard.writeText(text).then(
                () => toast(successMsg),
                () => fallbackCopy(text, successMsg)
            );
        } else {
            fallbackCopy(text, successMsg);
        }
    }
    function fallbackCopy(text, successMsg) {
        const textarea = document.createElement('textarea');
        textarea.value = text;
        textarea.style.position = 'fixed';
        textarea.style.top = '0';
        textarea.style.left = '0';
        textarea.style.opacity = '0';
        document.body.appendChild(textarea);
        textarea.focus();
        textarea.select();
        try {
            const success = document.execCommand('copy');
            if (success) toast(successMsg);
            else toast('复制失败，请手动选择文本复制', 'error');
        } catch (err) {
            toast('复制出错，请手动复制', 'error');
        }
        document.body.removeChild(textarea);
    }

    async function checkLoginStatus() {

            const res = await fetch('/check-login');

    }

    async function authFetch(url, options = {}) {

    async function loadServerConfig() {
        try {
            const res = await authFetch('/get-config');
            if (!res) return;
            const data = await res.json();
            if (data.status === 'success' && data.yaml) {
                document.getElementById('yamlOutput').value = data.yaml;
                currentYaml = data.yaml;
                parseNodesFromYaml(data.yaml);
            } else {
                document.getElementById('yamlOutput').value = '';
                currentYaml = '';
                nodes = [];
                render();
                document.getElementById('dnsUpstream').value = '8.8.8.8:53';

    }

    // 从 YAML 解析节点，提取监听端口、备注以及 DNS 上游
    function parseNodesFromYaml(yamlText) {
        try {
            const config = jsyaml.load(yamlText);
            const chains = config?.chains || [];
            const services = config?.services || [];
            const newNodes = [];

            // 提取 DNS 上游地址（从第一个服务中获取）
            let dnsUpstream = '8.8.8.8:53';
            if (services.length) {
                const first = services[0];
                const dnsArr = first.handler?.dns;
                if (Array.isArray(dnsArr) && dnsArr.length && dnsArr[0].addr) {
                    dnsUpstream = dnsArr[0].addr;
                }
            }
            document.getElementById('dnsUpstream').value = dnsUpstream;

            const chainPortMap = {};
            services.forEach(s => {
                const portMatch = s.addr?.match(/:(\d+)$/);
                const port = portMatch ? parseInt(portMatch[1]) : null;
                const chain = s.handler?.chain;
                if (port && chain && !chainPortMap[chain]) {
                    chainPortMap[chain] = port;
                }
            });

            chains.forEach(chain => {
                const chainName = chain.name;
                const listenPort = chainPortMap[chainName] || '';
                chain.hops?.forEach(hop => {
                    hop.nodes?.forEach(node => {
                        if (node.connector?.type === 'socks5' && node.addr) {
                            newNodes.push({
                                addr: node.addr,
                                username: node.connector.auth?.username || '',
                                password: node.connector.auth?.password || '',
                                retry: node.connector.retry || 2,
                                remark: node.connector?.metadata?.remark || '',
                                listenPort: listenPort
                            });
                        }
                    });
                });
            });
            nodes = newNodes;
            render();

            const firstSS = services.find(s => s.handler?.type === 'ss' || s.handler?.type === 'ssu');
            if (firstSS) {
                document.getElementById('ssMethod').value = firstSS.handler.auth?.username || '';
                document.getElementById('ssPassword').value = firstSS.handler.auth?.password || '';
                if (nodes.length && nodes[0].listenPort) {
                    document.getElementById('startPort').value = nodes[0].listenPort;
                } else {
                    document.getElementById('startPort').value = parseInt(firstSS.addr?.split(':')[1]) || 46001;
                }
            }
            saveCache();
        } catch (e) { console.error('解析 YAML 失败', e); toast('服务器配置格式错误', 'error'); }
    }

    async function checkNode(index) {
        const node = nodes[index];
        const tcpCell = document.querySelector(`#tcp-${index}`);
        const udpCell = document.querySelector(`#udp-${index}`);
        if (!tcpCell || !udpCell) return;
        tcpCell.innerHTML = '⏳ 检测中...';
        udpCell.innerHTML = '⏳ 检测中...';

        try {
            const res = await authFetch('/check-node', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({addr: node.addr, username: node.username, password: node.password})
            });
            if (res) {
                const data = await res.json();
                let tcpHtml = data.tcp
                    ? `<span class="node-status" style="background:#dcfce7;color:#166534;">🟢 TCP</span><br><small>${data.tcp_latency}ms</small>`
                    : `<span class="node-status" style="background:#fee2e2;color:#991b1b;" title="${data.tcp_msg}">🔴 TCP</span><br><small>${data.tcp_latency}ms</small>`;
                let udpHtml = data.udp
                    ? `<span class="node-status" style="background:#dcfce7;color:#166534;">🟢 UDP</span><br><small>${data.udp_latency}ms</small>`
                    : `<span class="node-status" style="background:#fee2e2;color:#991b1b;" title="${data.udp_msg}">🔴 UDP</span><br><small>${data.udp_latency}ms</small>`;
                tcpCell.innerHTML = tcpHtml;
                udpCell.innerHTML = udpHtml;
            } else {
                tcpCell.innerHTML = '<span class="node-status" style="background:#fef3c7;color:#92400e;">❓</span>';
                udpCell.innerHTML = '<span class="node-status" style="background:#fef3c7;color:#92400e;">❓</span>';
            }
        } catch (e) {
            tcpCell.innerHTML = '<span class="node-status" style="background:#fef3c7;color:#92400e;">❓</span>';
            udpCell.innerHTML = '<span class="node-status" style="background:#fef3c7;color:#92400e;">❓</span>';
        }
    }

    async function checkAllNodes() {
        const btn = document.getElementById('checkAllBtn');
        btn.disabled = true;
        btn.textContent = '⏳ 检测中...';

        }

        btn.textContent = '🔍 检测所有节点';
        toast('所有节点检测完成');
    }

    function render() {
        const tbody = document.getElementById('tbody');
        if (!nodes.length) {
            tbody.innerHTML = '<tr><td colspan="10" style="text-align:center">暂无节点</td></tr>';
            document.getElementById('nodeCount').textContent = '0 个节点';
            return;
        }
        tbody.innerHTML = nodes.map((n,i) => `
            <tr>
                <td>${i+1}</td>
                <td><input class="listenPort" type="number" value="${n.listenPort || ''}" placeholder="自动" style="width:80px;"></td>
                <td><input class="addr" value="${escHtml(n.addr)}"></td>
                <td><input class="user" value="${escHtml(n.username)}"></td>
                <td><input class="pass" value="${escHtml(n.password)}"></td>
                <td><input class="remark" value="${escHtml(n.remark || '')}" placeholder="备注"></td>
                <td><input class="retry" type="number" min="0" max="10" value="${n.retry ?? 2}" style="width:60px;"></td>
                <td id="tcp-${i}" class="node-status-td">-</td>
                <td id="udp-${i}" class="node-status-td">-</td>
                <td>
                    <button class="check-single" data-idx="${i}" title="检测节点">🔍</button>
                    <button class="del" data-idx="${i}">🗑️</button>
                </td>
            </tr>`).join('');
        document.querySelectorAll('.listenPort').forEach((inp,i) => inp.onchange = () => { nodes[i].listenPort = parseInt(inp.value) || ''; saveCache(); });
        document.querySelectorAll('.addr').forEach((inp,i) => inp.onchange = () => { nodes[i].addr = inp.value; saveCache(); });
        document.querySelectorAll('.user').forEach((inp,i) => inp.onchange = () => { nodes[i].username = inp.value; saveCache(); });
        document.querySelectorAll('.pass').forEach((inp,i) => inp.onchange = () => { nodes[i].password = inp.value; saveCache(); });
        document.querySelectorAll('.remark').forEach((inp,i) => inp.onchange = () => { nodes[i].remark = inp.value; saveCache(); });
        document.querySelectorAll('.retry').forEach((inp,i) => inp.onchange = () => { nodes[i].retry = parseInt(inp.value) || 2; saveCache(); });
        document.querySelectorAll('.check-single').forEach(btn => {
            btn.onclick = () => checkNode(parseInt(btn.dataset.idx));
        });
        document.querySelectorAll('.del').forEach(btn => btn.onclick = () => {

    // 解析添加：自动分配监听端口
    function parseBatch() {
        const text = document.getElementById('batchInput').value;
        const lines = text.split(/\r?\n/);
        let added = 0;

        let nextPort = parseInt(document.getElementById('startPort').value) || 46001;
        nodes.forEach(n => {
            if (n.listenPort && !isNaN(n.listenPort) && n.listenPort >= nextPort) {
                nextPort = Math.max(nextPort, n.listenPort + 1);
            }
        });

        for (let line of lines) {
            line = line.trim();
            if (!line) continue;
            let raw = line.startsWith('socks5://') ? line.slice(9) : line;
            let parts = raw.split(':');
            if (parts.length >= 4) {
                let host = parts[0], port = parts[1], user = parts[2], pass = parts.slice(3).join(':');
                if (host && port && user && pass && !isNaN(parseInt(port))) {
                    let addr = `${host}:${port}`;
                    if (!nodes.some(n => n.addr === addr && n.username === user)) {
                        nodes.push({
                            addr,
                            username: user,
                            password: pass,
                            retry: 2,
                            remark: '',
                            listenPort: nextPort
                        });
                        nextPort++;
                        added++;
                    }
                }
            }
        }
        if (added > 0) {
            render();
            saveCache();
            toast(`成功添加 ${added} 个节点，已自动分配监听端口`);
        } else {
            toast('未添加新节点，检查格式', 'error');
        }
    }

    function extractToBatchInput() {
        const yamlText = document.getElementById('yamlOutput').value.trim();
        if (!yamlText) { toast('YAML 编辑器为空', 'error'); return; }
        try {
            const config = jsyaml.load(yamlText);
            const chains = config?.chains || [];
            const services = config?.services || [];
            const chainPortMap = {};
            services.forEach(s => {
                const portMatch = s.addr?.match(/:(\d+)$/);
                const port = portMatch ? parseInt(portMatch[1]) : null;
                const chain = s.handler?.chain;
                if (port && chain && !chainPortMap[chain]) {
                    chainPortMap[chain] = port;
                }
            });
            const newNodes = [];
            chains.forEach(chain => {
                const listenPort = chainPortMap[chain.name] || '';
                chain.hops?.forEach(hop => {
                    hop.nodes?.forEach(node => {
                        if (node.connector?.type === 'socks5' && node.addr) {
                            newNodes.push({
                                addr: node.addr,
                                username: node.connector.auth?.username || '',
                                password: node.connector.auth?.password || '',
                                retry: node.connector.retry || 2,
                                remark: node.connector?.metadata?.remark || '',
                                listenPort: listenPort
                            });
                        }
                    });
                });
            });
            if (newNodes.length === 0) { toast('YAML 中没有 socks5 节点', 'error'); return; }
            nodes = newNodes;
            render();
            saveCache();
            const lines = nodes.map(n => `${n.addr}:${n.username}:${n.password}`);
            document.getElementById('batchInput').value = lines.join('\n');
            toast(`已从 YAML 提取 ${nodes.length} 个节点`);
        } catch (e) { toast('YAML 格式错误', 'error'); }
    }

    // 生成 YAML，包含 DNS 代理配置
    function generateYaml() {
        const startPort = parseInt(document.getElementById('startPort').value);
        const method = document.getElementById('ssMethod').value.trim() || 'chacha20-ietf-poly1305';
        const pass = document.getElementById('ssPassword').value || '123456';
        const dnsUpstream = document.getElementById('dnsUpstream').value.trim() || '8.8.8.8:53';
        if (!nodes.length) { toast('请先添加节点','error'); return; }

        let services = [], chains = [];
        nodes.forEach((n,i) => {
            const port = (n.listenPort && !isNaN(n.listenPort)) ? parseInt(n.listenPort) : (startPort + i);
            // TCP 服务，handler 加入 dns
            services.push({
                name: `ss-tcp-${port}`,
                addr: `:${port}`,
                handler: {
                    type: "ss",
                    auth: { username: method, password: pass },
                    chain: `chain-${port}`,
                    dns: [
                        { addr: dnsUpstream, chain: `chain-${port}` }
                    ]
                },
                listener: { type: "tcp" }
            });
            // UDP 服务，handler 加入 dns
            services.push({
                name: `ss-udp-${port}`,
                addr: `:${port}`,
                handler: {
                    type: "ssu",
                    auth: { username: method, password: pass },
                    chain: `chain-${port}`,
                    dns: [
                        { addr: dnsUpstream, chain: `chain-${port}` }
                    ]
                },
                listener: { type: "udp" }
            });
            const metadata = { udp: true, udpBufferSize: 8192, notls: true };
            if (n.remark) metadata.remark = n.remark;
            chains.push({
                name: `chain-${port}`,
                hops: [{
                    name: "hop-0",
                    nodes: [{
                        name: "node-0",
                        addr: n.addr,
                        connector: {
                            type: "socks5",
                            auth: { username: n.username, password: n.password },
                            metadata: metadata,
                            mux: { type: "socks5", idleTimeout: "60s" },
                            dialer: { type: "tcp", timeout: "15s" },
                            retry: n.retry ?? 2
                        }
                    }]
                }]
            });
        });
        currentYaml = jsyaml.dump({ services, chains }, { indent: 2, lineWidth: -1 });
        document.getElementById('yamlOutput').value = currentYaml;
    }
