function setup(cv){const dpr=devicePixelRatio||1;const r=cv.getBoundingClientRect();cv.width=r.width*dpr;cv.height=(cv.height||220)*dpr;const x=cv.getContext('2d');x.scale(dpr,dpr);return [x,r.width,cv.height/dpr];}
function areaChart(cv,data,col){
  const [x,w,h]=setup(cv);x.clearRect(0,0,w,h);
  const max=Math.max(...data)*1.15,pad=6;const sx=w/(data.length-1);
  const y=v=>h-pad-(v/max)*(h-pad*2);
  const g=x.createLinearGradient(0,0,0,h);g.addColorStop(0,col+'55');g.addColorStop(1,col+'00');
  x.beginPath();x.moveTo(0,h);data.forEach((v,i)=>x.lineTo(i*sx,y(v)));x.lineTo(w,h);x.closePath();x.fillStyle=g;x.fill();
  x.beginPath();data.forEach((v,i)=>i?x.lineTo(i*sx,y(v)):x.moveTo(0,y(v)));x.strokeStyle=col;x.lineWidth=2;x.stroke();
  const lv=y(data[data.length-1]);x.beginPath();x.arc(w-1,lv,3,0,7);x.fillStyle=col;x.fill();
}

  const max=Math.max(...data),min=Math.min(...data),sx=w/(data.length-1);
  const y=v=>h-2-((v-min)/(max-min||1))*(h-4);
  x.beginPath();data.forEach((v,i)=>i?x.lineTo(i*sx,y(v)):x.moveTo(0,y(v)));x.strokeStyle=col;x.lineWidth=1.6;x.stroke();
}
