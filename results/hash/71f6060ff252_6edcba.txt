        document.addEventListener("DOMContentLoaded",()=>{

            enviar_pregunta.addEventListener("click", async () => {
                const pregunta = inputPregunta.value;
                const btnChat  = document.querySelector(".widget-btn");
                btnChat.click();

                const waitForElement = (selector) => {
                    return new Promise((resolve) => {
                        const interval = setInterval(() => {
                            const element = document.querySelector(selector);
                            if (element) {
                                clearInterval(interval);
                                resolve(element);
                            }
                        }, 100); 
                    });
                };

                const inputChat = await waitForElement("#editable");
                const sendButton = await waitForElement(".kit-widget-chat-input__send-btn");

                inputChat.textContent = pregunta;
                const event = new Event('input', { bubbles: true });
                inputChat.dispatchEvent(event);

                setTimeout(() => {
                    sendButton.click();
                }, 500);
            });
            
        })
