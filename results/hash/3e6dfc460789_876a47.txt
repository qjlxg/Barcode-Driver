// Build options
Object.entries(DATA).forEach(([key, q]) => {
  const el = document.getElementById(q.container);
  q.items.forEach((item, i) => {
    const div = document.createElement('div');
    div.className = 'opt';
    div.innerHTML = `<div class="opt-radio"></div><div class="opt-body"><div class="opt-level">${item.l}</div><div class="opt-desc">${item.d}</div></div>`;
    div.onclick = () => {
      el.querySelectorAll('.opt').forEach(o => o.classList.remove('selected'));
      div.classList.add('selected');
      scores[key] = {
       level: i,
       text: item.d
      };
      document.getElementById(q.btn).disabled = false;
    };
    el.appendChild(div);
  });
});

function setStep(n) {
  document.querySelectorAll('.step').forEach(s => s.classList.remove('active'));
  const el = document.getElementById(STEPS[n]);
  el.classList.add('active');
  cur = n;
  const total = STEPS.length - 1;
  const pct = Math.round((n / total) * 100);
  document.getElementById('pbar').style.width = pct + '%';
  const label = n === 0 ? '' : n === total ? 'Результат' : `Шаг ${n} из ${total - 1}`;
  document.getElementById('step-counter').textContent = label;
  document.getElementById('main-card').style.animation = 'none';
  requestAnimationFrame(() => { document.getElementById('main-card').style.animation = ''; });
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function goNext() { setStep(cur + 1); }
function goPrev() { setStep(cur - 1); }

function pickChip(el, group) {
  document.querySelectorAll(`#chips-${group} .chip`).forEach(c => c.classList.remove('selected'));
  el.classList.add('selected');
  chipSelections[group] = el.textContent;
}

function pickMultiChip(el, group, max) {
  const isSelected = el.classList.contains('selected');
  const current = document.querySelectorAll(`#chips-${group} .chip.selected`);
  if (!isSelected && current.length >= max) return;
  el.classList.toggle('selected');
  chipSelections[group] = Array.from(document.querySelectorAll(`#chips-${group} .chip.selected`)).map(c => c.textContent);
}

function getTrack() {
  const ai = scores.ai || 0, vc = scores.vc || 0, n8n = scores.n8n || 0;
  const mkt = scores.mkt || 0, sales = scores.sales || 0, asset = scores.asset || 0;
  const tech = (ai + vc + n8n) / 3;
  const comm = (mkt + sales) / 2;
  if (asset >= 3) return 5;
  if (tech >= 4 && comm >= 3) return 6;
  if (tech >= 3 && comm <= 2) return 2;
  if (tech <= 2 && comm >= 3) return 3;
  if (tech >= 2 && comm >= 2) return 4;
  return 1;
}

function makeDots(val, max) {
  let h = '<div class="dots-row">';
  for (let i = 0; i < max; i++) h += `<div class="dot${i < val ? ' on' : ''}"></div>`;
  return h + '</div>';
}

}

function sendToSheets(d) {
  if (!SHEETS_URL) return;
  const status = document.getElementById('sheets-status');
  const row = {
    date: d.date,
    name: d.name,
    contact: d.contact,
    track: `${d.trackN} — ${d.t.name}`,
    ai: scores.ai?.text || '', 
    vc: scores.vc?.text || '',
    n8n: scores.n8n?.text || '',
    mkt: scores.mkt?.text || '',
    sales: scores.sales?.text || '',
    mind: d.mind,
    asset: d.asset,
    goalFin: d.goalFin,
    goalProd: d.goalProd,
    role: d.role,
    blockers: d.blockers,
    time: d.time,
  };
  fetch(SHEETS_URL, {
    method: 'POST',
    mode: 'no-cors',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(row),
  })
  .then(() => {
    if (status) status.innerHTML = '<div class="sheets-notice success">✓ Карточка отправлена куратору</div>';
  })
  .catch(() => {
    if (status) status.innerHTML = '<div class="sheets-notice error">Не удалось отправить карточку — сохрани скриншот или скачай файл.</div>';
  });
}
