async function startRecording() {
  if (mediaRec) return;
  if (!navigator.mediaDevices?.getUserMedia) {
    alert('Голосовые сообщения работают только через https://95.85.235.78');
    return;
  }
  let stream;
  try { stream = await navigator.mediaDevices.getUserMedia({ audio: true, video: false }); }
  catch(e) {
    alert(e.name === 'NotAllowedError'
      ? 'Разреши доступ к микрофону в настройках браузера'
      : 'Ошибка микрофона: ' + e.message);
    return;
  }

  // Web Audio для визуализации
  audioCtx  = new (window.AudioContext || window.webkitAudioContext)();
  analyser  = audioCtx.createAnalyser();
  analyser.fftSize = 128;
  audioCtx.createMediaStreamSource(stream).connect(analyser);

  recCancelled = false;
  audioChunks  = [];
  waveHistory  = [];
  recSecs      = 0;

  const mimeType = ['audio/webm;codecs=opus','audio/webm','audio/ogg;codecs=opus','audio/ogg']
    .find(t => MediaRecorder.isTypeSupported(t)) || '';
  mediaRec = new MediaRecorder(stream, mimeType ? { mimeType } : {});
  mediaRec.ondataavailable = e => { if (e.data.size > 0) audioChunks.push(e.data); };
  mediaRec.onstop = () => {
    stream.getTracks().forEach(t => t.stop());
    audioCtx?.close(); audioCtx = null; analyser = null;
    cancelAnimationFrame(animFrame);
    if (!recCancelled && audioChunks.length) uploadVoice();
  };
  mediaRec.start(100);

  // UI
  document.getElementById('rec-ui').classList.add('show');
  document.getElementById('input-row').style.display = 'none';
  document.getElementById('mic-btn').classList.add('rec');

  // Таймер
  recTimer = setInterval(() => {
    recSecs++;
    document.getElementById('rec-time').textContent = fmtTime(recSecs);
    if (recSecs >= 120) stopRecording(true);
  }, 1000);

  // Запускаем визуализацию
  initCanvas();
  drawLiveWave();
}

function stopRecording(send) {
  if (!mediaRec) return;
  clearInterval(recTimer);
  cancelAnimationFrame(animFrame);
  recCancelled = !send;
  if (mediaRec.state !== 'inactive') mediaRec.stop();
  mediaRec = null;
  document.getElementById('rec-ui').classList.remove('show');
  document.getElementById('input-row').style.display = 'flex';
  document.getElementById('mic-btn').classList.remove('rec');
  document.getElementById('rec-time').textContent = '0:00';
  // Очищаем canvas
  const cv = document.getElementById('rec-canvas');
  cv.getContext('2d').clearRect(0, 0, cv.width, cv.height);
}

function cancelRecording() { stopRecording(false); }

async function uploadVoice() {
  const mimeType = mediaRec?._mimeType || audioChunks[0]?.type || 'audio/webm';
  const ext = mimeType.includes('ogg') ? 'ogg' : 'webm';
  const blob = new Blob(audioChunks, { type: mimeType });
  if (blob.size < 300) return;
  const fd = new FormData();
  fd.append('file', blob, `voice-${Date.now()}.${ext}`);
  try {
    const r = await fetch('/api/upload', { method:'POST', headers:{ Authorization:'Bearer '+token }, body:fd });
    const d = await r.json();
    if (!r.ok) throw new Error(d.error);
    socket.emit('send_message', { roomId, text: null, mediaUrl: d.url, mediaType: 'audio' });
  } catch(e) { alert('Ошибка отправки: ' + e.message); }
}

function drawLiveWave() {
  if (!analyser) return;
  const cv  = document.getElementById('rec-canvas');
  const ctx = cv.getContext('2d');
  const W = cv.width, H = cv.height;
  const buf = new Uint8Array(analyser.frequencyBinCount);
  analyser.getByteFrequencyData(buf);

  // Среднее значение громкости
  const avg = buf.reduce((a,b) => a+b, 0) / buf.length;
  waveHistory.push(avg);
  if (waveHistory.length > W) waveHistory.shift();

  ctx.clearRect(0, 0, W, H);

  // Рисуем историю как бегущую волну
  const barW   = 3;
  const gap    = 2;
  const total  = barW + gap;
  const count  = Math.floor(W / total);
  const start  = Math.max(0, waveHistory.length - count);

  for (let i = 0; i < count; i++) {
    const val = waveHistory[start + i] ?? 0;
    const barH = Math.max(3, (val / 255) * (H - 4));
    const x    = i * total;
    const y    = (H - barH) / 2;

    // Градиент: тихо — приглушённый жёлтый, громко — яркий
    const alpha = 0.3 + (val / 255) * 0.7;
    ctx.fillStyle = `rgba(250,182,34,${alpha})`;

    // Скруглённые прямоугольники
    ctx.beginPath();
    ctx.roundRect(x, y, barW, barH, 2);
    ctx.fill();
  }

  animFrame = requestAnimationFrame(drawLiveWave);
}

function buildWave(n, seed) {
  let h = '', s = seed || 42;
  for (let i = 0; i < n; i++) {
    s = (s * 1664525 + 1013904223) & 0xffffffff;
    const height = 4 + Math.abs(Math.sin(i * 0.7 + s * 0.00001) * 12) + (s >>> 28);
    h += `<div class="audio-wave-bar" style="height:${Math.round(height)}px"></div>`;
  }
  return h;
}

function renderAudioPlayer(url, id) {
  const seed = id.split('').reduce((a,c)=>a+c.charCodeAt(0),0);
  return `
    <div class="msg-audio-wrap" id="ap-${id}">
      <audio id="aud-${id}" src="${url}" preload="metadata"
        style="display:none"
        ontimeupdate="_auUpdate('${id}')"
        onloadedmetadata="_auMeta('${id}')"
        onended="_auEnd('${id}')"
        onerror="_auErr('${id}','${url}')">
      </audio>
      <button class="audio-play-btn" id="apb-${id}" onclick="_auToggle('${id}')">▶</button>
      <div class="audio-right">
        <div class="audio-wave" id="aw-${id}" onclick="_auSeek(event,'${id}')">${buildWave(32, seed)}</div>
        <div class="audio-meta">
          <span class="audio-dur" id="ad-${id}">0:00</span>
        </div>
      </div>
    </div>`;
}

function _auEl(id) { return document.getElementById('aud-'+id); }

function _auToggle(id) {
  const a = _auEl(id); if (!a) return;
  if (a.paused) {
    // Пауза всех остальных
    document.querySelectorAll('.msg-audio-wrap audio').forEach(el => {
      if (el !== a && !el.paused) { el.pause(); document.getElementById('apb-'+el.id.slice(4)).textContent = '▶'; }
    });
    a.play().catch(e => {
      console.error('play failed:', e);
      _auErr(id, a.src);
    });
    document.getElementById('apb-'+id).textContent = '⏸';
  } else {
    a.pause();
    document.getElementById('apb-'+id).textContent = '▶';
  }
}

function _auUpdate(id) {
  const a = _auEl(id); if (!a || !a.duration) return;
  const bars = document.querySelectorAll('#aw-'+id+' .audio-wave-bar');
  const played = Math.floor((a.currentTime / a.duration) * bars.length);
  bars.forEach((b,i) => b.classList.toggle('played', i < played));
  document.getElementById('ad-'+id).textContent = fmtTime(Math.floor(a.currentTime));
}

function _auMeta(id) {
  const a = _auEl(id); if (!a) return;
  const dur = isFinite(a.duration) ? Math.round(a.duration) : 0;
  document.getElementById('ad-'+id).textContent = fmtTime(dur);
}

function _auEnd(id) {
  document.getElementById('apb-'+id).textContent = '▶';
  document.querySelectorAll('#aw-'+id+' .audio-wave-bar').forEach(b => b.classList.remove('played'));
  const a = _auEl(id); if (a) { a.currentTime = 0; }
}

function _auErr(id, url) {
  // Fallback — показываем нативный <audio controls>
  const wrap = document.getElementById('ap-'+id);
  if (!wrap) return;
  wrap.innerHTML = `<audio controls src="${url}" style="max-width:280px;border-radius:8px;"></audio>`;
}

function _auSeek(e, id) {
  const a = _auEl(id); if (!a || !a.duration) return;
  const wave = document.getElementById('aw-'+id);
  const rect = wave.getBoundingClientRect();
  a.currentTime = Math.max(0, Math.min(1, (e.clientX - rect.left) / rect.width)) * a.duration;
}

async function onFilePick(e) {
  const files = [...e.target.files];
  if (!files.length) return;
  e.target.value = '';
  const preview = document.getElementById('up-preview');
  preview.style.display = 'flex';

  for (const file of files) {
    const thumb = document.createElement('div');
    thumb.className = 'up-thumb';
    thumb.innerHTML = `<div class="upload-spinner"></div>`;
    preview.appendChild(thumb);

    try {
      const fd = new FormData();
      fd.append('file', file);
      const r = await fetch('/api/upload', { method:'POST', headers:{ Authorization:'Bearer '+token }, body:fd });
      const data = await r.json();
      if (!r.ok) throw new Error(data.error);
      pendingMedia.push({ url: data.url, type: data.type });
      toggleSendMic();

      const rm = document.createElement('button');
      rm.className = 'up-rm';
      rm.textContent = '×';
      const idx = pendingMedia.length - 1;
      rm.onclick = () => { pendingMedia.splice(idx, 1); thumb.remove(); if (!pendingMedia.length) { preview.style.display = 'none'; } toggleSendMic(); };

      if (data.type === 'image') {
        thumb.innerHTML = `<img src="${data.url}" alt="">`;
      } else {
        thumb.innerHTML = `<video src="${data.url}"></video>`;
      }
      thumb.appendChild(rm);
    } catch(err) {
      thumb.innerHTML = `<span style="font-size:11px;color:var(--danger)">Ошибка</span>`;
      setTimeout(() => thumb.remove(), 2000);
    }
  }
}

    );
    return;
  }
  try {
    localStream = await navigator.mediaDevices.getUserMedia({ audio: true, video: false });
  } catch(e) {
    const msg = e.name === 'NotAllowedError'
      ? 'Доступ к микрофону запрещён. Разреши его в настройках браузера.'
      : 'Нет доступа к микрофону: ' + e.message;
    alert(msg);
    return;
  }
  inVoice = true;
  document.getElementById('voice-btn').classList.add('on');
  document.getElementById('vbtn-txt').textContent = 'Выйти из голоса';
  socket.emit('voice_join', roomId);
}

function leaveVoice() {
  if (!inVoice) return;
  inVoice = false;
  if (localStream) { localStream.getTracks().forEach(t => t.stop()); localStream = null; }
  Object.keys(peers).forEach(id => { peers[id].close(); delete peers[id]; });
  document.querySelectorAll('.voice-audio').forEach(a => a.remove());
  document.getElementById('voice-btn').classList.remove('on');
  document.getElementById('vbtn-txt').textContent = 'Войти в голос';
  document.getElementById('voice-bar').classList.remove('show');
  if (roomId) socket.emit('voice_leave', roomId);
}

function makePeer(remoteId) {
  if (peers[remoteId]) peers[remoteId].close();
  const pc = new RTCPeerConnection(ICE);
  peers[remoteId] = pc;
  if (localStream) localStream.getTracks().forEach(t => pc.addTrack(t, localStream));
  pc.ontrack = e => {
    let a = document.getElementById('va-' + remoteId);
    if (!a) { a = document.createElement('audio'); a.id = 'va-' + remoteId; a.className = 'voice-audio'; a.autoplay = true; document.body.appendChild(a); }
    a.srcObject = e.streams[0];
  };
  pc.onicecandidate = e => {
    if (e.candidate) socket.emit('voice_signal', { to: remoteId, signal: { type: 'candidate', candidate: e.candidate } });
  };
  pc.oniceconnectionstatechange = () => {
    if (['failed','disconnected','closed'].includes(pc.iceConnectionState)) {
      pc.close(); delete peers[remoteId];
      const a = document.getElementById('va-' + remoteId); if (a) a.remove();
    }
  };
  return pc;
}

// New user joins — they create offers to all existing peers
socket.on('voice_existing_peers', async existingPeers => {
  for (const peer of existingPeers) {
    const pc = makePeer(peer.socketId);
    const offer = await pc.createOffer();
    await pc.setLocalDescription(offer);
    socket.emit('voice_signal', { to: peer.socketId, signal: { type: 'offer', sdp: pc.localDescription } });
  }
});

// Existing user notified about new joiner — wait for their offer
socket.on('voice_peer_joined', ({ socketId }) => {
  makePeer(socketId); // prepare, offer will come from them
});

// Unified signal handler
socket.on('voice_signal', async ({ from, signal }) => {
  if (!inVoice) return;
  let pc = peers[from];

  if (signal.type === 'offer') {
    if (!pc) pc = makePeer(from);
    await pc.setRemoteDescription(new RTCSessionDescription(signal.sdp));
    const answer = await pc.createAnswer();
    await pc.setLocalDescription(answer);
    socket.emit('voice_signal', { to: from, signal: { type: 'answer', sdp: pc.localDescription } });
  } else if (signal.type === 'answer') {
    if (pc) await pc.setRemoteDescription(new RTCSessionDescription(signal.sdp));
  } else if (signal.type === 'candidate') {
    if (pc) { try { await pc.addIceCandidate(new RTCIceCandidate(signal.candidate)); } catch {} }
  }
});

socket.on('voice_peer_left', ({ socketId }) => {
  if (peers[socketId]) { peers[socketId].close(); delete peers[socketId]; }
  const a = document.getElementById('va-' + socketId); if (a) a.remove();
});

async function api(url, body, tok) {
  const opts = { headers: { 'Content-Type':'application/json' } };
  if (tok) opts.headers['Authorization'] = 'Bearer ' + tok;
  if (body) { opts.method = 'POST'; opts.body = JSON.stringify(body); }
  const r = await fetch(url, opts);
  const d = await r.json();
  if (!r.ok) throw new Error(d.error || 'Ошибка');
  return d;
}

}
