const addCommas = n => n.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',');

class LoycoSlider {
    constructor(config) {
        this.isDragging = false;
        this.percentage = 0;
        this.step = 0;
        this.step_value = 0;

        this.config = {
            id: Array.from({ length: 12 }, () => String.fromCharCode(65 + Math.random() * 26)).join(''),
            currency: 'NOK',
            language:'EN',
            budgetSegment: null,
            programId: null,
            bar_background_color:'#424242',
            bar_color: 'linear-gradient(90deg, rgba(56,198,189,1) 0%, rgba(37,167,179,1) 50%, rgba(14,127,166,1) 100%)',
            grab_color: 'linear-gradient(65deg, rgba(56,198,189,1) 0%, rgba(37,167,179,1) 50%, rgba(14,127,166,1) 100%)',
            text_color: 'linear-gradient(65deg, rgb(32, 153, 145) 0%, rgba(37,167,179,1) 50%, rgba(14,127,166,1) 100%)',
            size: '1',
            API_URL: 'https://ias.loyco.io/pricesteps/',
            ...config,
        };

        this.lang = {
            EN:{
                members:'contacts',
                month:'month',
                pr_mbr:'per contact'
            },
            NO:{
                members: 'kontakter',
                month: 'måned',
                pr_mbr: 'per kontakt'
            }
        }

        this.id = this.config.id;

        if(this.config.element == null) return;

        const onFetch = (data) => {
            this.config.steps = data.map((x) => x.members);
            this.config.price = data.map((x) => x.price);
            this.step_value = this.config.steps[0]
            this.create();
        };

        this.config.API_URL += `?currency=${this.config.currency}&language=${this.config.language}`;
        if (this.config.budgetSegment !== null) this.config.API_URL += `&budgetsegment=${this.config.budgetSegment}`;
        if (this.config.programId !== null) this.config.API_URL += `&programid=${this.config.programId}`;

        fetch(this.config.API_URL, { headers: { "Content-Type": "application/json" } })
            .then((x) => x.json())
            .then(onFetch);
    }

    create() {
        if (!document.querySelector('#loyco_slider_css')) {
            const css = document.createElement('style');
            css.id = 'loyco_slider_css';
            css.innerText = `
                .loyco_slider {
					opacity: 0;
                    animation: fadeIn 0.7s forwards;
                    height: 30px;
                    border-radius: 3px;
                    width: 100%;
                    outline: 2px solid rgba(126, 126, 126, 0.664);
                    position: relative;
                }
                .loyco_slider > div:nth-child(1) {
                    border-radius: 3px;
                    height: 30px;
                    position: absolute;
                }
                .loyco_slider > div:nth-child(2) {
                    width: 50px;
                    pointer-events: auto;
                    height: 50px;
                    outline: 2px solid rgba(126, 126, 126, 0.664);
                    position: absolute;
                    border-radius: 100%;
                    margin-top: -10px;
                    margin-left: -10px;
                    overflow: hidden;
                    cursor: grab;
                    z-index: 2;
                }
                .loyco_slider_step_line {
                    width: 4px;
                    height: 30px;
                    background-color: rgba(126, 126, 126, 1);
                    position: absolute;
                    top: 0;
                    transform: translateX(-50%);
                    z-index: 1;
                }
                @keyframes fadeIn {
                    from { opacity: 0; }
                    to { opacity: 1; }
                }
                .floating-text {
                    position: absolute;
                    top: 60px;
                    text-rendering: optimizeLegibility;
                    -webkit-font-smoothing: antialiased; /* For WebKit browsers */
                    -moz-osx-font-smoothing: grayscale;  /* For macOS Firefox */
                    transform: translateX(-50%);
                    width: 230px;
                    background: rgb(254,254,254); /* Box background */
                    color: #363738;
                    padding: 8px;
                    border-radius: 6px;
                    text-align: center;
                    pointer-events: none;
                    white-space: nowrap;
                    font-size:19px;
                    font-family: Arial;
                    box-shadow: 0 0 0 1.5px #363738; /* White outline around the box */
                }

                .floating-text > div > a {
                    -webkit-background-clip: text;
                    -webkit-text-fill-color: transparent;
                    font-weight:bold;
                }

                .floating-text > div:nth-child(1){
                    font-size:25px;
                }

                .floating-text::before {
                    content: '';
                    position: absolute;
                    top: -12px; /* Slightly above the arrow */
                    left: var(--arrow-offset, 50%);
                    transform: translateX(-50%);
                    width: 0;
                    height: 0;

                    /* White outline for the arrow */
                    border-left: 12px solid transparent;
                    border-right: 12px solid transparent;
                    border-bottom: 12px solid #363738;
                    z-index: -1; /* Ensure it's behind the main arrow */
                }

                .floating-text::after {
                    content: '';
                    position: absolute;
                    top: -10px;
                    left: var(--arrow-offset, 50%);
                    transform: translateX(-50%);
                    width: 0;
                    height: 0;

                    /* Main white arrow */
                    border-left: 10px solid transparent;
                    border-right: 10px solid transparent;
                    border-bottom: 10px solid rgb(254,254,254); /* Main arrow color */
                }

            `;
            document.head.appendChild(css);
        }

        this.container = document.createElement('div');
        this.container.id = this.id + '_loyco_slider';
        this.container.classList.add('loyco_slider');
        this.container.style.background = this.config.bar_background_color

        this.container.innerHTML = `
            <div id="${this.id}_loyco_slider_bg" style="background: ${this.config.bar_color}"></div>
            <div id="${this.id}_loyco_slider_grab" style="background: ${this.config.grab_color}"></div>
            <div id="${this.id}_floating_text" class="floating-text">0</div>
        `;

        this.config.element.appendChild(this.container);

        this.grab = document.querySelector(`#${this.id}_loyco_slider_grab`);
        this.bg = document.querySelector(`#${this.id}_loyco_slider_bg`);
        this.floatingText = document.querySelector(`#${this.id}_floating_text`);
        this.max = this.container.clientWidth;
        this.bg.style.backgroundSize = `${this.max}px 30px`;

        // Add lines for each step
        if (this.config.steps && this.config.steps.length > 2) {
            const totalSteps = this.config.steps.length - 1;
            this.config.steps.forEach((_, index) => {
                if (index === 0 || index === totalSteps) return; // Skip first and last steps

                const stepPosition = (index / totalSteps) * 100;
                const stepLine = document.createElement('div');
                stepLine.classList.add('loyco_slider_step_line');
                stepLine.style.left = `${stepPosition}%`;
                this.container.appendChild(stepLine);
            });
        }

        const updateFloatingTextPosition = () => {
            const grabRect = this.grab.getBoundingClientRect();
            const containerRect = this.container.getBoundingClientRect();
        
            const grabCenter = grabRect.left - containerRect.left + grabRect.width / 2; // Exact center of the grab ball
            const floatingWidth = 230; // Width of the floating text
            const margin = 20; // Additional margin for offset beyond the bar width
        
            // Adjust floating text position to stay within slider boundaries
            let leftPosition = grabCenter;
            if (leftPosition - floatingWidth / 2 < -margin) {
                leftPosition = floatingWidth / 2 - margin;
            } else if (leftPosition + floatingWidth / 2 > this.max + margin) {
                leftPosition = this.max - floatingWidth / 2 + margin;
            }
        
            // Set floating text position
            this.floatingText.style.left = `${leftPosition}px`;
        
            // Calculate and set the arrow's offset to point at the grab ball's center
            const arrowOffset = grabCenter - (leftPosition - floatingWidth / 2); // Relative to floating text
            this.floatingText.style.setProperty('--arrow-offset', `${arrowOffset}px`);
        
            // Update the displayed text
            this.floatingText.innerHTML = `
            <div><a style="background:${this.config.text_color};-webkit-background-clip: text;">${addCommas(this.step_value)}</a> ${this.lang[this.config.language].members}</div>
            <div style="margin-top:4px;"><a style="background:${this.config.text_color};-webkit-background-clip: text;">${this.config.currency} ${addCommas(this.config.price[this.step])}</a> /${this.lang[this.config.language].month}</div>
            <div><a style="background:${this.config.text_color};-webkit-background-clip: text;">${this.config.currency} ${(this.config.price[this.step]/this.step_value).toFixed(3)}</a> ${this.lang[this.config.language].pr_mbr}</div>`;
        };
        
        const adjustGrabPositionOnResize = () => {
            this.max = this.container.clientWidth;
            this.bg.style.backgroundSize = `${this.max}px 30px`;
            const mouseX = (this.percentage / 100) * this.max;
            this.grab.style.left = mouseX - 20 + "px";
            this.bg.style.width = this.percentage + "%";
            updateFloatingTextPosition();
        };

        const handleMouseDown = (event) => {
            event.preventDefault();
            this.isDragging = true;
            document.body.style.cursor = "grabbing";
        };

        const handleMouseUp = () => {
            this.isDragging = false;
            document.body.style.cursor = "default";
            this.snapToStep();
            updateFloatingTextPosition();
        };

        const handleMouseMove = (event) => {
            if (!this.isDragging) return;
        
            const containerRect = this.bg.parentElement.getBoundingClientRect();
            let mouseX = event.clientX - containerRect.left;
        
            mouseX = Math.max(0, Math.min(this.max, mouseX));
        
            this.percentage = (mouseX / this.max) * 100;
            this.updateStepOnMove();
            this.grab.style.left = mouseX - 20 + "px";
            this.bg.style.width = this.percentage + "%";
        
            updateFloatingTextPosition();
        };
        
        this.updateStepOnMove = () => {
            const totalSteps = this.config.steps.length - 1;
            const newStep = Math.round((this.percentage / 100) * totalSteps);

            if (newStep !== this.step) {
                this.step = newStep;
                this.step_value = this.config.steps[this.step];
                this.onUpdate(this.step, this.step_value);
            }
        };

        this.snapToStep = () => {
            if (!this.config.steps || this.config.steps.length === 0) return;

            const totalSteps = this.config.steps.length - 1;
            this.step = Math.round((this.percentage / 100) * totalSteps);
            this.step_value = this.config.steps[this.step];

            this.percentage = (this.step / totalSteps) * 100;
            const mouseX = (this.percentage / 100) * this.max;
            this.grab.style.left = mouseX - 20 + "px";
            this.bg.style.width = this.percentage + "%";

            this.onUpdate(this.step, this.step_value);
            updateFloatingTextPosition();
        };

        this.grab.addEventListener("mousedown", handleMouseDown);
        document.addEventListener("mouseup", handleMouseUp);
        document.addEventListener("mousemove", handleMouseMove);
        window.addEventListener("resize", adjustGrabPositionOnResize);

        this.container.addEventListener("click", (event) => {
            const containerRect = this.container.getBoundingClientRect();
            let mouseX = event.clientX - containerRect.left;

            mouseX = Math.max(0, Math.min(this.max, mouseX));
            this.percentage = (mouseX / this.max) * 100;
            this.snapToStep();
        });

        adjustGrabPositionOnResize();
        this.config.onLoad && this.config.onLoad(this)
    }

    onUpdate() {
        this.config.onUpdate && this.config.onUpdate(this.step, this.step_value, this.config?.price?.[this.step], this.percentage, this);
    }
}

document.addEventListener("DOMContentLoaded", function () {

            }
        }

    });
