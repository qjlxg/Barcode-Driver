function money(usd) {
    if (usd == null) return '—';
    if (CUR.rub) return Math.round(usd * CUR.rate).toLocaleString('ru-RU') + ' ₽';
    return '$ ' + Number(usd).toFixed(Number(usd) < 1 ? 3 : 2);
}

function codeToFlag(code) { if (!code || code.length !== 2) return ''; const b = 0x1F1E6 - 65; return String.fromCodePoint(code.toUpperCase().charCodeAt(0)+b, code.toUpperCase().charCodeAt(1)+b); }

// ── Price block ──
function priceBlock(p) {
    if (p.price_orig_usd && p.price_orig_usd > p.price_usd) {
        return `<div class="text-right"><div class="font-black" style="color:var(--accent)">${money(p.price_usd)}</div><div class="line-through text-[10px]" style="color:var(--muted)">${money(p.price_orig_usd)}</div></div>`;
    }
    return `<div class="font-black" style="color:var(--accent)">${money(p.price_usd)}</div>`;
}

// ── Accounts ──
function renderAccounts() {
    const box = document.getElementById('shop-list');
    if (!CATALOG) return;
    const q = (document.getElementById('acc-search').value || '').toLowerCase().trim();
    let list = CATALOG.accounts.slice();
    if (q) list = list.filter(p => (p.name||'').toLowerCase().includes(q) || (p.phone_prefix||'').toLowerCase().includes(q));
    if (accFavOnly) list = list.filter(p => isFav(p.id));
    if (accSort === 'asc')  list.sort((a,b) => (a.price_usd||0)-(b.price_usd||0));
    if (accSort === 'desc') list.sort((a,b) => (b.price_usd||0)-(a.price_usd||0));
    if (!list.length) {
        box.innerHTML = accFavOnly ? emptyHtml('⭐', t('fav_empty'), t('fav_hint'))
            : (q ? emptyHtml('🔎', t('not_found'), '') : emptyHtml('📭', t('no_accounts'), t('check_later')));
        return;
    }
    box.innerHTML = list.map(p => `
        <div onclick='openProduct(${JSON.stringify({kind:"account",id:p.id,name:p.name,desc:p.description,price:p.price_usd,orig:p.price_orig_usd,available:p.available})})'
             class="card p-4 flex justify-between items-center gap-3 cursor-pointer active:scale-[.98] transition-transform">
            <div class="min-w-0">
                <div class="font-bold truncate">${escapeHtml(p.name)}</div>
                <div class="text-[10px] uppercase font-bold tracking-wider mt-0.5" style="color:var(--muted)">${p.phone_prefix ? escapeHtml(p.phone_prefix)+' · ' : ''}${p.available} ${t('pieces')}</div>
            </div>
            <div class="shrink-0">${priceBlock(p)}</div>
        </div>`).join('');
}

async function refreshPremiumPrices() {
    const recipient = recipientVal();
    for (const chip of document.querySelectorAll('#options-sub .dur')) {
        const a = chip.dataset.amount;
        api(`/api/price?type=PREMIUM&amount=${a}` + (recipient ? `&recipient=${encodeURIComponent(recipient)}` : '')).then(r => {
            const el = chip.querySelector('.price');
            premPrices[a] = (r && r.ok) ? r.price_usd : null;
            if (el) el.innerText = money(premPrices[a]);
            updatePremTotal();
        }).catch(()=>{});
    }
}
function premCount() {
    if (premBulk) return splitList(document.getElementById('prem-usernames').value).length || 1;
    return 1;
}
function updatePremTotal() {
    if (premType !== 'sub') return;
    const chip = document.querySelector('#options-sub .dur.active');
    const ctaSum = document.getElementById('prem-cta-sum');
    let total = null;
    if (chip) { const unit = premPrices[chip.dataset.amount]; if (unit != null) total = money(unit * premCount()); }
    document.getElementById('prem-total').innerText = total || '—';
    ctaSum.innerText = total || '';
}

let starsTimer = null;
function onStarsQty() {
    const qty = parseInt(document.getElementById('stars-qty').value || '0', 10);
    let recipient = recipientVal();
    if (premBulk) { const list = splitList(document.getElementById('prem-usernames').value); recipient = list[0] || ''; }
    if (!qty || qty < 1) {
        document.getElementById('star-unit').innerText = '—';
        if (premType === 'stars') { document.getElementById('prem-total').innerText = '—'; document.getElementById('prem-cta-sum').innerText = ''; }
        return;
    }
    clearTimeout(starsTimer);
    starsTimer = setTimeout(() => {
        api(`/api/price?type=STARS&amount=${qty}` + (recipient ? `&recipient=${encodeURIComponent(recipient)}` : '')).then(r => {
            const ok = r && r.ok && r.price_usd != null;
            const count = premCount();
            const total = ok ? r.price_usd * count : null;
            if (premType === 'stars') document.getElementById('prem-total').innerText = total != null ? money(total) : '—';
            document.getElementById('star-unit').innerText = ok ? money(r.price_usd / qty) : '—';
            if (premType === 'stars') document.getElementById('prem-cta-sum').innerText = total != null ? money(total) : '';
        }).catch(()=>{});
    }, 350);
}

function renderServiceParams(sid) { const fs = SVC_FIELDS[sid]; return fs ? fs.map(fieldHtml).join('') : ''; }
function collectServiceParams(sid) {
    const fs = SVC_FIELDS[sid] || [];
    const p = {};
    for (const f of fs) {
        if (f.type === 'interests') {
            const box = document.getElementById('param-'+f.key);
            p[f.key] = box ? [...box.querySelectorAll('.int-btn.interest-on')].map(b=>b.dataset.ikey) : [];
            continue;
        }
        if (f.type === 'textauto') {
            if (autoState[f.key]) { p[f.key] = ''; continue; }
            const el = document.getElementById('param-'+f.key);
            const val = (el && el.value || '').trim();
            if (!val) { tg.showAlert('Заполните: ' + f.label); return null; }
            p[f.key] = val; continue;
        }
        const el = document.getElementById('param-'+f.key);
        if (!el) continue;
        const val = (el.value || '').trim();
        if (f.key === 'ad_message' && (val.length < 5 || val.length > 160)) {
            tg.showAlert(`Текст объявления: от 5 до 160 символов (сейчас ${val.length})`); return null;
        }
        if (f.type !== 'select' && !val) { tg.showAlert('Заполните: ' + f.label); return null; }
        p[f.key] = el.value;
    }
    return p;
}

async function submitOrder(btn, payload) {
    orderBusy = true; setCtaLoading(btn, true);
    try {
        const res = await api('/api/order', {method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(payload)});
        handleOrderResult(res);
    } finally { orderBusy = false; setCtaLoading(btn, false); }
}
async function placeProductOrder(btn) {
    if (orderBusy || !currentProduct) return;
    let payload;
    if (currentProduct.kind === 'proxy' || currentProduct.kind === 'proxy_bulk') {
        payload = {kind: currentProduct.kind};
    } else {
        const qty = parseInt(document.getElementById('p-qty').value || '1', 10);
        payload = {kind: currentProduct.kind, id: currentProduct.id, quantity: qty};
        if (currentProduct.kind === 'account') payload.format = selectedFormat;
        if (currentProduct.kind === 'service') {
            if (svcBulk) {
                const links = splitList(document.getElementById('p-links').value);
                if (!links.length) { tg.showAlert('Укажите ссылки'); return; }
                payload.links = links;

                const link = document.getElementById('p-link').value.trim();
                if (!link) { tg.showAlert('Укажите ссылку'); return; }
                payload.link = link;
            }
            const params = collectServiceParams(currentProduct.id);
            if (params === null) return;
            if (Object.keys(params).length) payload.params = params;
        }
    }
    await submitOrder(btn, payload);

function flipCoinTo(result) {
    const inner = document.getElementById('fg-coin-inner');
    const edgeVisual = document.getElementById('fg-coin-edge-visual');
    if (edgeVisual) { edgeVisual.classList.add('hidden'); edgeVisual.classList.remove('wobbling', 'bouncing'); }
    if (inner) { inner.classList.remove('edge-flatspin'); inner.style.opacity = '1'; }
    const targetMod = result === 'tails' ? 180 : 0;
    const currentMod = ((coinRotation % 360) + 360) % 360;
    const delta = (targetMod - currentMod + 360) % 360;
    coinRotation += 360 * 4 + delta;
    if (inner) inner.style.transform = 'rotateY(' + coinRotation + 'deg)';
}

// 0 — покачивается на ребре, 1 — подпрыгивает на ребре, 2 — крутится волчком плашмя.
function triggerCoinEdge(variant) {
    const inner = document.getElementById('fg-coin-inner');
    const edgeVisual = document.getElementById('fg-coin-edge-visual');
    if (variant === 2) {
        // "Волчок": сама 3D-монета уходит в быстрое плоское вращение — грани
        // остаются видимыми (не переключаемся на отдельный edge-bar).
        if (inner) {
            inner.classList.remove('edge-flatspin');
            void inner.offsetHeight;
            inner.classList.add('edge-flatspin');
        }
        return;
    }
    if (inner) inner.style.opacity = '0';
    if (!edgeVisual) return;
    const cls = variant === 1 ? 'bouncing' : 'wobbling';
    edgeVisual.classList.remove('hidden', 'wobbling', 'bouncing');
    void edgeVisual.offsetHeight;
    edgeVisual.classList.add(cls);
}
function resetCoinVisual() {
    const inner = document.getElementById('fg-coin-inner');
    const coinEl = document.querySelector('.coin3d');
    const edgeVisual = document.getElementById('fg-coin-edge-visual');
    if (coinEl) coinEl.classList.remove('flying');
    if (edgeVisual) { edgeVisual.classList.add('hidden'); edgeVisual.classList.remove('wobbling', 'bouncing'); }
    if (inner) {
        inner.classList.remove('edge-flatspin');
        inner.style.opacity = '1';
        inner.style.transform = 'rotateY(' + coinRotation + 'deg)';
    }
}

function buildFestivalReel() {
    const sectors = (FESTIVAL_STATE && FESTIVAL_STATE.wheel_sectors) || [];
    const track = document.getElementById('fg-reel');
    if (!track || !sectors.length) return;
    let html = '';
    for (let c = 0; c < REEL_CYCLES; c++) {
        sectors.forEach((s, i) => {
            const isMiss = !s.value;
            if (isMiss) {
                html += `<div class="reel-item"><div class="reel-ticket reel-miss">`
                      + `<div class="reel-medal">${REEL_MISS_ICON}</div>`
                      + `<span class="reel-amount">${t('festival_miss_label')}</span></div></div>`;
            } else {
                const grad = REEL_GRADIENTS[i % REEL_GRADIENTS.length];
                html += `<div class="reel-item"><div class="reel-ticket" style="background:${grad}">`
                      + `<div class="reel-medal">${REEL_GEM_ICON}</div>`
                      + `<span class="reel-amount">${escapeHtml(s.label)}</span></div></div>`;
            }
        });
    }
    track.innerHTML = html;
    reelOffset = 0;
    track.style.transition = 'none';
    track.style.transform = 'translateY(' + reelCenterAdjust() + 'px)';
}
function spinReelTo(logicalIdx) {
    const sectors = (FESTIVAL_STATE && FESTIVAL_STATE.wheel_sectors) || [];
    const n = sectors.length;
    const track = document.getElementById('fg-reel');
    if (!track || !n) return;
    const centerAdjust = reelCenterAdjust();
    const cycleHeight = REEL_ITEM_H * n;
    const currentMod = ((reelOffset % cycleHeight) + cycleHeight) % cycleHeight;
    track.style.transition = 'none';
    track.style.transform = 'translateY(' + (-(currentMod) + centerAdjust) + 'px)';
    void track.offsetHeight; // форсируем reflow, чтобы сброс позиции не анимировался

    // currentMod всегда кратен REEL_ITEM_H, т.е. соответствует физическому
    // индексу текущего центрального билета. Раньше цель считалась как
    // currentMod + 3*cycleHeight + logicalIdx*ITEM_H — из-за этого физический
    // индекс, а не логический, определял итоговый билет, и после первого же
    // спина лента показывала не тот сектор, что реально выпал на сервере
    // (сдвиг накапливался от спина к спину). Считаем целевой физический индекс
    // явно, чтобы (индекс % n) гарантированно совпадал с logicalIdx.
    const currentPhysIdx = currentMod / REEL_ITEM_H;
    const cyclesForward = 3;
    const deltaMod = (((logicalIdx - (currentPhysIdx % n)) % n) + n) % n;
    const targetPhysIdx = currentPhysIdx + cyclesForward * n + deltaMod;
    reelOffset = targetPhysIdx * REEL_ITEM_H;

    requestAnimationFrame(() => {
        track.style.transition = 'transform 3.6s cubic-bezier(.12,.72,.16,1)';
        track.style.transform = 'translateY(' + (-(reelOffset) + centerAdjust) + 'px)';
    });
}

function updateFestivalCoinsDisplay(coins) {
    // fg-balance стоит рядом с отдельной декоративной иконкой в баре — там
    // эмодзи не дублируем, иначе видно два кружка монетки подряд.
    const withEmoji = '🪙 ' + coins;
    ['festival-coins', 'home-festival-coins'].forEach(id => {
        const el = document.getElementById(id);
        if (el) el.textContent = withEmoji;
    });
    const balEl = document.getElementById('fg-balance');
    if (balEl) balEl.textContent = withEmoji;
}

async function loadFestival() {
    let r;
    try {
        r = await api('/api/festival/state');
    } catch (e) {
        console.error('[Festival] state request failed', e);
        const coinsEl = document.getElementById('festival-coins');
        if (coinsEl) coinsEl.textContent = t('festival_error');
        return;
    }
    if (!r || !r.ok) {
        console.error('[Festival] bad state response', r);
        const coinsEl = document.getElementById('festival-coins');
        if (coinsEl) coinsEl.textContent = t('festival_error');
        return;
    }
    FESTIVAL_STATE = r;
    updateFestivalCoinsDisplay(r.coins);
    const costEl = document.getElementById('festival-cost');
    if (costEl) costEl.textContent = r.cost_coins + ' 🪙';
    const costLineEl = document.getElementById('fg-cost-line');
    if (costLineEl) costLineEl.textContent = t('festival_cost_label') + ': ' + r.cost_coins + ' 🪙';
    const noteEl = document.getElementById('festival-rtp-note');
    if (noteEl) noteEl.textContent = t('festival_rtp_note').replace('{min}', r.win_min_usd).replace('{max}', r.win_max_usd);
    buildFestivalReel();
}

function festivalGameMeta(game) {
    if (game === 'rps') return { icon: '✊', title: t('festival_game_rps'), desc: t('festival_rps_rules') };
    if (game === 'wheel') return { icon: '🎡', title: t('festival_game_wheel'), desc: t('festival_wheel_rules') };
    return { icon: '🪙', title: t('festival_game_coin'), desc: t('festival_coin_rules') };
}

function updateFestivalPlayButtonState() {
    const btn = document.getElementById('fg-play-btn');
    const label = document.getElementById('fg-play-btn-label');
    if (!btn) return;
    if (label) label.textContent = currentFestivalGame === 'wheel' ? t('festival_spin_btn') : t('festival_play_btn');
    const needsChoice = currentFestivalGame === 'rps' || currentFestivalGame === 'coinflip';
    btn.disabled = festivalBusy || (needsChoice && !festivalChoice);
}

function setFestivalPlayBusy(busy) {
    // Ширина кнопки фиксирована в CSS (#fg-play-btn), спиннер накладывается
    // абсолютным позиционированием поверх скрытого текста — кнопка не "прыгает".
    const btn = document.getElementById('fg-play-btn');
    if (!btn) return;
    btn.classList.toggle('cta-busy', busy);
    if (busy) {
        btn.disabled = true;
        if (!btn.querySelector('.fg-play-spinner-wrap')) {
            const wrap = document.createElement('span');
            wrap.className = 'fg-play-spinner-wrap';
            const sp = document.createElement('span');
            sp.className = 'spinner-sm';
            wrap.appendChild(sp);
            btn.appendChild(wrap);
        }
    } else {
        const wrap = btn.querySelector('.fg-play-spinner-wrap');
        if (wrap) wrap.remove();
    }
}

function selectFestivalChoice(choice, btn) {
    if (festivalBusy) return;
    festivalChoice = choice;
    const group = currentFestivalGame === 'rps' ? '.rps-btn' : '#fg-coin-controls button';
    document.querySelectorAll(group).forEach(b => b.classList.remove('selected'));
    if (btn) btn.classList.add('selected');
    haptic();
    updateFestivalPlayButtonState();
}

function confirmFestivalPlay() {
    if (festivalBusy) return;
    if (currentFestivalGame === 'wheel') { playFestival(null); return; }
    if (!festivalChoice) return;
    playFestival(festivalChoice);
}

function openFestivalGame(game) {
    if (!FESTIVAL_STATE) { loadFestival(); }
    currentFestivalGame = game;
    festivalChoice = null;
    const meta = festivalGameMeta(game);
    document.getElementById('fg-icon').textContent = meta.icon;
    document.getElementById('fg-title').textContent = meta.title;
    document.getElementById('fg-desc').textContent = meta.desc;
    const costLineEl = document.getElementById('fg-cost-line');
    if (costLineEl && FESTIVAL_STATE) costLineEl.textContent = t('festival_cost_label') + ': ' + FESTIVAL_STATE.cost_coins + ' 🪙';
    document.getElementById('fg-rps-stage').classList.toggle('hidden', game !== 'rps');
    document.getElementById('fg-coin-stage').classList.toggle('hidden', game !== 'coinflip');
    document.getElementById('fg-wheel-stage').classList.toggle('hidden', game !== 'wheel');
    document.getElementById('fg-result').classList.add('hidden');
    document.getElementById('fg-result').classList.remove('reveal');

    if (game === 'rps') {
        document.getElementById('fg-rps-arena').classList.add('hidden');
        document.getElementById('fg-rps-arena').classList.remove('flex');
        document.getElementById('fg-rps-user').textContent = '✊';
        document.getElementById('fg-rps-user').classList.remove('shaking', 'popping', 'bump-l', 'hf-l');
        document.getElementById('fg-rps-bot').textContent = '🤖';
        document.getElementById('fg-rps-bot').classList.remove('shaking', 'popping', 'bump-r', 'hf-r');
        document.getElementById('fg-rps-spark').classList.remove('popping');
        document.querySelectorAll('.rps-btn').forEach(b => { b.disabled = false; b.classList.remove('selected'); });
    }
    if (game === 'coinflip') {
        document.querySelectorAll('#fg-coin-controls button').forEach(b => { b.disabled = false; b.classList.remove('selected'); });
        resetCoinVisual();

    if (game === 'wheel' && FESTIVAL_STATE) {
        buildFestivalReel();
    }

    updateFestivalPlayButtonState();
    document.getElementById('festival-game-page').classList.remove('hidden');
    window.scrollTo(0, 0);
    haptic();
}
function closeFestivalGame() {
    document.getElementById('festival-game-page').classList.add('hidden');

async function playFestival(choice) {
    if (festivalBusy) return;
    if (!FESTIVAL_STATE || !FESTIVAL_STATE.enabled) { tg.showAlert(t('festival_disabled')); return; }
    if (FESTIVAL_STATE.coins < FESTIVAL_STATE.cost_coins) { tg.showAlert(t('festival_insufficient')); return; }
    festivalBusy = true;

    const game = currentFestivalGame;
    setFestivalPlayBusy(true);

    if (game === 'rps') {
        document.querySelectorAll('.rps-btn').forEach(b => b.disabled = true);
        const arena = document.getElementById('fg-rps-arena');
        arena.classList.remove('hidden'); arena.classList.add('flex');
        const userEl = document.getElementById('fg-rps-user'), botEl = document.getElementById('fg-rps-bot');
        userEl.textContent = RPS_EMOJI[choice] || '✊'; botEl.textContent = '🤖';
        userEl.classList.remove('popping'); botEl.classList.remove('popping');
        userEl.classList.add('shaking'); botEl.classList.add('shaking');
    }
    if (game === 'coinflip') {
        document.querySelectorAll('#fg-coin-controls button').forEach(b => b.disabled = true);
        resetCoinVisual();
    }
    document.getElementById('fg-result').classList.add('hidden');
    document.getElementById('fg-result').classList.remove('reveal');

    let coinFlewAway = false;
    let rpsTieVariant = 0, coinEdgeVariant = 0;
    try {
        // Минимальная задержка держит кнопку в состоянии "играем" до старта визуальной
        // анимации — результат никогда не показывается раньше, чем она закончится.
        const [r] = await Promise.all([
            api('/api/festival/play', {
                method: 'POST', headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ game, choice })
            }),
            festSleep(game === 'rps' ? 1350 : 200)
        ]);
        if (!r || !r.ok) { tg.showAlert(t('festival_error')); return; }

        if (game === 'rps') {
            const userEl = document.getElementById('fg-rps-user'), botEl = document.getElementById('fg-rps-bot');
            userEl.classList.remove('shaking'); botEl.classList.remove('shaking');
            const userMove = (r.detail && r.detail.user_move) || choice;
            const botMove = (r.detail && r.detail.bot_move) || 'rock';
            userEl.textContent = RPS_EMOJI[userMove] || '✊';
            botEl.textContent = RPS_EMOJI[botMove] || '✊';
            if (r.refunded) {
                // Ничья — 3 разных прикола вместо одной и той же анимации.
                rpsTieVariant = Math.floor(Math.random() * 3);
                if (rpsTieVariant === 1) {
                    userEl.classList.add('bump-l'); botEl.classList.add('bump-r');
                    const spark = document.getElementById('fg-rps-spark');
                    spark.classList.remove('popping'); void spark.offsetHeight; spark.classList.add('popping');
                    await festSleep(650);
                } else if (rpsTieVariant === 2) {
                    userEl.classList.add('hf-l'); botEl.classList.add('hf-r');
                    const spark = document.getElementById('fg-rps-spark');
                    spark.classList.remove('popping'); void spark.offsetHeight; spark.classList.add('popping');
                    await festSleep(800);
                } else {
                    userEl.classList.add('popping'); botEl.classList.add('popping');
                    await festSleep(500);
                }
            } else {
                userEl.classList.add('popping'); botEl.classList.add('popping');
                await festSleep(500);
            }
        } else if (game === 'coinflip') {
            if (r.refunded) {
                // Ребро — 3 разных прикола вместо одной и той же анимации.
                coinEdgeVariant = Math.floor(Math.random() * 3);
                triggerCoinEdge(coinEdgeVariant);
                await festSleep(coinEdgeVariant === 2 ? 1550 : (coinEdgeVariant === 1 ? 1200 : 1450));
            } else {
                // Проигрыш иногда оформляем как "монетка укатилась со стола" — просто
                // альтернативная анимация неудачи для разнообразия, на шанс не влияет.
                coinFlewAway = !r.won && Math.random() < 0.4;
                if (coinFlewAway) {
                    coinRotation += 360 * 2;
                    const inner = document.getElementById('fg-coin-inner');
                    if (inner) inner.style.transform = 'rotateY(' + coinRotation + 'deg)';
                    const coinEl = document.querySelector('.coin3d');
                    if (coinEl) coinEl.classList.add('flying');
                    await festSleep(1200);
                } else {
                    flipCoinTo((r.detail && r.detail.result) || choice);
                    await festSleep(1750);
                }
            }
        } else if (game === 'wheel') {
            const idx = (r.detail && typeof r.detail.sector_index === 'number') ? r.detail.sector_index : 0;
            spinReelTo(idx);
            await festSleep(3700);
        }

        // Баланс и карточка результата обновляются вместе — только после того как
        // анимация полностью отыграла, а не раньше.
        FESTIVAL_STATE.coins = r.coins;
        updateFestivalCoinsDisplay(r.coins);

        const box = document.getElementById('fg-result');
        box.classList.remove('hidden');
        box.classList.add('reveal');
        if (r.refunded && game === 'coinflip') {
            const edgeIcons = ['🪙⚖️', '🌀', '🎩'];
            box.innerHTML = `<div class="text-3xl mb-2">${edgeIcons[coinEdgeVariant]}</div><div class="font-black text-lg" style="color:var(--muted)">${t('festival_coin_edge_' + (coinEdgeVariant + 1))}</div>`;
            haptic('light');
        } else if (r.refunded && game === 'rps') {
            const tieIcons = ['🤝', '👊', '🙌'];
            box.innerHTML = `<div class="text-3xl mb-2">${tieIcons[rpsTieVariant]}</div><div class="font-black text-lg" style="color:var(--muted)">${t('festival_tie_banner_' + (rpsTieVariant + 1))}</div>`;
            haptic('light');
        } else if (r.won) {
            box.innerHTML = `<div class="text-3xl mb-2">🏆</div><div class="font-black text-lg" style="color:var(--green)">${t('festival_win_banner')}</div><div class="text-2xl font-black mt-1">${money(r.prize_usd)}</div>`;
            haptic('medium');
        } else if (coinFlewAway) {
            box.innerHTML = `<div class="text-3xl mb-2">🪙💨</div><div class="font-black text-lg" style="color:var(--muted)">${t('festival_coin_flew_away')}</div>`;
            haptic('light');
        } else {
            box.innerHTML = `<div class="text-3xl mb-2">😔</div><div class="font-black text-lg" style="color:var(--muted)">${t('festival_lose_banner')}</div>`;
            haptic('light');
        }
    } catch (e) {
        console.error('[Festival] play request failed', e);
        tg.showAlert(t('festival_error'));
    } finally {
        festivalBusy = false;
        setFestivalPlayBusy(false);
        if (game === 'rps') document.querySelectorAll('.rps-btn').forEach(b => b.disabled = false);
        if (game === 'coinflip') document.querySelectorAll('#fg-coin-controls button').forEach(b => b.disabled = false);
        updateFestivalPlayButtonState();
    }
}
